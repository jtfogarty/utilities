1//! Core serialization traits and binary wire format implementations.
2//!
3//! This module defines the [`Serializable`](crate::serialize::Serializable) trait and implements it
4//! for common Rust types, providing a language-agnostic binary protocol for WASM guest-host
5//! communication.
6//!
7//! # Wire Format
8//!
9//! The serialization format is designed to be:
10//! - **Compact**: Minimal overhead for small types
11//! - **Deterministic**: Same input always produces same output
12//! - **Language-agnostic**: Can be implemented in any language
13//! - **Little-endian**: All multi-byte integers use little-endian byte order
14//!
15//! ## Transfer Layout
16//!
17//! When transferred across WASM boundaries, all serialized data is prefixed with a length:
18//! ```text
19//! [4-byte length (u32, LE)][serialized data]
20//! ```
21//!
22//! ## Type Formats
23//!
24//! - **Primitives**: Direct byte encoding (String: UTF-8, numbers: LE bytes, bool: 0/1)
25//! - **Enums**: Tag byte + optional payload (Option, Result, Bound)
26//! - **Collections**: Length-prefixed elements (Vec, tuples)
27//! - **Complex**: FlatBuffers protocol (Value, Kind)
28//!
29//! See individual type implementations for detailed format specifications.
30

46/// A wrapper around serialized binary data.
47///
48/// This type holds the raw bytes of a serialized value and can be transferred
49/// across WASM boundaries using the [`Transfer`] trait.
50///
51/// # Memory Layout
52///
53/// When transferred, `Serialized` data uses this layout:
54/// ```text
55/// [4-byte length (u32, LE)][data bytes]
56/// ```
57pub struct Serialized(pub bytes::Bytes);
58
59// Guest side implementation (sync)
60impl Transfer for Serialized {
61	fn transfer(self, controller: &mut dyn MemoryController) -> Result<Ptr> {
62		let len = 4 + self.0.len();
63		let ptr = controller.alloc(len as u32)?;
64		let mem = controller.mut_mem(ptr, len as u32);
65		mem[0..4].copy_from_slice(&(self.0.len() as u32).to_le_bytes());
66		mem[4..len].copy_from_slice(&self.0);
67		Ok(ptr.into())
68	}
69
70	fn receive(ptr: Ptr, controller: &mut dyn MemoryController) -> Result<Self> {
71		let mem = controller.mut_mem(*ptr, 4);
72		let len = u32::from_le_bytes(mem[0..4].try_into()?);
73		let data = controller.mut_mem(*ptr + 4, len).to_vec();
74		#[allow(clippy::unnecessary_cast)]
75		controller.free(*ptr, 4 + len as u32)?;
76		Ok(Serialized(data.into()))
77	}
78}

80// Host side implementation (async)
81#[cfg(feature = "host")]
82#[async_trait]
83impl AsyncTransfer for Serialized {
84	async fn transfer(self, controller: &mut dyn AsyncMemoryController) -> Result<Ptr> {
85		let len = 4 + self.0.len();
86		let ptr = controller.alloc(len as u32).await?;
87		let mem = controller.mut_mem(ptr, len as u32)?;
88		let len_bytes = (self.0.len() as u32).to_le_bytes();
89		mem[0..4].copy_from_slice(len_bytes.as_slice());
90		mem[4..len].copy_from_slice(&self.0);
91		Ok(ptr.into())
92	}
93
94	async fn receive(ptr: Ptr, controller: &mut dyn AsyncMemoryController) -> Result<Self> {
95		let mem = controller.mut_mem(*ptr, 4)?;
96		let len = u32::from_le_bytes(mem[0..4].try_into()?);
97		let data = controller.mut_mem(*ptr + 4, len)?.to_vec();
98		#[allow(clippy::unnecessary_cast)]
99		controller.free(*ptr, 4 + len as u32).await?;
100		Ok(Serialized(data.into()))
101	}
102}

104// Guest side implementation for Serializable (sync)
105impl<T: Serializable> Transfer for T {
106	fn transfer(self, controller: &mut dyn MemoryController) -> Result<Ptr> {
107		Transfer::transfer(self.serialize()?, controller)
108	}
109
110	fn receive(ptr: Ptr, controller: &mut dyn MemoryController) -> Result<Self> {
111		Self::deserialize(Transfer::receive(ptr, controller)?)
112	}
113}

128/// A trait for types that can be serialized to and deserialized from a binary format.
129///
130/// This trait defines the core serialization protocol used for cross-language communication
131/// in the Surrealism ecosystem. Unlike Serde, this provides complete control over the wire
132/// format, ensuring compatibility across different programming languages.
133///
134/// # Implementation Requirements
135///
136/// Implementers must ensure:
137/// - **Deterministic**: Same input always produces same output
138/// - **Round-trip safe**: `deserialize(serialize(x))` == `x`
139/// - **Little-endian**: All multi-byte integers use little-endian byte order
140/// - **Documented**: Wire format must be clearly documented
141///
142/// # Example
143///
144/// ```rust,ignore
145/// use surrealism_types::serialize::{Serializable, Serialized};
146///
147/// impl Serializable for MyType {
148///     fn serialize(self) -> Result<Serialized> {
149///         // Convert to bytes...
150///         Ok(Serialized(bytes))
151///     }
152///
153///     fn deserialize(serialized: Serialized) -> Result<Self> {
154///         // Convert from bytes...
155///         Ok(my_value)
156///     }
157/// }
158/// ```
159pub trait Serializable: Sized {
160	/// Serialize this value into binary format.
161	///
162	/// # Errors
163	///
164	/// Returns an error if serialization fails (e.g., invalid UTF-8, allocation failure).
165	fn serialize(self) -> Result<Serialized>;
166
167	/// Deserialize a value from binary format.
168	///
169	/// # Errors
170	///
171	/// Returns an error if:
172	/// - The data is malformed or truncated
173	/// - The data doesn't match the expected format
174	/// - Type-specific validation fails
175	fn deserialize(serialized: Serialized) -> Result<Self>;
176}
177
178impl<T: SurrealValue> Serializable for SerializableArg<T> {
179	fn serialize(self) -> Result<Serialized> {
180		self.0.into_value().serialize()
181	}
182
183	fn deserialize(serialized: Serialized) -> Result<Self> {
184		Ok(SerializableArg(T::from_value(surrealdb_types::Value::deserialize(serialized)?)?))
185	}
186}

192/// String serialization.
193///
194/// Wire format: Raw UTF-8 bytes (no null terminator)
195/// ```text
196/// [UTF-8 bytes...]
197/// ```
198impl Serializable for String {
199	fn serialize(self) -> Result<Serialized> {
200		Ok(Serialized(self.as_bytes().to_vec().into()))
201	}
202
203	fn deserialize(serialized: Serialized) -> Result<Self> {
204		String::from_utf8(serialized.0.to_vec())
205			.map_err(|e| anyhow::anyhow!("Invalid UTF-8 string: {}", e))
206	}
207}

209/// f64 (64-bit floating point) serialization.
210///
211/// Wire format: 8 bytes, little-endian IEEE 754
212/// ```text
213/// [8 bytes: f64 LE]
214/// ```
215impl Serializable for f64 {
216	fn serialize(self) -> Result<Serialized> {
217		Ok(Serialized(self.to_le_bytes().to_vec().into()))
218	}
219
220	fn deserialize(serialized: Serialized) -> Result<Self> {
221		if serialized.0.len() != 8 {
222			return Err(anyhow::anyhow!("Expected 8 bytes for f64, got {}", serialized.0.len()));
223		}
224
225		Ok(f64::from_le_bytes(serialized.0[..8].try_into()?))
226	}
227}

330/// [`Result<T, E>`] serialization with tag-based discrimination.
331///
332/// Wire format:
333/// ```text
334/// Ok(T):  [0x00][serialized T]
335/// Err(E): [0x01][serialized E]
336/// ```
337impl<T: Serializable, E: Serializable> Serializable for Result<T, E> {
338	fn serialize(self) -> Result<Serialized> {
339		match self {
340			Ok(value) => {
341				let serialized = value.serialize()?;
342				let mut result = Vec::with_capacity(1 + serialized.0.len());
343				result.push(0); // First byte is 0 for Ok
344				result.extend_from_slice(&serialized.0);
345				Ok(Serialized(result.into()))
346			}
347			Err(error) => {
348				let serialized = error.serialize()?;
349				let mut result = Vec::with_capacity(1 + serialized.0.len());
350				result.push(1); // First byte is 1 for Err
351				result.extend_from_slice(&serialized.0);
352				Ok(Serialized(result.into()))
353			}
354		}
355	}
356
357	fn deserialize(serialized: Serialized) -> Result<Self> {
358		if serialized.0.is_empty() {
359			return Err(anyhow::anyhow!("Empty serialized data"));
360		}
361
362		match serialized.0[0] {
363			0 => {
364				// Ok variant - deserialize the value from remaining bytes
365				let value_bytes = Serialized(serialized.0.slice(1..));
366				let value = T::deserialize(value_bytes)?;
367				Ok(Ok(value))
368			}
369			1 => {
370				// Err variant - extract error string from remaining bytes
371				let error_bytes = &serialized.0[1..];
372				let error = E::deserialize(Serialized(error_bytes.to_vec().into()))?;
373				Ok(Err(error))
374			}
375			_ => Err(anyhow::anyhow!("Invalid Result variant byte")),
376		}
377	}
378}

380/// [`anyhow::Result<T>`] serialization.
381///
382/// Wire format: Same as `Result<T, String>` (error converted to string)
383/// ```text
384/// Ok(T):        [0x00][serialized T]
385/// Err(String):  [0x01][error message as UTF-8]
386/// ```
387impl<T: Serializable> Serializable for anyhow::Result<T> {
388	fn serialize(self) -> Result<Serialized> {
389		self.map_err(|e| e.to_string()).serialize()
390	}
391
392	fn deserialize(serialized: Serialized) -> Result<Self> {
393		Ok(Result::<T, String>::deserialize(serialized)?.map_err(|e| anyhow::anyhow!(e)))
394	}

397/// [`Option<T>`] serialization with tag-based discrimination.
398///
399/// Wire format:
400/// ```text
401/// Some(T): [0x00][serialized T]
402/// None:    [0x01]
403/// ```
404impl<T: Serializable> Serializable for Option<T> {
405	fn serialize(self) -> Result<Serialized> {
406		match self {
407			Some(value) => {
408				let serialized = value.serialize()?;
409				let mut result = Vec::with_capacity(1 + serialized.0.len());
410				result.push(0); // First byte is 0 for Some
411				result.extend_from_slice(&serialized.0);
412				Ok(Serialized(result.into()))
413			}
414			None => Ok(Serialized(vec![1].into())),
415		}
416	}
417
418	fn deserialize(serialized: Serialized) -> Result<Self> {
419		if serialized.0.is_empty() {
420			return Err(anyhow::anyhow!("Empty serialized data"));
421		}
422
423		match serialized.0[0] {
424			0 => {
425				// Some variant - deserialize the value from remaining bytes
426				let value_bytes = Serialized(serialized.0.slice(1..));
427				let value = T::deserialize(value_bytes)?;
428				Ok(Some(value))
429			}
430			1 => {
431				// None variant
432				Ok(None)
433			}
434			_ => Err(anyhow::anyhow!("Invalid Option variant byte")),
435		}
436	}
437}

439/// [`Vec<T>`] serialization with length-prefixed elements.
440///
441/// Wire format:
442/// ```text
443/// [4-byte count (u32 LE)]
444/// [4-byte len1 (u32 LE)][element1 data]
445/// [4-byte len2 (u32 LE)][element2 data]
446/// ...
447/// ```
448impl<T: Serializable> Serializable for Vec<T> {
449	fn serialize(self) -> Result<Serialized> {
450		let mut result = (self.len() as u32).to_le_bytes().to_vec();
451		for value in self {
452			let serialized = value.serialize()?;
453			result.extend_from_slice(&(serialized.0.len() as u32).to_le_bytes());
454			result.extend_from_slice(&serialized.0);
455		}
456		Ok(Serialized(result.into()))
457	}
458
459	fn deserialize(serialized: Serialized) -> Result<Self> {
460		if serialized.0.is_empty() {
461			return Err(anyhow::anyhow!("Empty serialized data"));
462		}
463
464		let len = u32::from_le_bytes(serialized.0[0..4].try_into()?) as usize;
465		let mut result = Vec::with_capacity(len);
466		let mut pos = 4;
467		for _ in 0..len {
468			let value_len = u32::from_le_bytes(serialized.0[pos..pos + 4].try_into()?) as usize;
469			let value_bytes = Serialized(serialized.0.slice(pos + 4..pos + 4 + value_len));
470			let value = T::deserialize(value_bytes)?;
471			result.push(value);
472			pos += 4 + value_len;
473		}
474		Ok(result)
475	}
476}

478/// [`std::ops::Bound<T>`] serialization for range bounds.
479///
480/// Wire format:
481/// ```text
482/// Unbounded:    [0x00]
483/// Included(T):  [0x01][serialized T]
484/// Excluded(T):  [0x02][serialized T]
485/// ```
486impl<T: Serializable> Serializable for Bound<T> {
487	fn serialize(self) -> Result<Serialized> {
488		let bytes = match self {
489			Bound::Unbounded => {
490				vec![0]
491			}
492			Bound::Included(value) => {
493				let serialized = value.serialize()?;
494				let mut result = Vec::with_capacity(1 + serialized.0.len());
495				result.push(1);
496				result.extend_from_slice(&serialized.0);
497				result
498			}
499			Bound::Excluded(value) => {
500				let serialized = value.serialize()?;
501				let mut result = Vec::with_capacity(1 + serialized.0.len());
502				result.push(2);
503				result.extend_from_slice(&serialized.0);
504				result
505			}
506		};
507
508		Ok(Serialized(bytes.into()))
509	}
510
511	fn deserialize(serialized: Serialized) -> Result<Self> {
512		let bytes = serialized.0;
513		match bytes[0] {
514			0 => Ok(Bound::Unbounded),
515			1 => {
516				let value = T::deserialize(Serialized(bytes.slice(1..)))?;
517				Ok(Bound::Included(value))
518			}
519			2 => {
520				let value = T::deserialize(Serialized(bytes.slice(1..)))?;
521				Ok(Bound::Excluded(value))
522			}
523			_ => Err(anyhow::anyhow!("Invalid Bound variant byte")),
524		}
525	}
526}

528/// A serializable range type that can represent any Rust range.
529///
530/// Wire format:
531/// ```text
532/// [4-byte beg_len (u32 LE)]
533/// [4-byte end_len (u32 LE)]
534/// [beg_bound data (beg_len bytes)]
535/// [end_bound data (end_len bytes)]
536/// ```
537///
538/// This allows representing `..`, `start..`, `..end`, `start..end`,
539/// `start..=end`, etc. by encoding the start and end bounds.
540#[derive(Debug)]
541pub struct SerializableRange<T: Serializable> {
542	/// The start bound of the range.
543	pub beg: Bound<T>,
544	/// The end bound of the range.
545	pub end: Bound<T>,
546}
547
548impl<T: Serializable + Clone> SerializableRange<T> {
549	/// Convert any range type into a `SerializableRange`.
550	///
551	/// # Note
552	///
553	/// This function clones the bounds because Rust's `IntoBounds` trait is unstable.
554	/// Once stabilized, this could avoid the clone.
555	pub fn from_range_bounds(range: impl std::ops::RangeBounds<T>) -> Result<Self> {
556		Ok(SerializableRange {
557			beg: range.start_bound().cloned(),
558			end: range.end_bound().cloned(),
559		})
560	}
561}
562
563impl<T: Serializable> Serializable for SerializableRange<T> {
564	fn serialize(self) -> Result<Serialized> {
565		let beg = self.beg.serialize()?;
566		let end = self.end.serialize()?;
567		let mut result = Vec::with_capacity(8 + beg.0.len() + end.0.len());
568		result.extend_from_slice(&(beg.0.len() as u32).to_le_bytes());
569		result.extend_from_slice(&(end.0.len() as u32).to_le_bytes());
570		result.extend_from_slice(&beg.0);
571		result.extend_from_slice(&end.0);
572		Ok(Serialized(result.into()))
573	}
574
575	fn deserialize(serialized: Serialized) -> Result<Self> {
576		let bytes = serialized.0;
577		let beg_len = u32::from_le_bytes(bytes[0..4].try_into()?) as usize;
578		let end_len = u32::from_le_bytes(bytes[4..8].try_into()?) as usize;
579		let beg = Bound::deserialize(Serialized(bytes.slice(8..8 + beg_len)))?;
580		let end = Bound::deserialize(Serialized(bytes.slice(8 + beg_len..8 + beg_len + end_len)))?;
581		let range = SerializableRange {
582			beg,
583			end,
584		};
585		Ok(range)
586	}
587}

589impl<T: Serializable> std::ops::RangeBounds<T> for SerializableRange<T> {

591		self.beg.as_ref()
592	}

594	fn end_bound(&self) -> Bound<&T> {

599/// Tuple serialization with length-prefixed elements.
600///
601/// Wire format for tuple `(A, B, ...)`:
602/// ```text
603/// [4-byte lenA (u32 LE)][element A data]
604/// [4-byte lenB (u32 LE)][element B data]
605/// ...
606/// ```
607///
608/// Tuples from 1 to 10 elements are supported.
609macro_rules! impl_tuple {
610    ($(($($name:ident),+)),+ $(,)?) => {
611        $(impl<$($name: Serializable),+> Serializable for ($($name,)+) {
612            fn serialize(self) -> Result<Serialized> {
613                #[allow(non_snake_case)]
614                let ($($name,)+) = self;
615                $(#[allow(non_snake_case)] let $name = $name.serialize()?;)+
616                let mut result = Vec::with_capacity(0 $(+ $name.0.len())+);
617                $(result.extend_from_slice(&($name.0.len() as u32).to_le_bytes());
618                result.extend_from_slice(&$name.0);)+
619                Ok(Serialized(result.into()))
620            }
621
622            fn deserialize(serialized: Serialized) -> Result<Self> {
623                let mut pos = 0;
624                $(
625                    let len = u32::from_le_bytes(serialized.0[pos..pos + 4].try_into()?) as usize;
626                    #[allow(non_snake_case)]
627                    let $name = $name::deserialize(Serialized(serialized.0.slice(pos + 4..pos + 4 + len)))?;
628                    pos += 4 + len;
629                )+
630                let _ = pos;
631                Ok(($($name,)+))
632            }
633        })+
634    }
635}
