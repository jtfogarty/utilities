18/// Custom fake [`Result`] generator.
19///
20/// # Examples
21///
22/// ```
23/// use fake::{Fake, ResultFaker};
24/// use fake::faker::name::en::Name;
25///
26/// // generate name on success but some error code on failure
27/// let f = ResultFaker::ok(Name());
28/// for _ in 0..2 {
29///     let a = f.fake::<Result<String, u8>>();
30/// }
31/// let f = ResultFaker::with(3.., 1..10);
32/// for _ in 0..5 {
33///     let a = f.fake::<Result<u32, usize>>();
34/// }
35/// ```
36pub struct ResultFaker<T, E, R> {
37    ok: T,
38    err: E,
39    err_rate: R,
40}
