8/* ABC without I, O and U

12The letter U where also not used because it could be confused with V letter.

15const LICENSE_CHARS: [char; 23] = [

20#[inline]
21fn numerify_licence_plate<R: Rng + ?Sized>(string: &str, rng: &mut R) -> String {
22    string
23        .chars()
24        .map(|x| match x {
25            '$' => *LICENSE_CHARS.choose(rng).unwrap(),
26            '#' => char::from_digit((0..10).fake_with_rng::<u32, _>(rng), 10).unwrap(),
27            other => other,
28        })
29        .collect()
30}
