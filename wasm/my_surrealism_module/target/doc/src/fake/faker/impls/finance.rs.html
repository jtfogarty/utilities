80fn split_number_to_digits(x: u32) -> Vec<u32> {
81    x.to_string()
82        .chars()

87impl<L: Data> Dummy<Isin<L>> for String {
88    fn dummy_with_rng<R: Rng + ?Sized>(_: &Isin<L>, rng: &mut R) -> Self {
89        let country_code = *ISO3166.choose(rng).unwrap();
90        let nsin = (1..10)
91            .map(|_x| *ALPHANUMERIC.choose(rng).unwrap())
92            .collect::<String>();
93        // Checksum calculation according to Luhn algorithm
94        let char_to_num: Vec<u32> = country_code
95            .chars()
96            .chain(nsin.chars())
97            .map(|x| {
98                if x.is_ascii_digit() {
99                    x.to_digit(10).unwrap()
100                } else {
101                    10 + ((x as u32) - ('A' as u32))
102                }
103            })
104            .flat_map(split_number_to_digits)
105            .collect();
106        let checksum_even = char_to_num.iter().rev().skip(1).step_by(2).sum::<u32>();
107        let checksum_odd = char_to_num
108            .iter()
109            .rev()
110            .step_by(2)
111            .map(|&x| x * 2)
112            .flat_map(split_number_to_digits)
113            .sum::<u32>();
114        let checksum_digit = (10 - ((checksum_even + checksum_odd) % 10)) % 10;
115        format!("{}{}{}", country_code, nsin, checksum_digit)
116    }
