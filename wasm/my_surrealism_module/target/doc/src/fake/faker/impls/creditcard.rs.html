1use crate::faker::creditcard::raw::CreditCardNumber;
2use crate::locales::Data;
3use crate::Dummy;
4use rand::{prelude::IteratorRandom, seq::SliceRandom, Rng};
5
6type PrefixCreditcard<'a> = (u8, Option<&'a [u8]>, &'a [usize]);
7
8static PREFIX_LENGTHS: &[PrefixCreditcard] = &[
9    (b'3', Some(b"47"), &[13]),    // American Express
10    (b'4', None, &[12, 15]),       // Visa
11    (b'5', Some(b"12345"), &[14]), // MasterCard
12];
13
14impl<L: Data> Dummy<CreditCardNumber<L>> for String {
15    fn dummy_with_rng<R: Rng + ?Sized>(_: &CreditCardNumber<L>, rng: &mut R) -> String {
16        let (prefix, opt_prefix, lens): PrefixCreditcard = *PREFIX_LENGTHS.choose(rng).unwrap();
17        let len = *lens.choose(rng).unwrap();
18        let mut bytes = vec![prefix];
19        if let Some(opts) = opt_prefix {
20            bytes.push(*opts.choose(rng).unwrap());
21        }
22        bytes.extend((1..len).map(|_| (b'0'..(b'9' + 1)).choose(rng).unwrap()));
23        let checksum = bytes
24            .iter()
25            .enumerate()
26            .map(|(i, b)| {
27                let v = (*b - b'0') as u32;
28                // for each even-indexed digit, take the cross sum of the double
29                if (i & 1) == 0 {
30                    if v > 4 {
31                        v * 2 - 9
32                    } else {
33                        v * 2
34                    }
35                } else {
36                    // for the odd-indexed just add the value
37                    v
38                }
39            })
40            .sum::<u32>();
41        bytes.push((((100000 - checksum) % 10) as u8) + b'0');
42        // this is safe because the prefix and all digits we add are valid UTF-8
43        unsafe { String::from_utf8_unchecked(bytes) }
44    }
45}
