8pub trait ContourExtension {
9    fn unsafe_area(&self) -> i64;
10    fn is_convex(&self) -> bool;
11    fn is_clockwise_ordered(&self) -> bool;
12    fn contains(&self, point: IntPoint) -> bool;
13    fn to_reversed(&self) -> IntContour;
14}
15
16impl ContourExtension for [IntPoint] {
17    /// The area of the `Path`.
18    /// - Returns: A positive double area if path is clockwise and negative double area otherwise.
19    fn unsafe_area(&self) -> i64 {
20        let n = self.len();
21        let mut p0 = self[n - 1];
22        let mut area: i64 = 0;
23
24        for &p1 in self.iter() {
25            let a = (p1.x as i64).wrapping_mul(p0.y as i64);
26            let b = (p1.y as i64).wrapping_mul(p0.x as i64);
27            area = area.wrapping_add(a).wrapping_sub(b);
28            p0 = p1;
29        }
30
31        area
32    }
33
34    /// Determines if the `Path` is convex.
35    ///
36    /// A convex polygon is a simple polygon (not self-intersecting) in which
37    /// the line segment between any two points along the boundary never
38    /// goes outside the polygon. This method assumes that the points in `Path`
39    /// are ordered (either clockwise or counter-clockwise) and the path is not
40    /// self-intersecting.
41    ///
42    /// - Returns: A Boolean value indicating whether the path is convex.
43    ///   - Returns `true` if the path is convex.
44    ///   - Returns `false` otherwise.
45    fn is_convex(&self) -> bool {
46        let n = self.len();
47        if n <= 2 {
48            return true;
49        }
50
51        let p0 = self[n - 2];
52        let mut p1 = self[n - 1];
53        let mut e0 = p1.subtract(p0);
54
55        let mut sign: i64 = 0;
56        for &p2 in self.iter() {
57            let e1 = p2.subtract(p1);
58            let cross = e1.cross_product(e0).signum();
59            if cross == 0 {
60                let dot = e1.dot_product(e0);
61                if dot == -1 {
62                    return false;
63                }
64            } else if sign == 0 {
65                sign = cross
66            } else if sign != cross {
67                return false;
68            }
69
70            e0 = e1;
71            p1 = p2;
72        }
73
74        true
75    }
76
77    /// The wind direction of the `Path`.
78    /// - Returns: A Boolean value indicating whether the path is clockwise ordered.
79    ///  - Returns `true` if the path is clockwise ordered.
80    ///  - Returns `false` otherwise.
81    #[inline(always)]
82    fn is_clockwise_ordered(&self) -> bool {
83        self.unsafe_area() >= 0
84    }
85
86    /// Checks if a point is contained within the `Path`.
87    /// - Parameter p: The `IntPoint` point to check.
88    /// - Returns: A boolean value indicating whether the point is within the path.
89    fn contains(&self, point: IntPoint) -> bool {
90        let n = self.len();
91        let mut is_contain = false;
92        let mut b = self[n - 1];
93        for &a in self.iter() {
94            let is_in_range = (a.y > point.y) != (b.y > point.y);
95            if is_in_range {
96                let dx = b.x - a.x;
97                let dy = b.y - a.y;
98                let sx = (point.y - a.y) * dx / dy + a.x;
99                if point.x < sx {
100                    is_contain = !is_contain;
101                }
102            }
103            b = a;
104        }
105
106        is_contain
107    }
108
109    #[inline]
110    fn to_reversed(&self) -> IntContour {
111        let mut contour = self.to_vec();
112        contour.reverse();
113        contour
114    }
115}
