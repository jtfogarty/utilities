7/// Represents the parsed kind syntax
8#[derive(Debug, Clone)]
9pub enum KindExpr {
10	/// Simple identifier like `string`, `bool`
11	Ident(Ident),
12	/// Parameterized kind like `record<user>`, `array<string, 10>`
13	Parameterized {
14		name: Ident,
15		params: Vec<KindParam>,
16	},
17	/// Union type like `string | bool`
18	Union(Vec<KindExpr>),
19	/// Literal value like `true`, `false`, `42`, `"hello"`
20	Literal(syn::Lit),
21	/// Kind:: prefix like `Kind::String`
22	KindPrefix(syn::Path),
23	/// Literal:: prefix like `Literal::Bool(true)`
24	LiteralPrefix(syn::Expr),
25	/// Parenthesized expression escape hatch like `(my_expr)`
26	Parenthesized(syn::Expr),
27	/// Object literal like `{ status: "OK", time: string }`
28	Object(Vec<(String, KindExpr)>),
29	/// Array literal like `[string, int, bool]`
30	Array(Vec<KindExpr>),
31}
32
33/// Parameters inside angle brackets
34#[derive(Debug, Clone)]
35#[allow(clippy::large_enum_variant)]
36pub enum KindParam {
37	/// An identifier parameter
38	Ident(Ident),
39	/// A literal parameter (numbers, strings)
40	Literal(syn::Lit),
41	/// A nested kind expression
42	Kind(KindExpr),
43}
44
45impl Parse for KindExpr {
46	/// Parse a `KindExpr` from the input token stream.
47	///
48	/// This is the main parsing entry point that handles union types (`|` operator)
49	/// and delegates to `parse_primary` for individual kind expressions.
50	fn parse(input: syn::parse::ParseStream) -> syn::Result<Self> {
51		let mut expr = Self::parse_primary(input)?;
52
53		// Check for union operator |
54		while input.peek(Token![|]) {
55			input.parse::<Token![|]>()?;
56			let right = Self::parse_primary(input)?;
57
58			expr = match expr {
59				KindExpr::Union(mut variants) => {
60					variants.push(right);
61					KindExpr::Union(variants)
62				}
63				_ => KindExpr::Union(vec![expr, right]),
64			};
65		}
66
67		Ok(expr)
68	}
69}

71impl KindExpr {
72	/// Parse a primary (non-union) kind expression.
73	///
74	/// Handles all non-union syntax including:
75	/// - Array literals: `[string, int]`
76	/// - Object literals: `{ key: value }`
77	/// - Parenthesized expressions: `(expr)`
78	/// - Literal values: `true`, `42`, `"hello"`
79	/// - Identifiers with optional parameters: `string`, `array<string>`, `record<user | post>`
80	/// - Path prefixes: `Kind::String`, `Literal::Bool(true)`
81	fn parse_primary(input: syn::parse::ParseStream) -> syn::Result<Self> {
82		// Check for array literals with square brackets
83		if input.peek(syn::token::Bracket) {
84			return Self::parse_array(input);
85		}
86
87		// Check for object literals with curly braces
88		if input.peek(syn::token::Brace) {
89			return Self::parse_object(input);
90		}
91
92		// Check for parenthesized expressions first
93		if input.peek(syn::token::Paren) {
94			let content;
95			syn::parenthesized!(content in input);
96			let expr: syn::Expr = content.parse()?;
97			return Ok(KindExpr::Parenthesized(expr));
98		}
99
100		// Check for literals
101		if input.peek(syn::LitBool)
102			|| input.peek(syn::LitInt)
103			|| input.peek(syn::LitFloat)
104			|| input.peek(syn::LitStr)
105		{
106			let lit: syn::Lit = input.parse()?;
107			return Ok(KindExpr::Literal(lit));
108		}
109
110		// Parse identifier or path
111		let name: Ident = input.parse()?;
112		let name_str = name.to_string();
113
114		// Special handling for boolean literals as identifiers
115		if name_str == "true" {
116			return Ok(KindExpr::Literal(syn::Lit::Bool(syn::LitBool::new(true, name.span()))));
117		} else if name_str == "false" {
118			return Ok(KindExpr::Literal(syn::Lit::Bool(syn::LitBool::new(false, name.span()))));
119		}
120
121		// Check for :: (path separator)
122		if input.peek(Token![::]) {
123			input.parse::<Token![::]>()?;
124
125			if name_str == "Kind" {
126				// Parse Kind::Something - just store the remaining path after ::
127				let path: syn::Path = input.parse()?;
128				return Ok(KindExpr::KindPrefix(path));
129			} else if name_str == "Literal" {
130				// Parse Literal::Something
131				let expr: syn::Expr = input.parse()?;
132				return Ok(KindExpr::LiteralPrefix(expr));
133			} else {
134				return Err(input.error("Only Kind:: and Literal:: prefixes are supported"));
135			}
136		}
137
138		// Check for angle brackets
139		if input.peek(Token![<]) {
140			input.parse::<Token![<]>()?;
141			let params = Self::parse_params(input)?;
142			input.parse::<Token![>]>()?;
143
144			Ok(KindExpr::Parameterized {
145				name,
146				params,
147			})
148		} else {
149			Ok(KindExpr::Ident(name))
150		}
151	}
152
153	/// Parse an array literal like `[string, int, bool]`.
154	///
155	/// Array literals represent a fixed sequence of kind types and are converted to
156	/// `Kind::Literal(KindLiteral::Array(...))`.
157	fn parse_array(input: syn::parse::ParseStream) -> syn::Result<Self> {
158		let content;
159		syn::bracketed!(content in input);
160
161		let mut elements = Vec::new();
162
163		while !content.is_empty() {
164			// Parse element expression
165			let element_expr = Self::parse(&content)?;
166			elements.push(element_expr);
167
168			// Optional trailing comma
169			if content.peek(Token![,]) {
170				content.parse::<Token![,]>()?;
171			}
172		}
173
174		Ok(KindExpr::Array(elements))
175	}
176
177	/// Parse an object literal like `{ status: string, time: datetime }`.
178	///
179	/// Object literals support both identifier keys (`status`) and string literal keys
180	/// (`"user-id"`) for keys with special characters. Values can be any kind expression.
181	/// The result is converted to `Kind::Literal(KindLiteral::Object(...))`.
182	fn parse_object(input: syn::parse::ParseStream) -> syn::Result<Self> {
183		let content;
184		syn::braced!(content in input);
185
186		let mut fields = Vec::new();
187
188		while !content.is_empty() {
189			// Parse key - can be identifier or string literal
190			let key_str = if content.peek(syn::LitStr) {
191				// String literal key like "user-id"
192				let lit: syn::LitStr = content.parse()?;
193				lit.value()
194			} else if content.peek(Ident) {
195				// Identifier key like status
196				let ident: Ident = content.parse()?;
197				ident.to_string()
198			} else {
199				return Err(content.error("Expected field name (identifier or string literal)"));
200			};
201
202			// Parse colon
203			content.parse::<Token![:]>()?;
204
205			// Parse value expression
206			let value_expr = Self::parse(&content)?;
207
208			fields.push((key_str, value_expr));
209
210			// Optional trailing comma
211			if content.peek(Token![,]) {
212				content.parse::<Token![,]>()?;
213			}
214		}
215
216		Ok(KindExpr::Object(fields))
217	}
218
219	/// Parse parameters inside angle brackets like `<string, 10>` or `<user | post>`.
220	///
221	/// Parameters can be:
222	/// - Identifiers: `string`, `user`
223	/// - Literals: `10`, `"value"`
224	/// - Parenthesized expressions: `(T::kind_of())`
225	/// - Nested kind expressions: `string | int`
226	///
227	/// Parameters are separated by commas or pipes (for union types).
228	fn parse_params(input: syn::parse::ParseStream) -> syn::Result<Vec<KindParam>> {
229		let mut params = Vec::new();
230
231		while !input.peek(Token![>]) && !input.is_empty() {
232			if input.peek(syn::token::Paren) {
233				// Parse parenthesized expression as a parameter
234				let kind_expr = Self::parse_primary(input)?;
235				params.push(KindParam::Kind(kind_expr));
236			} else if input.peek(syn::Lit) {
237				// Parse literal (numbers, strings)
238				let lit: syn::Lit = input.parse()?;
239				params.push(KindParam::Literal(lit));
240			} else if input.peek(Ident) {
241				// Look ahead to see if this is a nested kind expression
242				let fork = input.fork();
243				let _: Ident = fork.parse()?;
244
245				if fork.peek(Token![<]) || fork.peek(Token![|]) {
246					// This is a nested kind expression
247					let kind_expr = input.parse()?;
248					params.push(KindParam::Kind(kind_expr));
249				} else {
250					// Simple identifier
251					let ident: Ident = input.parse()?;
252					params.push(KindParam::Ident(ident));
253				}
254			} else {
255				return Err(
256					input.error("Expected identifier, literal, or parenthesized expression")
257				);
258			}
259
260			// Handle comma or pipe separator
261			if input.peek(Token![,]) {
262				input.parse::<Token![,]>()?;
263			} else if input.peek(Token![|]) && !input.peek2(Token![>]) {
264				// This | is not the end, continue parsing parameters
265				input.parse::<Token![|]>()?;
266			} else if !input.peek(Token![>]) {
267				break;
268			}
269		}
270
271		Ok(params)
272	}
273}

275/// Main entry point for the `kind!` procedural macro.
276///
277/// This macro provides a convenient DSL for creating `Kind` values with support for:
278/// - Basic types: `string`, `bool`, `int`, `array`, `record`, etc.
279/// - Parameterized types: `array<string>`, `set<int, 5>`, `record<user | post>`
280/// - Union types: `string | bool | int`
281/// - Literals: `true`, `false`, `42`, `"hello"`
282/// - Object literals: `{ status: string, "user-id": int }`
283/// - Array literals: `[string, int, bool]`
284/// - Escape hatches: `(expr)` for arbitrary expressions
285/// - Prefixes: `Kind::String`, `Literal::Bool(true)`
286///
287/// # Examples
288///
289/// ```ignore
290/// use surrealdb_types::kind;
291///
292/// // Basic types
293/// let k = kind!(string);
294/// let k = kind!(array<string>);
295///
296/// // Union types
297/// let k = kind!(string | int | bool);
298///
299/// // Object literals
300/// let k = kind!({ status: string, time: datetime });
301///
302/// // Dynamic kinds with escape hatch
303/// let k = kind!(array<(T::kind_of())>);
304/// ```
305pub fn kind(input: TokenStream) -> TokenStream {
306	let expr = parse_macro_input!(input as KindExpr);
307	let tokens = generate_kind_code(&expr);
308	TokenStream::from(tokens)
309}
310
311/// Generate Rust code for a parsed `KindExpr`.
312///
313/// Converts the parsed kind expression into tokens that construct the appropriate
314/// `surrealdb_types::Kind` variant at compile time.
315fn generate_kind_code(expr: &KindExpr) -> TokenStream2 {
316	match expr {
317		KindExpr::Ident(ident) => generate_simple_kind(ident),
318		KindExpr::Parameterized {
319			name,
320			params,
321		} => generate_parameterized_kind(name, params),
322		KindExpr::Union(variants) => {
323			let variant_codes: Vec<_> = variants.iter().map(generate_kind_code).collect();
324			quote! {
325				surrealdb_types::Kind::Either(vec![#(#variant_codes),*])
326			}
327		}
328		KindExpr::Literal(lit) => generate_literal_kind(lit),
329		KindExpr::KindPrefix(path) => {
330			// Convert Kind::Something to surrealdb_types::Kind::Something
331			quote! { surrealdb_types::Kind::#path }
332		}
333		KindExpr::LiteralPrefix(expr) => {
334			quote! {
335				surrealdb_types::Kind::Literal(surrealdb_types::KindLiteral::#expr)
336			}
337		}
338		KindExpr::Parenthesized(expr) => {
339			quote! { #expr }
340		}
341		KindExpr::Object(fields) => {
342			let field_codes: Vec<_> = fields
343				.iter()
344				.map(|(key, value)| {
345					let value_code = generate_kind_code(value);
346					quote! { (#key.to_string(), #value_code) }
347				})
348				.collect();
349
350			quote! {
351				surrealdb_types::Kind::Literal(
352					surrealdb_types::KindLiteral::Object(
353						std::collections::BTreeMap::from([#(#field_codes),*])
354					)
355				)
356			}
357		}
358		KindExpr::Array(elements) => {
359			let element_codes: Vec<_> = elements.iter().map(generate_kind_code).collect();
360
361			quote! {
362				surrealdb_types::Kind::Literal(
363					surrealdb_types::KindLiteral::Array(vec![#(#element_codes),*])
364				)
365			}
366		}
367	}
368}

370/// Generate code for literal kind values like `true`, `42`, `"hello"`.
371///
372/// Converts parsed literals into `Kind::Literal(KindLiteral::...)` expressions.
373fn generate_literal_kind(lit: &syn::Lit) -> TokenStream2 {
374	match lit {
375		syn::Lit::Bool(lit_bool) => {
376			let value = lit_bool.value;
377			quote! {
378				surrealdb_types::Kind::Literal(surrealdb_types::KindLiteral::Bool(#value))
379			}
380		}
381		syn::Lit::Int(lit_int) => {
382			if let Ok(value) = lit_int.base10_parse::<i64>() {
383				quote! {
384					surrealdb_types::Kind::Literal(surrealdb_types::KindLiteral::Integer(#value))
385				}
386			} else {
387				quote! { compile_error!("Integer literal out of range") }
388			}
389		}
390		syn::Lit::Float(lit_float) => {
391			if let Ok(value) = lit_float.base10_parse::<f64>() {
392				quote! {
393					surrealdb_types::Kind::Literal(surrealdb_types::KindLiteral::Float(#value))
394				}
395			} else {
396				quote! { compile_error!("Float literal out of range") }
397			}
398		}
399		syn::Lit::Str(lit_str) => {
400			let value = lit_str.value();
401			quote! {
402				surrealdb_types::Kind::Literal(surrealdb_types::KindLiteral::String(#value.to_string()))
403			}
404		}
405		_ => quote! { compile_error!("Unsupported literal type") },
406	}
407}

409/// Generate code for simple (non-parameterized) kind identifiers.
410///
411/// Maps simple identifiers like `string`, `bool`, `int` to their corresponding
412/// `Kind` enum variants. For example, `string` becomes `Kind::String`.
413fn generate_simple_kind(ident: &Ident) -> TokenStream2 {
414	let name = ident.to_string();
415	match name.as_str() {
416		"any" => quote! { surrealdb_types::Kind::Any },
417		"none" => quote! { surrealdb_types::Kind::None },
418		"null" => quote! { surrealdb_types::Kind::Null },
419		"bool" => quote! { surrealdb_types::Kind::Bool },
420		"bytes" => quote! { surrealdb_types::Kind::Bytes },
421		"datetime" => quote! { surrealdb_types::Kind::Datetime },
422		"decimal" => quote! { surrealdb_types::Kind::Decimal },
423		"duration" => quote! { surrealdb_types::Kind::Duration },
424		"float" => quote! { surrealdb_types::Kind::Float },
425		"int" => quote! { surrealdb_types::Kind::Int },
426		"number" => quote! { surrealdb_types::Kind::Number },
427		"object" => quote! { surrealdb_types::Kind::Object },
428		"string" => quote! { surrealdb_types::Kind::String },
429		"uuid" => quote! { surrealdb_types::Kind::Uuid },
430		"regex" => quote! { surrealdb_types::Kind::Regex },
431		"range" => quote! { surrealdb_types::Kind::Range },
432		"table" => quote! { surrealdb_types::Kind::Table(vec![]) },
433		"record" => quote! { surrealdb_types::Kind::Record(vec![]) },
434		"geometry" => quote! { surrealdb_types::Kind::Geometry(vec![]) },
435		"set" => quote! { surrealdb_types::Kind::Set(Box::new(surrealdb_types::Kind::Any), None) },
436		"array" => {
437			quote! { surrealdb_types::Kind::Array(Box::new(surrealdb_types::Kind::Any), None) }
438		}
439		"file" => quote! { surrealdb_types::Kind::File(vec![]) },
440		"function" => quote! { surrealdb_types::Kind::Function(None, None) },
441		_ => quote! { compile_error!(concat!("Unknown kind: ", #name)) },
442	}
443}

445/// Generate code for parameterized kinds like `record<user>`, `array<string, 10>`.
446///
447/// Handles kinds that take parameters in angle brackets:
448/// - `record<user | post>` - record tables
449/// - `array<string, 10>` - array with element type and optional size
450/// - `set<int, 5>` - set with element type and optional size
451/// - `geometry<point | polygon>` - geometry types
452/// - `file<bucket1 | bucket2>` - file buckets
453fn generate_parameterized_kind(name: &Ident, params: &[KindParam]) -> TokenStream2 {
454	let name_str = name.to_string();
455
456	match name_str.as_str() {
457		"table" => {
458			let mut tables = Vec::new();
459			for param in params {
460				match param {
461					KindParam::Ident(ident) => {
462						tables.push(ident.to_string());
463					}
464					KindParam::Kind(KindExpr::Union(variants)) => {
465						// Handle union inside table parameters: table<user | post>
466						for variant in variants {
467							if let KindExpr::Ident(ident) = variant {
468								tables.push(ident.to_string());
469							}
470						}
471					}
472					_ => {} // Ignore other parameter types
473				}
474			}
475
476			quote! {
477				surrealdb_types::Kind::Table(vec![#(#tables.into()),*])
478			}
479		}
480		"record" => {
481			let mut tables = Vec::new();
482			for param in params {
483				match param {
484					KindParam::Ident(ident) => {
485						tables.push(ident.to_string());
486					}
487					KindParam::Kind(KindExpr::Union(variants)) => {
488						// Handle union inside record parameters: record<user | post>
489						for variant in variants {
490							if let KindExpr::Ident(ident) = variant {
491								tables.push(ident.to_string());
492							}
493						}
494					}
495					_ => {} // Ignore other parameter types
496				}
497			}
498
499			quote! {
500				surrealdb_types::Kind::Record(vec![#(#tables.into()),*])
501			}
502		}
503		"array" => {
504			if params.is_empty() {
505				quote! { surrealdb_types::Kind::Array(Box::new(surrealdb_types::Kind::Any), None) }
506			} else {
507				let inner_kind = generate_param_code(&params[0]);
508				let size = if params.len() > 1 {
509					if let KindParam::Literal(syn::Lit::Int(int_lit)) = &params[1] {
510						let val = int_lit.base10_parse::<u64>().unwrap_or(0);
511						quote! { Some(#val) }
512					} else {
513						quote! { None }
514					}
515				} else {
516					quote! { None }
517				};
518
519				quote! {
520					surrealdb_types::Kind::Array(Box::new(#inner_kind), #size)
521				}
522			}
523		}
524		"set" => {
525			if params.is_empty() {
526				quote! { surrealdb_types::Kind::Set(Box::new(surrealdb_types::Kind::Any), None) }
527			} else {
528				let inner_kind = generate_param_code(&params[0]);
529				let size = if params.len() > 1 {
530					if let KindParam::Literal(syn::Lit::Int(int_lit)) = &params[1] {
531						let val = int_lit.base10_parse::<u64>().unwrap_or(0);
532						quote! { Some(#val) }
533					} else {
534						quote! { None }
535					}
536				} else {
537					quote! { None }
538				};
539
540				quote! {
541					surrealdb_types::Kind::Set(Box::new(#inner_kind), #size)
542				}
543			}
544		}
545		"geometry" => {
546			let mut geom_kinds = Vec::new();
547			for param in params {
548				match param {
549					KindParam::Ident(ident) => {
550						let geom_name = ident.to_string();
551						match geom_name.as_str() {
552							"point" => {
553								geom_kinds.push(quote! { surrealdb_types::GeometryKind::Point })
554							}
555							"line" => {
556								geom_kinds.push(quote! { surrealdb_types::GeometryKind::Line })
557							}
558							"polygon" => {
559								geom_kinds.push(quote! { surrealdb_types::GeometryKind::Polygon })
560							}
561							"multipoint" => geom_kinds
562								.push(quote! { surrealdb_types::GeometryKind::MultiPoint }),
563							"multiline" => {
564								geom_kinds.push(quote! { surrealdb_types::GeometryKind::MultiLine })
565							}
566							"multipolygon" => geom_kinds
567								.push(quote! { surrealdb_types::GeometryKind::MultiPolygon }),
568							"collection" => geom_kinds
569								.push(quote! { surrealdb_types::GeometryKind::Collection }),
570							_ => {}
571						}
572					}
573					KindParam::Kind(KindExpr::Union(variants)) => {
574						// Handle union inside geometry parameters: geometry<point | polygon>
575						for variant in variants {
576							if let KindExpr::Ident(ident) = variant {
577								let geom_name = ident.to_string();
578								match geom_name.as_str() {
579									"point" => geom_kinds
580										.push(quote! { surrealdb_types::GeometryKind::Point }),
581									"line" => geom_kinds
582										.push(quote! { surrealdb_types::GeometryKind::Line }),
583									"polygon" => geom_kinds
584										.push(quote! { surrealdb_types::GeometryKind::Polygon }),
585									"multipoint" => geom_kinds
586										.push(quote! { surrealdb_types::GeometryKind::MultiPoint }),
587									"multiline" => geom_kinds
588										.push(quote! { surrealdb_types::GeometryKind::MultiLine }),
589									"multipolygon" => geom_kinds.push(
590										quote! { surrealdb_types::GeometryKind::MultiPolygon },
591									),
592									"collection" => geom_kinds
593										.push(quote! { surrealdb_types::GeometryKind::Collection }),
594									_ => {}
595								}
596							}
597						}
598					}
599					_ => {} // Ignore other parameter types
600				}
601			}
602
603			quote! {
604				surrealdb_types::Kind::Geometry(vec![#(#geom_kinds),*])
605			}
606		}
607		"file" => {
608			let mut buckets = Vec::new();
609			for param in params {
610				match param {
611					KindParam::Ident(ident) => {
612						buckets.push(ident.to_string());
613					}
614					KindParam::Kind(KindExpr::Union(variants)) => {
615						// Handle union inside file parameters: file<images | videos>
616						for variant in variants {
617							if let KindExpr::Ident(ident) = variant {
618								buckets.push(ident.to_string());
619							}
620						}
621					}
622					_ => {} // Ignore other parameter types
623				}
624			}
625
626			quote! {
627				surrealdb_types::Kind::File(vec![#(#buckets.to_string()),*])
628			}
629		}
630		_ => quote! { compile_error!(concat!("Unknown parameterized kind: ", #name_str)) },
631	}
632}

634/// Generate code for a single parameter value.
635///
636/// Parameters can be identifiers, literals, or nested kind expressions.
637/// This function delegates to the appropriate generation function based on the parameter type.
638fn generate_param_code(param: &KindParam) -> TokenStream2 {
639	match param {
640		KindParam::Ident(ident) => generate_simple_kind(ident),
641		KindParam::Literal(lit) => quote! { #lit },
642		KindParam::Kind(kind) => generate_kind_code(kind),
643	}
644}
