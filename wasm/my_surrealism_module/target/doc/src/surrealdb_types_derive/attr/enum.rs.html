1use syn::{Attribute, Ident, LitStr};
2
3use crate::SkipContent;
4
5#[derive(Debug, Default)]
6pub struct EnumAttributes {
7	/// Whether the enum is untagged
8	pub untagged: bool,
9	/// Tag field name for internally/adjacently tagged enums
10	pub tag: Option<String>,
11	/// Content field name for adjacently tagged enums
12	pub content: Option<String>,
13	/// Enum-level content skipping. `Always` for `#[surreal(skip_content)]`,
14	/// `If(path)` for `#[surreal(skip_content_if = "predicate")]`.
15	pub skip_content: Option<SkipContent>,
16	/// Whether to transform variant names to uppercase
17	pub casing: Option<Casing>,
18}
19
20impl EnumAttributes {
21	pub fn parse(attrs: &[Attribute]) -> Self {
22		let mut enum_attrs = Self::default();
23
24		for attr in attrs {
25			if attr.path().is_ident("surreal") {
26				attr.parse_nested_meta(|meta| {
27					if meta.path.is_ident("untagged") {
28						enum_attrs.untagged = true;
29					} else if meta.path.is_ident("tag") {
30						if let Ok(value) = meta.value()
31							&& let Ok(lit_str) = value.parse::<LitStr>()
32						{
33							enum_attrs.tag = Some(lit_str.value());
34						}
35					} else if meta.path.is_ident("content") {
36						if let Ok(value) = meta.value()
37							&& let Ok(lit_str) = value.parse::<LitStr>()
38						{
39							enum_attrs.content = Some(lit_str.value());
40						}
41					} else if meta.path.is_ident("skip_content") {
42						if enum_attrs.skip_content.is_some() {
43							panic!(
44								"Cannot use both skip_content and skip_content_if on the same enum"
45							);
46						}
47						enum_attrs.skip_content = Some(SkipContent::Always);
48					} else if meta.path.is_ident("skip_content_if") {
49						if enum_attrs.skip_content.is_some() {
50							panic!(
51								"Cannot use both skip_content and skip_content_if on the same enum"
52							);
53						}
54						if let Ok(value) = meta.value()
55							&& let Ok(lit_str) = value.parse::<LitStr>()
56						{
57							enum_attrs.skip_content = Some(SkipContent::If(
58								syn::parse_str(&lit_str.value())
59									.expect("skip_content_if must be a valid path"),
60							));
61						}
62					} else if meta.path.is_ident("uppercase") {
63						enum_attrs.casing = Some(Casing::Uppercase);
64					} else if meta.path.is_ident("lowercase") {
65						enum_attrs.casing = Some(Casing::Lowercase);
66					}
67					Ok(())
68				})
69				.ok();
70			}
71		}
72
73		enum_attrs
74	}
75
76	pub fn variant_string(&self, variant: &Ident) -> String {
77		match self.casing {
78			Some(Casing::Uppercase) => variant.to_string().to_uppercase(),
79			Some(Casing::Lowercase) => variant.to_string().to_lowercase(),
80			None => variant.to_string(),
81		}
82	}
83}
