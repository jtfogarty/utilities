2use quote::quote;
3use syn::{Attribute, LitStr, Path};

5/// Configuration for the crate path used in generated code.
6///
7/// This allows users to customize which crate path the derive macro uses,
8/// which is useful when re-exporting the types from another crate.
9#[derive(Debug, Clone)]
10pub struct CratePath {
11	path: TokenStream,
12}
13
14impl CratePath {
15	/// Parse the crate path from attributes.
16	///
17	/// Looks for `#[surreal(crate = "path::to::crate")]` attribute.
18	/// If not specified, defaults to `::surrealdb::types` when the
19	/// `sdk-path` feature is enabled, otherwise `::surrealdb_types`.
20	pub fn parse(attrs: &[Attribute]) -> Self {
21		for attr in attrs {
22			if attr.path().is_ident("surreal") {
23				let mut crate_path = None;
24
25				let _ = attr.parse_nested_meta(|meta| {
26					if meta.path.is_ident("crate")
27						&& let Ok(value) = meta.value()
28						&& let Ok(lit_str) = value.parse::<LitStr>()
29						&& let Ok(path) = syn::parse_str::<Path>(&lit_str.value())
30					{
31						crate_path = Some(quote! { #path });
32					}
33					Ok(())
34				});
35
36				if let Some(path) = crate_path {
37					return Self {
38						path,
39					};
40				}
41			}
42		}
43
44		Self::default()
45	}
46
47	/// Get the token stream for Value type
48	pub fn value(&self) -> TokenStream {
49		let base = &self.path;
50		quote! { #base::Value }
51	}
52

109impl Default for CratePath {
110	fn default() -> Self {
111		let path = if cfg!(feature = "sdk-path") {
112			quote! { ::surrealdb::types }
113		} else {
114			quote! { ::surrealdb_types }
115		};
