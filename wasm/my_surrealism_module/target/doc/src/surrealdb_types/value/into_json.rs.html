8impl Value {
9	/// Converts the value into a json representation of the value.
10	/// Returns None if there are non serializable values present in the value.
11	// TODO: Remove the JsonValue intermediate and implement a json formatter for
12	// Value.
13	pub fn into_json_value(self) -> JsonValue {
14		// This function goes through some extra length to manually implement the
15		// encoding into json value. This is done to ensure clarity and stability in
16		// regards to how the value variants are converted.
17
18		match self {
19			// These value types are simple values which
20			// can be used in query responses sent to
21			// the client.
22			Value::None | Value::Null => JsonValue::Null,
23			Value::Bool(boolean) => JsonValue::Bool(boolean),
24			Value::Number(number) => match number {
25				Number::Int(int) => JsonValue::Number(JsonNumber::from(int)),
26				Number::Float(float) => {
27					// This is replicating serde_json::to_string default behavior.
28					// f64 will convert to null if it is either NaN of infinity.
29					if let Some(number) = JsonNumber::from_f64(float) {
30						JsonValue::Number(number)
31					} else {
32						JsonValue::Null
33					}
34				}
35				Number::Decimal(decimal) => JsonValue::String(decimal.to_string()),
36			},
37			Value::String(strand) => JsonValue::String(strand),
38			Value::Duration(duration) => JsonValue::String(duration.to_string()),
39			Value::Datetime(datetime) => {
40				JsonValue::String(datetime.0.to_rfc3339_opts(SecondsFormat::AutoSi, true))
41			}
42			Value::Uuid(uuid) => {
43				// This buffer is the exact size needed to be able to encode the uuid.
44				let mut buffer = [0u8; uuid::fmt::Hyphenated::LENGTH];
45				let string = (*uuid.0.hyphenated().encode_lower(&mut buffer)).to_string();
46				JsonValue::String(string)
47			}
48			Value::Array(array) => JsonValue::Array(
49				array.0.into_iter().map(Value::into_json_value).collect::<Vec<JsonValue>>(),
50			),
51			Value::Set(set) => JsonValue::Array(
52				set.0.into_iter().map(Value::into_json_value).collect::<Vec<JsonValue>>(),
53			),
54			Value::Object(object) => {
55				let mut map = Map::with_capacity(object.len());
56				for (k, v) in object.0 {
57					map.insert(k, v.into_json_value());
58				}
59				JsonValue::Object(map)
60			}
61			Value::Geometry(geo) => geometry_into_json_value(geo),
62			Value::Bytes(bytes) => {
63				JsonValue::Array(bytes.0.into_iter().map(|x| JsonValue::Number(x.into())).collect())
64			}
65			Value::Table(table) => JsonValue::String(table.to_string()),
66			Value::RecordId(thing) => JsonValue::String(thing.to_sql()),
67			// TODO: Maybe remove
68			Value::Regex(regex) => JsonValue::String(regex.to_sql()),
69			Value::File(file) => JsonValue::String(file.to_sql()),
70			// This kind of breaks the behaviour
71			// TODO: look at the serialization here.
72			Value::Range(range) => JsonValue::String(range.to_sql()),
73		}
74	}
75}

77fn geometry_into_json_value(geo: Geometry) -> JsonValue {
78	match geo {
79		Geometry::Point(point) => json!({
80			"type": "Point",
81			"coordinates": point_into_json_value(point)
82		}),
83		Geometry::Line(line_string) => {
84			json!({
85				"type": "LineString",
86				"coordinates": line_into_json_value(line_string)
87			})
88		}
89		Geometry::Polygon(polygon) => {
90			json!({
91				"type": "Polygon",
92				"coordinates": polygon_into_json_value(polygon)
93			})
94		}
95		Geometry::MultiPoint(multi_point) => {
96			json!({
97				"type": "MultiPoint",
98				"coordinates": multi_point.into_iter().map(point_into_json_value).collect::<Vec<_>>(),
99			})
100		}
101		Geometry::MultiLine(multi_line_string) => {
102			json!({
103				"type": "MultiLineString",
104				"coordinates": multi_line_string.into_iter().map(line_into_json_value).collect::<Vec<_>>(),
105			})
106		}
107		Geometry::MultiPolygon(multi_polygon) => {
108			json!({
109				"type": "MultiPolygon",
110				"coordinates": multi_polygon.into_iter().map(polygon_into_json_value).collect::<Vec<_>>(),
111			})
112		}
113		Geometry::Collection(items) => {
114			json!({
115				"type": "GeometryCollection",
116				"geometries": items.into_iter().map(geometry_into_json_value).collect::<Vec<_>>(),
117			})
118		}
119	}
120}

122fn point_into_json_value(point: Point) -> JsonValue {
123	vec![JsonValue::from(point.x()), JsonValue::from(point.y())].into()
124}
125
126fn line_into_json_value(line_string: LineString) -> JsonValue {
127	line_string.points().map(point_into_json_value).collect::<Vec<_>>().into()
128}
129
130fn polygon_into_json_value(polygon: Polygon) -> JsonValue {
131	let mut coords =
132		vec![polygon.exterior().points().map(point_into_json_value).collect::<Vec<_>>()];
133
134	for int in polygon.interiors() {
135		let int = int.points().map(point_into_json_value).collect::<Vec<_>>();
136		coords.push(int);
137	}
138	coords.into()
139}
