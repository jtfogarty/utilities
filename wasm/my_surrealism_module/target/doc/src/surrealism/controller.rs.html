1use anyhow::Result;
2use surrealism_types::controller::MemoryController;

4use crate::memory::{__sr_alloc, __sr_free};

6/// A controller struct that manages memory operations in a WASM environment.
7///
8/// This struct implements the `MemoryController` trait, providing methods for
9/// allocating, freeing, and accessing mutable memory slices. It acts as a bridge
10/// between Rust code and external memory management functions (e.g., `__sr_alloc`
11/// and `__sr_free`), which are likely defined in a WASM host or runtime.
12pub struct Controller {}
13
14impl MemoryController for Controller {
15	/// Allocates a block of memory with the specified length and alignment.
16	///
17	/// This method calls an external allocation function (`__sr_alloc`) to reserve
18	/// memory in the WASM linear memory space.
19	///
20	/// # Parameters
21	/// - `len`: The length of the memory block to allocate (in bytes).
22	/// - `align`: The alignment requirement for the allocated memory (in bytes).
23	///
24	/// # Returns
25	/// A `Result` containing the starting pointer (as `u32`) to the allocated memory
26	/// on success, or an error if allocation fails.
27	///
28	/// # Errors
29	/// Returns an error if the underlying allocation function fails (though in this
30	/// implementation, it assumes success and wraps the result in `Ok`).
31	fn alloc(&mut self, len: u32) -> Result<u32> {
32		let result = __sr_alloc(len);
33		if result == 0 {
34			anyhow::bail!("Memory allocation failed");
35		}
36		Ok(result)
37	}
38
39	/// Frees a previously allocated block of memory.
40	///
41	/// This method calls an external free function (`__sr_free`) to release the memory
42	/// block starting at the given pointer.
43	///
44	/// # Parameters
45	/// - `ptr`: The starting pointer to the memory block to free.
46	/// - `len`: The length of the memory block being freed (in bytes).
47	///
48	/// # Returns
49	/// A `Result` indicating success (`Ok(())`) or failure.
50	///
51	/// # Errors
52	/// In this implementation, it assumes success and always returns `Ok(())`. Potential
53	/// errors from the underlying free function are not propagated.
54	fn free(&mut self, ptr: u32, len: u32) -> Result<()> {
55		let result = __sr_free(ptr, len);
56		if result == 0 {
57			anyhow::bail!("Memory deallocation failed");
58		}
59		Ok(())
60	}
61
62	/// Returns a mutable slice to a region of memory.
63	///
64	/// This method creates a mutable reference to a slice of bytes in memory starting
65	/// at the given pointer and spanning the specified length. It uses unsafe Rust
66	/// to construct the slice from raw parts.
67	///
68	/// # Parameters
69	/// - `ptr`: The starting pointer to the memory region.
70	/// - `len`: The length of the memory region (in bytes).
71	///
72	/// # Returns
73	/// A mutable slice (`&mut [u8]`) representing the memory region.
74	///
75	/// # Safety
76	/// This function is unsafe because it assumes the pointer is valid, properly aligned,
77	/// and that the memory region is mutable and not accessed concurrently. Incorrect
78	/// usage may lead to undefined behavior, such as memory corruption or data races.
79	fn mut_mem(&mut self, ptr: u32, len: u32) -> &mut [u8] {
80		unsafe {
81			let ptr = ptr as usize as *mut u8;
82			std::slice::from_raw_parts_mut(ptr, len as usize)
83		}
84	}
85}
