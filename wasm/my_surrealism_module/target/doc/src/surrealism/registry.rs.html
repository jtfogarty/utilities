11/// Represents a wrapped function in the Surrealism framework.
12///
13/// This struct encapsulates a callable function `F` that accepts arguments of type `A`
14/// and returns a value of type `R`. It is designed for use in environments like WASM,
15/// where functions need to be registered and invoked with type-safe argument handling
16/// and memory transfer mechanisms.
17///
18/// The struct uses `PhantomData` to maintain type information for `A` and `R` without
19/// storing actual instances, ensuring compile-time type safety.
20///
21/// # Type Parameters
22/// - `A`: The argument type, which must implement `Args` for argument processing, `Debug` for
23///   debugging, and be `'static + Send + Sync` for thread-safety and lifetime.
24/// - `R`: The return type, which must implement `Transferrable<CResult<Value>>` for
25///   serialization/deserialization, `Debug` for debugging, and be `'static + Send + Sync`.
26/// - `F`: The function type, a closure or function pointer that takes `A` and returns `R`, also
27///   requiring `'static + Send + Sync`.
28///
29/// # Fields
30/// - `function`: The wrapped callable.
31/// - `_phantom`: Phantom data to hold type information for `A` and `R`.
32pub struct SurrealismFunction<A, R, F>
33where
34	A: 'static + Send + Sync + Args + Debug,
35	R: 'static + Send + Sync + SurrealValue + Debug,
36	F: 'static + Send + Sync + Fn(A) -> Result<R, String>,
37{
38	function: F,
39	_phantom: PhantomData<(A, R)>,
40}
41
42impl<A, R, F> SurrealismFunction<A, R, F>
43where
44	A: 'static + Send + Sync + Args + Debug,
45	R: 'static + Send + Sync + SurrealValue + Debug,
46	F: 'static + Send + Sync + Fn(A) -> Result<R, String>,
47{
48	/// Creates a new `SurrealismFunction` from the given callable.
49	///
50	/// # Parameters
51	/// - `function`: The function or closure to wrap.
52	///
53	/// # Returns
54	/// A new instance of `SurrealismFunction`.
55	pub fn from(function: F) -> Self {
56		Self {
57			function,
58			_phantom: Default::default(),
59		}
60	}
61
62	/// Retrieves the kinds (types) of the function's arguments.
63	///
64	/// This method uses the `Args` trait to determine the SQL kinds for each argument.
65	///
66	/// # Returns
67	/// A vector of `sql::Kind` representing the argument types.
68	pub fn args(&self) -> Vec<surrealdb_types::Kind> {
69		A::kinds()
70	}
71
72	/// Retrieves the kind (type) of the function's return value.
73	///
74	/// This method uses the `KindOf` trait to determine the SQL kind for the return type.
75	///
76	/// # Returns
77	/// The `sql::Kind` of the return value.
78	pub fn returns(&self) -> surrealdb_types::Kind {
79		R::kind_of()
80	}
81
82	/// Invokes the wrapped function with the provided arguments.
83	///
84	/// # Parameters
85	/// - `args`: The arguments to pass to the function.
86	///
87	/// # Returns
88	/// A `Result` containing the function's return value `R` on success, or an error.
89	///
90	/// # Errors
91	/// Propagates any error from the wrapped function if it returns a `Result`.
92	pub fn invoke(&self, args: A) -> Result<Result<R, String>> {
93		Ok((self.function)(args))
94	}
95
96	/// Prepares the argument kinds for raw transfer over memory.
97	///
98	/// This method converts the argument kinds into a transferable array and transfers
99	/// it using the provided memory controller.
100	///
101	/// # Parameters
102	/// - `controller`: A mutable reference to a `MemoryController` for allocation and transfer.
103	///
104	/// # Returns
105	/// A `Result` containing the transferred array of kinds on success, or an error.
106	///
107	/// # Errors
108	/// - If converting kinds to transferable types fails.
109	/// - If transferring the array fails.
110	pub fn args_raw(&self, controller: &mut dyn MemoryController) -> Result<Ptr> {
111		self.args().transfer(controller)
112	}
113
114	/// Prepares the return kind for raw transfer over memory.
115	///
116	/// This method converts the return kind into a transferable type and transfers
117	/// it using the provided memory controller.
118	///
119	/// # Parameters
120	/// - `controller`: A mutable reference to a `MemoryController` for allocation and transfer.
121	///
122	/// # Returns
123	/// A `Result` containing the transferred kind on success, or an error.
124	///
125	/// # Errors
126	/// - If converting the kind to a transferable type fails.
127	/// - If transferring the kind fails.
128	pub fn returns_raw(&self, controller: &mut dyn MemoryController) -> Result<Ptr> {
129		self.returns().transfer(controller)
130	}
131
132	/// Invokes the wrapped function using raw transferred arguments.
133	///
134	/// This method accepts raw transferred arguments, deserializes them into `A`,
135	/// invokes the function, and transfers the result back as a `CResult<Value>`.
136	///
137	/// # Parameters
138	/// - `controller`: A mutable reference to a `MemoryController` for allocation and transfer.
139	/// - `args`: The transferred array of argument values.
140	///
141	/// # Returns
142	/// A `Result` containing the transferred result on success, or an error.
143	///
144	/// # Errors
145	/// - If accepting/deserializing arguments fails.
146	/// - If invoking the function fails.
147	/// - If transferring the result fails.
148	pub fn invoke_raw(&self, controller: &mut dyn MemoryController, args: Ptr) -> Result<Ptr> {
149		let args = A::from_values(Vec::<surrealdb_types::Value>::receive(args, controller)?)
150			.map_err(|e| anyhow::anyhow!("{}", e))?;
151		self.invoke(args)?.map(SerializableArg::from).transfer(controller)
152	}
153}
