25/// An `f64` wrapper implementing `Ord` and `Eq`.
26///
27/// This is only used as part of bulk loading.
28/// All input coordinates are checked with `validate_coordinate` before they are used, hence
29/// `Ord` and `Eq` should always be well-defined.
30#[derive(Debug, PartialEq, PartialOrd, Clone, Copy)]
31struct FloatOrd<S>(S);

33fn bulk_load_distance_fn(center: Point2<f64>, position: Point2<f64>) -> impl Ord {
34    (
35        Reverse(FloatOrd(center.distance_2(position))),
36        FloatOrd(position.x),
37        FloatOrd(position.y),
38    )
39}

53/// Implements a circle-sweep bulk loading algorithm for efficient initialization of Delaunay
54/// triangulations.
55///
56/// The algorithm is motivated by:
57///
58/// A faster circle-sweep Delaunay triangulation algorithm
59/// Ahmad Biniaz, Gholamhossein Dastghaibyfard
60/// Advances in Engineering Software,
61/// Volume 43, Issue 1,
62/// 2012,
63/// <https://doi.org/10.1016/j.advengsoft.2011.09.003>
64///
65/// Or alternatively: <http://cglab.ca/~biniaz/papers/Sweep%20Circle.pdf>
66///
67/// # Overview
68///
69/// The major reason for the algorithm's good performance lies in an efficient lookup structure
70/// for finding *hull edges* at a certain *angle*.
71/// "angle" always refers to the angle of a vertex to a center point which is calculated first.
72/// The lookup structure is implemented by the `Hull` struct. It has a `get` and `insert` method
73/// which can quickly find and update the edges of the hull at a given angle.
74///
75/// The algorithm is roughly compromised of these steps:
76///
77///  1. Calculate the median position of all vertices. We call this position `initial_center`.
78///  2. Sort all vertices along their distance to this center.
79///  3. Build a seed triangulation by inserting vertices (beginning with the closest vertex) into an
80///     empty triangulation. Stop once the triangulation has at least one inner face.
81///  4. Calculate the final center. The final center is some point inside the seed triangulation (e.g.
82///     the average its vertices)
83///  5. Initiate the `Hull` lookup structure with the seed triangulation.
84///  6. Insert all remaining vertices, beginning with the vertex closest to `initial_center`.
85///     This can be done efficiently as the edge "closest" to the new vertex can be identified quickly
86///     with `Hull.get`. After each insertion, the hull is partially patched to be more convex
87///  7. After all vertices have been inserted: The hull is not necessarily convex. Fill any "hole"
88///     in the hull by a process comparable to the graham scan algorithm.
89///
90/// # Some details
91///
92/// "angle" does not refer to an actual angle in radians but rather to an approximation that doesn't
93/// require trigonometry for calculation. See method `pseudo_angle` for more information.
94pub fn bulk_load<V, T>(mut elements: Vec<V>) -> Result<T, InsertionError>
95where
96    V: HasPosition,
97    T: Triangulation<Vertex = V>,
98{
99    if elements.is_empty() {
100        return Ok(T::new());
101    }
102
103    let mut point_sum = Point2::<f64>::new(0.0, 0.0);
104
105    for element in &elements {
106        crate::validate_vertex(element)?;
107        let position = element.position();
108
109        point_sum = point_sum.add(position.to_f64());
110    }
111
112    // Set the initial center to the average of all positions. This should be a good choice for most triangulations.
113    //
114    // The research paper uses a different approach by taking the center of the points' bounding box.
115    // However, this position might be far off the center off mass if the triangulation has just a few outliers.
116    // This could lead to a very uneven angle distribution as nearly all points are might be in a very small angle segment
117    // around the center. This degrades the hull-structure's lookup and insertion performance.
118    // For this reason, taking the average appears to be a safer option as most vertices should be distributed around the
119    // initial center.
120    let initial_center = point_sum.mul(1.0 / (elements.len() as f64));
121
122    let mut result = T::with_capacity(elements.len(), elements.len() * 3, elements.len() * 2);
123
124    // Sort by distance, smallest values last. This allows to pop values depending on their distance.
125    elements.sort_unstable_by_key(|e| bulk_load_distance_fn(initial_center, e.position().to_f64()));
126
127    let mut hull = loop {
128        let Some(next) = elements.pop() else {
129            return Ok(result);
130        };
131
132        result.insert(next)?;
133
134        if result.num_vertices() < 3 {
135            continue;
136        }
137
138        let thee_vertices = [
139            FixedVertexHandle::new(result.num_vertices() - 3),
140            FixedVertexHandle::new(result.num_vertices() - 2),
141            FixedVertexHandle::new(result.num_vertices() - 1),
142        ];
143
144        if let Some(hull) = try_get_hull_center(&result, thee_vertices)
145            .and_then(|center| Hull::from_triangulation(&result, center))
146        {
147            hull_sanity_check(&result, &hull);
148            break hull;
149        }
150    };
151
152    if elements.is_empty() {
153        return Ok(result);
154    }
155
156    let mut buffer = Vec::new();
157    let mut skipped_elements = Vec::new();
158
159    while let Some(next) = elements.pop() {
160        skipped_elements.extend(single_bulk_insertion_step(
161            &mut result,
162            false,
163            &mut hull,
164            VertexToInsert::NewVertex(next),
165            &mut buffer,
166        ))
167    }
168
169    if cfg!(any(fuzzing, test)) {
170        hull_sanity_check(&result, &hull);
171    }
172
173    fix_convexity(&mut result);
174
175    for skipped in skipped_elements {
176        result.insert_with_hint_option_impl(skipped, None);
177    }
178
179    Ok(result)
180}

205/// Efficient bulk loading of a constraint delaunay triangulation, including both vertices and constraint edges.
206/// See [ConstrainedDelaunayTriangulation::bulk_load_cdt] for a related example and documentation.
207///
208/// This function does not panic if any two constraints intersect.
209/// It will instead call `on_conflict_found` on all edges that could not be added as they intersect a previously added constraint.
210pub fn try_bulk_load_cdt<V, DE, UE, F, L>(
211    mut elements: Vec<V>,
212    mut edges: Vec<[usize; 2]>,
213    mut on_conflict_found: impl FnMut([usize; 2]),
214) -> Result<ConstrainedDelaunayTriangulation<V, DE, UE, F, L>, InsertionError>
215where
216    V: HasPosition,
217    DE: Default,
218    UE: Default,
219    F: Default,
220    L: HintGenerator<<V as HasPosition>::Scalar>,
221{
222    if elements.is_empty() {
223        return Ok(ConstrainedDelaunayTriangulation::new());
224    }
225
226    let element_indices = sort_and_dedup(&mut elements, &mut edges)?;
227
228    if elements.len() < 3 {
229        // Special case for fewer than 3 vertices. Starting the hull calculation requires at least 3 vertices.
230        let mut result = ConstrainedDelaunayTriangulation::new();
231        for elem in elements {
232            result.insert(elem)?;
233        }
234
235        for &[from, to] in &edges {
236            result.add_constraint(FixedVertexHandle::new(from), FixedVertexHandle::new(to));
237        }
238
239        return Ok(result);
240    }
241
242    let mut buffer = Vec::new();
243
244    let mut skipped_vertices = HashSet::new();
245    let mut skipped_edges = HashSet::new();
246
247    let fix = FixedVertexHandle::new;
248
249    let mut add_constraints_for_new_vertex =
250        |result: &mut ConstrainedDelaunayTriangulation<V, DE, UE, F>,
251         edges: &mut Vec<[usize; 2]>,
252         skipped_vertices: &HashSet<usize>,
253         index| {
254            while let Some([from, to]) = edges.last().copied() {
255                if skipped_vertices.contains(&from) || skipped_vertices.contains(&to) {
256                    skipped_edges.insert([from, to]);
257                    edges.pop();
258                    continue;
259                }
260
261                // Check if the next edge creates any constraint edge
262                if from == index {
263                    let [from, to] = [from, to].map(fix);
264                    // Insert constraint edge
265                    if result.try_add_constraint(from, to).is_empty() {
266                        on_conflict_found([from.index(), to.index()]);
267                    }
268                    edges.pop();
269                } else {
270                    break;
271                }
272            }
273        };
274
275    // Goes through all element indices, closest to farthest away
276    let mut elements_iter = element_indices.iter().rev().copied();
277
278    let first = fix(elements_iter.next().unwrap());
279    let second = fix(elements_iter.next().unwrap());
280
281    let mut result = ConstrainedDelaunayTriangulation::<V, DE, UE, F, LastUsedVertexHintGenerator>::create_with_fixed_vertices(
282        elements,
283        first,
284        second,
285    );
286
287    let third = result.insert_with_hint_option_impl(
288        VertexToInsert::ExistingVertex(fix(elements_iter.next().unwrap())),
289        Some(first),
290    );
291
292    add_constraints_for_new_vertex(&mut result, &mut edges, &skipped_vertices, first.index());
293    add_constraints_for_new_vertex(&mut result, &mut edges, &skipped_vertices, second.index());
294    add_constraints_for_new_vertex(&mut result, &mut edges, &skipped_vertices, third.index());
295
296    let mut last_3_vertices = [first, second, third];
297    let mut next_handle_to_drop = 0;
298
299    let mut hull = loop {
300        if let Some(hull) = try_get_hull_center(&result, last_3_vertices)
301            .and_then(|center| Hull::from_triangulation(&result, center))
302        {
303            break hull;
304        }
305
306        let Some(index) = elements_iter.next() else {
307            return Ok(result.adjust_hint_generator());
308        };
309
310        let next_handle = result.insert_with_hint_option_impl(
311            VertexToInsert::ExistingVertex(FixedVertexHandle::new(index)),
312            Some(first),
313        );
314
315        add_constraints_for_new_vertex(&mut result, &mut edges, &skipped_vertices, index);
316
317        last_3_vertices[next_handle_to_drop] = next_handle;
318        next_handle_to_drop = (next_handle_to_drop + 1) % 3;
319    };
320
321    let mut require_convexity = true;
322
323    for next_index in elements_iter {
324        // Optimization: Once all edges are inserted, convexity isn't required anymore.
325        require_convexity &= !edges.is_empty();
326
327        if single_bulk_insertion_step(
328            &mut result,
329            require_convexity,
330            &mut hull,
331            VertexToInsert::ExistingVertex(FixedVertexHandle::new(next_index)),
332            &mut buffer,
333        )
334        .is_some()
335        {
336            skipped_vertices.insert(next_index);
337        }
338
339        add_constraints_for_new_vertex(&mut result, &mut edges, &skipped_vertices, next_index);
340    }
341
342    if cfg!(any(fuzzing, test)) {
343        hull_sanity_check(&result, &hull);
344    }
345
346    if !require_convexity {
347        fix_convexity(&mut result);
348    }
349
350    for element in skipped_vertices {
351        result.insert_with_hint_option_impl(
352            VertexToInsert::ExistingVertex(FixedVertexHandle::new(element)),
353            Some(first),
354        );
355    }
356
357    for [from, to] in skipped_edges {
358        let [from, to] = [from, to].map(FixedVertexHandle::new);
359        if result.try_add_constraint(from, to).is_empty() {
360            on_conflict_found([from.index(), to.index()]);
361        }
362    }
363
364    assert_eq!(edges, Vec::<[usize; 2]>::new());
365
366    Ok(result.adjust_hint_generator())
367}

369fn sort_and_dedup<V>(
370    elements: &mut Vec<V>,
371    edges: &mut [[usize; 2]],
372) -> Result<Vec<usize>, InsertionError>
373where
374    V: HasPosition,
375{
376    let mut point_sum = Point2::<f64>::new(0.0, 0.0);
377    for element in &*elements {
378        crate::validate_vertex(element)?;
379        let position = element.position();
380
381        point_sum = point_sum.add(position.to_f64());
382    }
383    let initial_center = point_sum.mul(1.0 / (elements.len() as f64));
384
385    for edge in edges.iter_mut() {
386        let [d1, d2] = edge.map(|vertex| {
387            bulk_load_distance_fn(initial_center, elements[vertex].position().to_f64())
388        });
389        if d1 > d2 {
390            edge.reverse();
391        }
392    }
393
394    edges.sort_unstable_by_key(|[from, _]| {
395        bulk_load_distance_fn(initial_center, elements[*from].position().to_f64())
396    });
397    let mut element_indices = (0..elements.len()).collect::<Vec<_>>();
398    element_indices.sort_unstable_by_key(|index| {
399        bulk_load_distance_fn(initial_center, elements[*index].position().to_f64())
400    });
401    let mut removed_indices = HashSet::new();
402    let mut last_position = None;
403    let mut duplicate_sets = Vec::new();
404    let mut current_duplicates = Vec::new();
405    let mut current_min_duplicate = element_indices[0];
406    element_indices.retain(|&index| {
407        let position = elements[index].position();
408
409        let mut retain = true;
410        if Some(position) == last_position {
411            retain = false; // Duplicate found - only the first index is kept
412
413            // Check if the index is the smallest index of the current duplicate set
414            let new_duplicate = if index < current_min_duplicate {
415                let previous_min = current_min_duplicate;
416                current_min_duplicate = index;
417                previous_min
418            } else {
419                index
420            };
421
422            removed_indices.insert(new_duplicate);
423            current_duplicates.push(new_duplicate);
424        } else {
425            if !current_duplicates.is_empty() {
426                if duplicate_sets.is_empty() {
427                    // Lazy initialize to avoid allocation. Each element is initially mapped to itself.
428                    duplicate_sets = (0..elements.len()).collect::<Vec<_>>();
429                }
430
431                for dupe in current_duplicates.iter() {
432                    duplicate_sets[*dupe] = current_min_duplicate;
433                }
434            }
435
436            // Reset duplicate information
437            current_duplicates.clear();
438            last_position = Some(position);
439            current_min_duplicate = index;
440        };
441
442        if !current_duplicates.is_empty() {
443            if duplicate_sets.is_empty() {
444                duplicate_sets = (0..elements.len()).collect::<Vec<_>>();
445            }
446
447            for dupe in current_duplicates.iter() {
448                duplicate_sets[*dupe] = current_min_duplicate;
449            }
450        }
451        retain
452    });
453
454    if !removed_indices.is_empty() {
455        let mut old_to_new = vec![usize::MAX; elements.len()];
456        let mut index_counter = 0;
457        let mut i = 0;
458        elements.retain(|_| {
459            old_to_new[i] = index_counter;
460            if removed_indices.contains(&i) {
461                old_to_new[i] = old_to_new[duplicate_sets[i]];
462                i += 1;
463                return false;
464            }
465            old_to_new[i] = index_counter;
466
467            i += 1;
468            index_counter += 1;
469            true
470        });
471
472        for elem in &mut element_indices {
473            *elem = old_to_new[*elem];
474        }
475
476        for &mut [ref mut from, ref mut to] in edges {
477            *from = old_to_new[*from];
478            *to = old_to_new[*to];
479        }
480    }
481    Ok(element_indices)
482}

484fn try_get_hull_center<V, T>(
485    result: &T,
486    three_vertices: [FixedVertexHandle; 3],
487) -> Option<Point2<f64>>
488where
489    V: HasPosition,
490    T: Triangulation<Vertex = V>,
491{
492    if result.all_vertices_on_line() {
493        return None;
494    }
495
496    let zero = <V as HasPosition>::Scalar::zero();
497    // Get new center that is usually within the convex hull
498    let mut sum_x = zero;
499    let mut sum_y = zero;
500
501    for handle in three_vertices.iter() {
502        let position = result.vertex(*handle).position();
503        sum_x = sum_x + position.x;
504        sum_y = sum_y + position.y;
505    }
506
507    // Note that we don't re-sort the elements according to their distance to the newest center. This doesn't seem to
508    // be required for the algorithms performance, probably due to the `center` being close to `initial_center`.
509    // As of now, it is unclear how to construct point sets that result in a `center` being farther off
510    // `initial center` and what the impact of this would be.
511    let center = Point2::new(sum_x, sum_y).mul(<V as HasPosition>::Scalar::from(1f32 / 3f32));
512
513    if let crate::PositionInTriangulation::OnFace(_) =
514        result.locate_with_hint(center, three_vertices[0])
515    {
516        return Some(center.to_f64());
517    }
518    None
519}

521/// Returns `handle` if it couldn't be inserted due to precision issues.
522#[inline(never)] // Prevent inlining for better profiling data
523fn single_bulk_insertion_step<TR, T>(
524    result: &mut TR,
525    require_convexity: bool,
526    hull: &mut Hull,
527    handle: VertexToInsert<TR::Vertex>,
528    buffer_for_edge_legalization: &mut Vec<FixedUndirectedEdgeHandle>,
529) -> Option<VertexToInsert<TR::Vertex>>
530where
531    T: HasPosition,
532    TR: Triangulation<Vertex = T>,
533{
534    let next_position = handle.position(result.s());
535
536    let current_angle = pseudo_angle(next_position.to_f64(), hull.center);
537
538    let edge = result.directed_edge(hull.get(current_angle));
539
540    let [from, to] = edge.positions();
541    if next_position == from || next_position == to {
542        return None;
543    }
544
545    if edge.side_query(next_position).is_on_right_side_or_on_line() {
546        // The position is, for some reason, not on the left side of the edge. This can e.g. happen
547        // if the vertices have the same angle. The safest way to include these elements appears to
548        // skip them and insert them individually at the end (albeit that's very slow)
549        return Some(handle);
550    }
551
552    assert!(edge.is_outer_edge());
553
554    let edge = edge.fix();
555
556    let handle = handle.resolve(result.s_mut());
557
558    let new_vertex =
559        dcel_operations::create_new_face_adjacent_to_edge(result.s_mut(), edge, handle);
560
561    let ccw_walk_start = result.directed_edge(edge).prev().rev().fix();
562    let cw_walk_start = result.directed_edge(edge).next().rev().fix();
563
564    // Check if the edge that was just connected requires legalization
565    result.legalize_edge(edge, false);
566
567    // At this stage the new vertex was successfully inserted. However, insertions like this will end
568    // up in a strongly *star shaped* triangulation instead of a nice nearly-convex blob of faces.
569    //
570    // To fix this, the algorithm proceeds by connecting some of the adjacent edges and forming new
571    // faces. A face is only created if all of its inner angles are less than 90 degrees. This
572    // tends to be a good heuristic that doesn't create too many skewed triangles which would need
573    // to be fixed later. Refer to the motivating research paper (see method `bulk_load`) for
574    // more information.
575    //
576    // Before:
577    //
578    // outer face
579    //
580    //       v <--- the new vertex
581    //      /\
582    //     /  \     +---- an edge that should potentially not be adjacent to the outer face
583    //    /    \    v
584    //   x0----x1--------x2
585    //
586    // After:
587    // *if* the angle between v->x1 and x1->x2 is smaller than 90°, the edge x2->v and its new
588    // adjacent face is created.
589    //
590    // This only applies to DTs: For CDTs, regular convexity is needed at all points to prevent
591    // constraint edges to leave the convex hull.
592    let mut current_edge = ccw_walk_start;
593    loop {
594        let handle = result.directed_edge(current_edge);
595        let prev = handle.prev();
596        let handle = handle.fix();
597
598        let [prev_from, prev_to] = prev.positions().map(|p| p.to_f64());
599        // `!point_projection.is_behind_edge` is used to identify if the new face's angle will be less
600        // than 90°
601        let angle_condition = require_convexity
602            || !math::is_behind_edge(next_position.to_f64(), prev_to, prev_from)
603            // Prevents inserting the same angle twice into the hull which can cause issues
604            || current_angle == pseudo_angle(prev_to, hull.center);
605
606        current_edge = prev.fix();
607
608        if angle_condition && prev.side_query(next_position).is_on_left_side() {
609            let new_edge = dcel_operations::create_single_face_between_edge_and_next(
610                result.s_mut(),
611                current_edge,
612            );
613
614            buffer_for_edge_legalization.clear();
615            buffer_for_edge_legalization.push(handle.as_undirected());
616            buffer_for_edge_legalization.push(current_edge.as_undirected());
617            result.legalize_edges_after_removal(buffer_for_edge_legalization, |_| false);
618
619            current_edge = new_edge;
620        } else {
621            break;
622        }
623    }
624
625    let mut current_edge = cw_walk_start;
626    // Same as before: Create faces if they will have inner angles less than 90 degrees. This loop
627    // goes in the other direction (clockwise). Refer to the code above for more comments.
628    loop {
629        let handle = result.directed_edge(current_edge);
630        let next = handle.next();
631        let handle = handle.fix();
632
633        let middle_position = next.from().position().to_f64();
634
635        let angle_condition = require_convexity
636            || !super::math::is_behind_edge(
637                middle_position,
638                next_position.to_f64(),
639                next.to().position().to_f64(),
640            )
641            || current_angle == pseudo_angle(middle_position, hull.center);
642
643        let next_fix = next.fix();
644        let is_on_left_side = next.side_query(next_position).is_on_left_side();
645        if angle_condition && is_on_left_side {
646            let new_edge = dcel_operations::create_single_face_between_edge_and_next(
647                result.s_mut(),
648                current_edge,
649            );
650
651            buffer_for_edge_legalization.clear();
652            buffer_for_edge_legalization.push(handle.as_undirected());
653            buffer_for_edge_legalization.push(next_fix.as_undirected());
654            result.legalize_edges_after_removal(buffer_for_edge_legalization, |_| false);
655
656            current_edge = new_edge;
657        } else {
658            break;
659        }
660    }
661
662    let new_vertex = result.vertex(new_vertex);
663    let outgoing_ch_edge = new_vertex.out_edges().find(|edge| edge.is_outer_edge());
664
665    // Fix the hull
666    if let Some(second_edge) = outgoing_ch_edge {
667        let first_edge = second_edge.prev();
668
669        let first_angle = pseudo_angle(first_edge.from().position().to_f64(), hull.center);
670        let second_angle = pseudo_angle(second_edge.to().position().to_f64(), hull.center);
671
672        hull.insert(
673            first_angle,
674            current_angle,
675            second_angle,
676            first_edge.fix(),
677            second_edge.fix(),
678        );
679
680        if cfg!(any(fuzzing, test)) {
681            hull_sanity_check(result, hull);
682        }
683    }
684
685    None
686}

688/// Makes the outer hull convex. Similar to a graham scan.
689fn fix_convexity<TR>(triangulation: &mut TR)
690where
691    TR: Triangulation,
692{
693    let mut edges_to_validate = Vec::with_capacity(2);
694    let mut convex_edges: Vec<FixedDirectedEdgeHandle> = Vec::with_capacity(64);
695
696    let mut current_fixed = triangulation.outer_face().adjacent_edge().unwrap().fix();
697
698    loop {
699        let current_handle = triangulation.directed_edge(current_fixed);
700        let next_handle = current_handle.next().fix();
701        convex_edges.push(current_fixed);
702        current_fixed = next_handle;
703        while let &[.., edge1_fixed, edge2_fixed] = &*convex_edges {
704            let edge1 = triangulation.directed_edge(edge1_fixed);
705            let edge2 = triangulation.directed_edge(edge2_fixed);
706
707            let target_position = edge2.to().position();
708            // Check if the new edge would violate the convex hull property by turning left
709            // The convex hull must only contain right turns
710            if edge1.side_query(target_position).is_on_left_side() {
711                // Violation detected. It is resolved by inserting a new edge
712                edges_to_validate.push(edge1.fix().as_undirected());
713                edges_to_validate.push(edge2.fix().as_undirected());
714
715                let new_edge = dcel_operations::create_single_face_between_edge_and_next(
716                    triangulation.s_mut(),
717                    edge1_fixed,
718                );
719
720                convex_edges.pop();
721                convex_edges.pop();
722                convex_edges.push(new_edge);
723
724                triangulation.legalize_edges_after_removal(&mut edges_to_validate, |_| false);
725            } else {
726                break;
727            }
728        }
729
730        if Some(&current_fixed) == convex_edges.get(1) {
731            break;
732        }
733    }
734}

736#[derive(Debug, Copy, Clone)]
737struct Segment {
738    from: FloatOrd<f64>,
739    to: FloatOrd<f64>,
740}
741
742impl Segment {
743    fn new(from: FloatOrd<f64>, to: FloatOrd<f64>) -> Self {
744        assert_ne!(from, to);
745        Self { from, to }
746    }
747
748    /// Returns `true` if this segment does not contain the angle 0.0.
749    ///
750    /// Pseudo angles wrap back to 0.0 after a full rotation.
751    fn is_non_wrapping_segment(&self) -> bool {
752        self.from < self.to
753    }
754
755    fn contains_angle(&self, angle: FloatOrd<f64>) -> bool {
756        if self.is_non_wrapping_segment() {
757            self.from <= angle && angle < self.to
758        } else {
759            self.from <= angle || angle < self.to
760        }
761    }
762}

764#[derive(Clone, Copy, Debug)]
765struct Node {
766    /// Pseudo-angle of this hull entry
767    angle: FloatOrd<f64>,
768
769    /// An edge leaving at this hull entry.
770    edge: FixedDirectedEdgeHandle,
771
772    /// Neighbors (indexes into the hull)
773    left: usize,
774    right: usize,
775}

777/// Implements an efficient angle-to-edge lookup for edges of the hull of a triangulation.
778///
779/// Refer to `bulk_load` (in `bulk_load.rs`) for more background on how this structure is being used.
780///
781/// It implements an efficient mapping of (pseudo-)angles to edges. To do so, it stores all inserted
782/// edges in a linked list backed by a vec. Finding an edge belonging to a given angle can always
783/// be done by iterating through this list until the target angle is found.
784/// The entries are stored in a consistent order (either clockwise or counterclockwise)
785///
786/// This naive sequential search is very slow as it needs to traverse half of the list on average.
787/// To speed things up, the space of valid angles (the half open interval [0, 1) )
788/// is partitioned into `n` equally sized buckets.
789/// For each bucket, `Hull` stores a reference to the list entry with the *biggest angle* that
790/// still belongs into that bucket. A sequential search will begin at this bucket and has to traverse
791/// only a few elements before finding the target angle.
792/// Since the number of buckets is re-adjusted depending on the number of hull entries, this mapping
793/// will now be in O(1) for reasonably evenly distributed triangulations.
794#[derive(Debug)]
795pub struct Hull {
796    buckets: Vec<usize>,
797    data: Vec<Node>,
798
799    center: Point2<f64>,
800
801    /// Unused indices in data which might be reclaimed later
802    empty: Vec<usize>,
803}
804
805impl Hull {
806    pub fn from_triangulation<T>(triangulation: &T, center: Point2<f64>) -> Option<Self>
807    where
808        T: Triangulation,
809    {
810        assert!(!triangulation.all_vertices_on_line());
811
812        let hull_size = triangulation.convex_hull_size();
813        let mut data = Vec::with_capacity(hull_size);
814
815        let mut prev_index = hull_size - 1;
816
817        let mut last_segment: Option<Segment> = None;
818        for (current_index, edge) in triangulation.convex_hull().enumerate() {
819            let angle_from = pseudo_angle(edge.from().position().to_f64(), center);
820            let angle_to = pseudo_angle(edge.to().position().to_f64(), center);
821
822            if let Some(segment) = last_segment {
823                if segment.contains_angle(angle_to) {
824                    // In rare cases angle_from will be larger than angle_to due to inaccuracies.
825                    return None;
826                }
827            }
828
829            if angle_from == angle_to || angle_from.0.is_nan() || angle_to.0.is_nan() {
830                // Should only be possible for very degenerate triangulations
831                return None;
832            }
833
834            last_segment = Some(Segment::new(angle_from, angle_to));
835
836            let next_index = (current_index + 1) % hull_size;
837
838            data.push(Node {
839                angle: angle_from,
840                edge: edge.fix(),
841                left: prev_index,
842                right: next_index,
843            });
844            prev_index = current_index;
845        }
846        let mut result = Self {
847            buckets: Vec::new(),
848            center,
849            data,
850            empty: Vec::new(),
851        };
852
853        const INITIAL_NUMBER_OF_BUCKETS: usize = 8;
854        result.initialize_buckets(INITIAL_NUMBER_OF_BUCKETS);
855
856        Some(result)
857    }
858
859    fn initialize_buckets(&mut self, target_size: usize) {
860        self.buckets.clear();
861        self.buckets.reserve(target_size);
862
863        const INVALID: usize = usize::MAX;
864        self.buckets
865            .extend(core::iter::repeat_n(INVALID, target_size));
866
867        let (first_index, current_node) = self
868            .data
869            .iter()
870            .enumerate()
871            .find(|(index, _)| !self.empty.contains(index))
872            .unwrap();
873
874        let mut current_index = first_index;
875        let first_bucket = self.ceiled_bucket(current_node.angle);
876        self.buckets[first_bucket] = current_index;
877
878        loop {
879            let current_node = self.data[current_index];
880            let segment = self.segment(&current_node);
881            let start_bucket = self.ceiled_bucket(segment.from);
882            let end_bucket = self.ceiled_bucket(segment.to);
883
884            self.update_bucket_segment(start_bucket, end_bucket, current_index);
885
886            current_index = current_node.right;
887
888            if current_index == first_index {
889                break;
890            }
891        }
892    }
893
894    /// Updates the hull after the insertion of a vertex.
895    ///
896    /// This method should be called after a vertex `v` has been inserted into the outer face of the
897    /// triangulation under construction.
898    ///
899    /// Such a vertex is guaranteed to have two outgoing edges that are adjacent to the convex hull,
900    /// e.g. `e0 -> v -> e1`
901    ///
902    /// In these scenarios, the parameters should be set as follows:
903    /// * `left_angle`: `pseudo_angle(e0.from())`
904    /// * `middle_angle`: `pseudo_angle(v.position())`
905    /// * `right_angle`: `pseudo_angle(e1.to())`
906    /// * `left_edge`: `e0.fix()`
907    /// * `right_edge`: `e1.fix()`
908    ///
909    /// Note that `left_angle` and `right_angle` must already be present in the hull. Otherwise,
910    /// calling this method will result in an endless loop.
911    fn insert(
912        &mut self,
913        left_angle: FloatOrd<f64>,
914        middle_angle: FloatOrd<f64>,
915        mut right_angle: FloatOrd<f64>,
916        left_edge: FixedDirectedEdgeHandle,
917        mut right_edge: FixedDirectedEdgeHandle,
918    ) {
919        let left_bucket = self.floored_bucket(left_angle);
920
921        let mut left_index = self.buckets[left_bucket];
922
923        loop {
924            let current_node = self.data[left_index];
925            if current_node.angle == left_angle {
926                break;
927            }
928            left_index = current_node.right;
929        }
930
931        let mut right_index = self.data[left_index].right;
932        loop {
933            let current_node = self.data[right_index];
934            if current_node.angle == right_angle {
935                break;
936            }
937
938            if cfg!(any(fuzzing, test)) {
939                assert!(!self.empty.contains(&right_index));
940            }
941
942            // Remove current_node - it is completely overlapped by the new segment
943            self.empty.push(right_index);
944            self.data[current_node.left].right = current_node.right;
945            self.data[current_node.right].left = current_node.left;
946            right_index = current_node.right;
947        }
948
949        let new_index = self.get_next_index();
950
951        if left_angle == middle_angle {
952            self.empty.push(left_index);
953            left_index = self.data[left_index].left;
954        } else {
955            self.data[left_index].edge = left_edge;
956        }
957
958        if right_angle == middle_angle {
959            if left_angle != right_angle {
960                self.empty.push(right_index);
961            }
962            right_edge = self.data[right_index].edge;
963            right_index = self.data[right_index].right;
964
965            right_angle = self.data[right_index].angle;
966        }
967
968        // Stitch the vertex between left_index and right_index
969        self.data[left_index].right = new_index;
970        self.data[right_index].left = new_index;
971
972        let new_node = Node {
973            angle: middle_angle,
974            edge: right_edge,
975            left: left_index,
976            right: right_index,
977        };
978
979        self.push_or_update_node(new_node, new_index);
980
981        // Update bucket entries appropriately
982        let left_bucket = self.ceiled_bucket(left_angle);
983        let middle_bucket = self.ceiled_bucket(middle_angle);
984        let right_bucket = self.ceiled_bucket(right_angle);
985
986        self.update_bucket_segment(left_bucket, middle_bucket, left_index);
987        self.update_bucket_segment(middle_bucket, right_bucket, new_index);
988
989        self.adjust_bucket_size_if_necessary();
990    }
991
992    fn get_next_index(&mut self) -> usize {
993        self.empty.pop().unwrap_or(self.data.len())
994    }
995
996    fn update_bucket_segment(&mut self, left_bucket: usize, right_bucket: usize, new_value: usize) {
997        if left_bucket <= right_bucket {
998            for current_bucket in &mut self.buckets[left_bucket..right_bucket] {
999                *current_bucket = new_value;
1000            }
1001        } else {
1002            // Wrap buckets
1003            for current_bucket in &mut self.buckets[left_bucket..] {
1004                *current_bucket = new_value;
1005            }
1006            for current_bucket in &mut self.buckets[..right_bucket] {
1007                *current_bucket = new_value;
1008            }
1009        }
1010    }
1011
1012    fn push_or_update_node(&mut self, node: Node, index: usize) {
1013        if let Some(existing_node) = self.data.get_mut(index) {
1014            *existing_node = node;
1015        } else {
1016            assert_eq!(self.data.len(), index);
1017            self.data.push(node);
1018        }
1019    }
1020
1021    /// Gets an edge of the hull which covers a given input angle.
1022    ///
1023    /// An edge is considered to cover an input angle if the input angle is contained in the angle
1024    /// segment spanned by `pseudo_angle(edge.from()) .. pseudo_angle(edge.from())`
1025    fn get(&self, angle: FloatOrd<f64>) -> FixedDirectedEdgeHandle {
1026        let mut current_handle = self.buckets[self.floored_bucket(angle)];
1027        loop {
1028            let current_node = self.data[current_handle];
1029            let left_angle = current_node.angle;
1030            let next_angle = self.data[current_node.right].angle;
1031
1032            if Segment::new(left_angle, next_angle).contains_angle(angle) {
1033                return current_node.edge;
1034            }
1035
1036            current_handle = current_node.right;
1037        }
1038    }
1039
1040    /// Looks up what bucket a given pseudo-angle will fall into.
1041    fn floored_bucket(&self, angle: FloatOrd<f64>) -> usize {
1042        ((angle.0 * self.buckets.len() as f64).floor() as usize) % self.buckets.len()
1043    }
1044
1045    fn ceiled_bucket(&self, angle: FloatOrd<f64>) -> usize {
1046        ((angle.0 * self.buckets.len() as f64).ceil() as usize) % self.buckets.len()
1047    }
1048
1049    fn segment(&self, node: &Node) -> Segment {
1050        Segment::new(node.angle, self.data[node.right].angle)
1051    }
1052
1053    fn adjust_bucket_size_if_necessary(&mut self) {
1054        let size = self.data.len() - self.empty.len();
1055        let num_buckets = self.buckets.len();
1056
1057        const MIN_NUMBER_OF_BUCKETS: usize = 16;
1058        if num_buckets * 2 < size {
1059            // Too few buckets - increase bucket count
1060            self.initialize_buckets(num_buckets * 2);
1061        } else if num_buckets > size * 4 && num_buckets > MIN_NUMBER_OF_BUCKETS {
1062            let new_size = num_buckets / 4;
1063            if new_size >= MIN_NUMBER_OF_BUCKETS {
1064                // Too many buckets - shrink
1065                self.initialize_buckets(new_size);
1066            }
1067        }
1068    }
1069}
1070
1071/// Returns a pseudo-angle in the 0-1 range, without expensive trigonometry functions
1072///
1073/// The angle has the following shape:
1074/// ```text
1075///              0.25
1076///               ^ y
1077///               |
1078///               |
1079///   0           |           x
1080///   <-----------o-----------> 0.5
1081///   1           |
1082///               |
1083///               |
1084///               v
1085///              0.75
1086/// ```
1087#[inline]
1088fn pseudo_angle(a: Point2<f64>, center: Point2<f64>) -> FloatOrd<f64> {
1089    let diff = a.sub(center);
1090    let p = diff.x / (diff.x.abs() + diff.y.abs());
1091    FloatOrd(1.0 - (if diff.y > 0.0 { 3.0 - p } else { 1.0 + p }) * 0.25)
1092}

1094fn hull_sanity_check(triangulation: &impl Triangulation, hull: &Hull) {
1095    let non_empty_nodes: Vec<_> = hull
1096        .data
1097        .iter()
1098        .enumerate()
1099        .filter(|(index, _)| !hull.empty.contains(index))
1100        .collect();
1101
1102    for (index, node) in &non_empty_nodes {
1103        let left_node = hull.data[node.left];
1104        let right_node = hull.data[node.right];
1105
1106        let edge = triangulation.directed_edge(node.edge);
1107        assert!(edge.is_outer_edge());
1108
1109        assert!(!hull.empty.contains(&node.left));
1110        assert!(!hull.empty.contains(&node.right));
1111
1112        assert_eq!(left_node.right, *index);
1113        assert_eq!(right_node.left, *index);
1114    }
1115
1116    for (bucket_index, bucket_node) in hull.buckets.iter().enumerate() {
1117        assert!(!hull.empty.contains(bucket_node));
1118        let bucket_start_angle = FloatOrd(bucket_index as f64 / hull.buckets.len() as f64);
1119
1120        for (node_index, node) in &non_empty_nodes {
1121            let segment = hull.segment(node);
1122
1123            if segment.contains_angle(bucket_start_angle) {
1124                // Make sure the bucket refers to the node with the smallest angle in the same bucket
1125                assert_eq!(node_index, bucket_node);
1126            }
1127        }
1128    }
1129}

1483        let data = vec![
1484            Point2::new(-2.7049442424493675e-11f64, -2.7049442424493268e-11),
1485            Point2::new(-2.7049442424493268e-11, -2.704944239760038e-11),

1488            Point2::new(-2.7049442424493268e-11, -2.704944239760038e-11),
1489            Point2::new(-2.7049442424493675e-11, 0.0),
1490        ];
1491
1492        let mut edges = Vec::<[usize; 2]>::new();
1493
1494        for p in &data {
1495            if crate::validate_coordinate(p.x).is_err() || crate::validate_coordinate(p.y).is_err()
1496            {
1497                return Ok(());
1498            }
1499            if p.x.abs() > 20.0 || p.y.abs() > 20.0 {
1500                return Ok(());
1501            }
1502        }
1503
1504        for &[from, to] in &edges {
1505            if from >= data.len() || to >= data.len() || from == to {
1506                return Ok(());
1507            }
1508        }
1509
1510        let mut reference_cdt =
1511            ConstrainedDelaunayTriangulation::<Point2<f64>>::bulk_load(data.clone()).unwrap();
1512
1513        let mut last_index = 0;
1514        for (index, [from, to]) in edges
1515            .iter()
1516            .copied()
1517            .map(|e| e.map(FixedVertexHandle::from_index))
1518            .enumerate()
1519        {
1520            if reference_cdt.can_add_constraint(from, to) {
1521                reference_cdt.add_constraint(from, to);
1522            } else {
1523                last_index = index;
1524                break;
1525            }
1526        }
1527
1528        edges.truncate(last_index);
1529
1530        let bulk_loaded =
1531            ConstrainedDelaunayTriangulation::<Point2<f64>>::bulk_load_cdt(data, edges).unwrap();
1532
1533        bulk_loaded.cdt_sanity_check();
1534
1535        Ok(())
1536    }
