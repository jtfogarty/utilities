26/// Cut off all edges to the left of the given edges and replace their face
27/// with the outer face.
28/// `edge_strip` should be a vec of connected edges ordered ccw.
29pub fn disconnect_edge_strip<V, DE, UE, F>(
30    dcel: &mut Dcel<V, DE, UE, F>,
31    edge_strip: Vec<FixedDirectedEdgeHandle>,
32) -> IsolateVertexResult
33where
34    DE: Default,
35    UE: Default,
36    F: Default,
37{
38    let mut edges_to_remove = Vec::new();
39    let mut faces_to_remove = Vec::new();
40
41    for edge in edge_strip {
42        let edge_handle = dcel.directed_edge(edge);
43        edges_to_remove.push(edge_handle.prev().as_undirected().fix());
44        faces_to_remove.extend(edge_handle.face().as_inner().map(|e| e.fix()));
45
46        let from = edge_handle.from().fix();
47
48        let prev = edge_handle.ccw().prev().fix();
49        dcel.half_edge_mut(prev).next = edge;
50        dcel.half_edge_mut(edge).prev = prev;
51
52        dcel.half_edge_mut(edge).face = OUTER_FACE_HANDLE;
53
54        dcel.faces[OUTER_FACE_HANDLE.index()].adjacent_edge = Some(edge);
55        dcel.vertices[from.index()].out_edge = Some(edge);
56    }
57
58    IsolateVertexResult {
59        new_edges: Vec::new(),
60        smallest_new_edge: FixedUndirectedEdgeHandle::new(0),
61        edges_to_remove,
62        faces_to_remove,
63    }
64}

66/// Removes a vertex from the DCEL.
67/// Then, the resulting hole will be re-triangulated with a triangle fan.
68pub fn isolate_vertex_and_fill_hole<V, DE, UE, F>(
69    dcel: &mut Dcel<V, DE, UE, F>,
70    border_loop: Vec<FixedDirectedEdgeHandle>,
71    vertex_to_remove: FixedVertexHandle,
72) -> IsolateVertexResult
73where
74    DE: Default,
75    UE: Default,
76    F: Default,
77{
78    let vertex = dcel.vertex(vertex_to_remove);
79
80    let mut edges_to_remove = Vec::new();
81    let mut faces_to_remove = Vec::new();
82
83    for edge in vertex.out_edges() {
84        edges_to_remove.push(edge.as_undirected().fix());
85        faces_to_remove.extend(edge.face().fix().as_inner());
86    }
87
88    remesh_edge_ring(dcel, border_loop, edges_to_remove, faces_to_remove)
89}

91pub fn remesh_edge_ring<V, DE, UE, F>(
92    dcel: &mut Dcel<V, DE, UE, F>,
93    mut border_loop: Vec<FixedDirectedEdgeHandle>,
94    edges_to_remove: Vec<FixedUndirectedEdgeHandle>,
95    faces_to_remove: Vec<FixedFaceHandle<InnerTag>>,
96) -> IsolateVertexResult
97where
98    DE: Default,
99    UE: Default,
100    F: Default,
101{
102    let mut new_edges = Vec::new();
103    let smallest_new_edge = FixedUndirectedEdgeHandle::new(dcel.edges.len());
104
105    // Re-mesh border loop by fanning out triangles
106    let mut inner_edge = border_loop.pop().unwrap();
107    let fan_origin = dcel.directed_edge(inner_edge).from().fix();
108    while border_loop.len() > 2 {
109        let outer_edge = border_loop.pop().unwrap();
110        let outer_edge_from = dcel.directed_edge(outer_edge).from().fix();
111        let outer_edge_to = dcel.directed_edge(outer_edge).to().fix();
112
113        let new_edge_handle = FixedDirectedEdgeHandle::new_normalized(dcel.edges.len());
114        let new_face_handle = FixedFaceHandle::new(dcel.faces.len());
115
116        // This edge needs to be created
117        //      |
118        //      |   +
119        //      |  /|
120        //      | / |
121        //      V/  |
122        //      /   |
123        //    t/   o|
124        //    /e    |
125        //   /   i  |
126        // fo-------+
127        //
128        // i = inner_edge
129        // o = outer_edge
130        // fo = fan_origin
131        // e = new_edge_handle
132        // t = new_twin_handle
133        let new_norm = HalfEdgeEntry {
134            next: inner_edge,
135            prev: outer_edge,
136            origin: outer_edge_to,
137            face: new_face_handle,
138        };
139        let new_twin = HalfEdgeEntry {
140            next: FixedDirectedEdgeHandle::max(), // Will be
141            prev: FixedDirectedEdgeHandle::max(), // fixed in
142            face: FixedFaceHandle::max(),         // the next iteration
143            origin: fan_origin,
144        };
145
146        let new_edge = EdgeEntry::new(new_norm, new_twin);
147
148        let new_face = FaceEntry {
149            adjacent_edge: Some(new_edge_handle),
150            data: F::default(),
151        };
152
153        dcel.half_edge_mut(outer_edge).face = new_face_handle;
154        dcel.half_edge_mut(outer_edge).next = new_edge_handle;
155        dcel.half_edge_mut(outer_edge).prev = inner_edge;
156
157        // Fix missing stuff from last iteration
158        dcel.half_edge_mut(inner_edge).prev = new_edge_handle;
159        dcel.half_edge_mut(inner_edge).next = outer_edge;
160        dcel.half_edge_mut(inner_edge).face = new_face_handle;
161
162        dcel.vertices[outer_edge_from.index()].out_edge = Some(outer_edge);
163
164        dcel.edges.push(new_edge);
165        new_edges.push(new_edge_handle.as_undirected());
166        dcel.faces.push(new_face);
167
168        // Prepare next iteration
169        inner_edge = new_edge_handle.rev();
170    }
171
172    // Fix last triangle
173    let inner_edge_next = border_loop.pop().unwrap();
174    let inner_edge_prev = border_loop.pop().unwrap();
175
176    // Create a new face for the last triangle
177    let new_face_handle = FixedFaceHandle::new(dcel.faces.len());
178    let new_face = FaceEntry {
179        adjacent_edge: Some(inner_edge),
180        data: F::default(),
181    };
182    dcel.half_edge_mut(inner_edge).face = new_face_handle;
183    dcel.faces.push(new_face);
184    dcel.half_edge_mut(inner_edge_prev).face = new_face_handle;
185    dcel.half_edge_mut(inner_edge_next).face = new_face_handle;
186
187    // Update edge links from last triangle
188    dcel.half_edge_mut(inner_edge).prev = inner_edge_prev;
189    dcel.half_edge_mut(inner_edge_prev).next = inner_edge;
190
191    dcel.half_edge_mut(inner_edge).next = inner_edge_next;
192    dcel.half_edge_mut(inner_edge_next).prev = inner_edge;
193
194    dcel.half_edge_mut(inner_edge_prev).prev = inner_edge_next;
195    dcel.half_edge_mut(inner_edge_next).next = inner_edge_prev;
196
197    // Update out_edge entries for the triangle's vertices
198    let prev_origin = dcel.half_edge_mut(inner_edge_prev).origin;
199    dcel.vertices[prev_origin.index()].out_edge = Some(inner_edge_prev);
200
201    let next_origin = dcel.half_edge_mut(inner_edge_next).origin;
202    dcel.vertices[next_origin.index()].out_edge = Some(inner_edge_next);
203
204    // fan_origin is the third vertex of the triangle
205    dcel.vertices[fan_origin.index()].out_edge = Some(inner_edge);
206
207    IsolateVertexResult {
208        new_edges,
209        smallest_new_edge,
210        edges_to_remove,
211        faces_to_remove,
212    }
213}

215pub fn create_single_face_between_edge_and_next<V, DE, UE, F>(
216    dcel: &mut Dcel<V, DE, UE, F>,
217    edge: FixedDirectedEdgeHandle,
218) -> FixedDirectedEdgeHandle
219where
220    DE: Default,
221    UE: Default,
222    F: Default,
223{
224    // Before:
225    //
226    //
227    //       ^
228    //         \
229    //  outer   \
230    //  face     \
231    //            \
232    // --- edge -> \
233    //
234    // After:
235    // Added a new edge between edge.next().to() and edge.from()
236    //
237    // new edge (return value)
238    //   |
239    //   |   ^
240    //   | /   \
241    //   v/ new \
242    //   / face  \
243    //  /         \
244    // /-- edge -> \
245
246    let edge_entry = *dcel.half_edge(edge);
247    let next_entry = *dcel.half_edge(edge_entry.next);
248    let next_to = dcel.directed_edge(edge_entry.next).to().fix();
249
250    let new_face_handle = FixedFaceHandle::<PossiblyOuterTag>::new(dcel.num_faces());
251
252    let inner_edge_entry = HalfEdgeEntry {
253        next: edge,
254        prev: edge_entry.next,
255        face: new_face_handle,
256        origin: next_to,
257    };
258
259    let outer_edge_entry = HalfEdgeEntry {
260        next: next_entry.next,
261        prev: edge_entry.prev,
262        face: OUTER_FACE_HANDLE,
263        origin: edge_entry.origin,
264    };
265
266    let new_edge_entry = EdgeEntry {
267        entries: [inner_edge_entry, outer_edge_entry],
268        directed_data: [DE::default(), DE::default()],
269        undirected_data: UE::default(),
270    };
271
272    let new_edge_handle = FixedUndirectedEdgeHandle::new(dcel.num_undirected_edges());
273
274    let new_inner_handle = FixedDirectedEdgeHandle::new_normalized(new_edge_handle.index());
275    let new_outer_handle = new_inner_handle.rev();
276
277    dcel.half_edge_mut(edge_entry.prev).next = new_outer_handle;
278    dcel.half_edge_mut(edge).prev = new_inner_handle;
279
280    dcel.half_edge_mut(edge_entry.next).next = new_inner_handle;
281    dcel.half_edge_mut(next_entry.next).prev = new_outer_handle;
282
283    dcel.half_edge_mut(edge).face = new_face_handle;
284    dcel.half_edge_mut(edge_entry.next).face = new_face_handle;
285
286    dcel.faces[OUTER_FACE_HANDLE.index()].adjacent_edge = Some(new_outer_handle);
287
288    dcel.edges.push(new_edge_entry);
289    dcel.faces.push(FaceEntry {
290        adjacent_edge: Some(new_inner_handle),
291        data: F::default(),
292    });
293
294    new_outer_handle
295}

297pub fn create_new_face_adjacent_to_edge<V, DE, UE, F>(
298    dcel: &mut Dcel<V, DE, UE, F>,
299    edge: FixedDirectedEdgeHandle,
300    new_vertex_handle: FixedVertexHandle,
301) -> FixedVertexHandle
302where
303    DE: Default,
304    UE: Default,
305    F: Default,
306{
307    // Before:
308    //
309    // --- edge --->
310    //
311    // After:
312    //
313    //  new_prev.rev
314    //  |   ^ <---- new
315    //  |  / \
316    //  | /   \
317    //  V/ new \<---new_next.rev
318    //  / face  \
319    // /         \
320    // --- edge -->
321    let edge_entry = *dcel.half_edge(edge);
322
323    let edge_from = edge_entry.origin;
324    let edge_to = dcel.directed_edge(edge).to().fix();
325
326    let new_next_handle = FixedUndirectedEdgeHandle::new(dcel.num_undirected_edges());
327    let new_prev_handle = FixedUndirectedEdgeHandle::new(dcel.num_undirected_edges() + 1);
328
329    let new_face_handle = FixedFaceHandle::<PossiblyOuterTag>::new(dcel.num_faces());
330
331    let new_next = HalfEdgeEntry {
332        next: new_prev_handle.normalized(),
333        prev: edge,
334        face: new_face_handle,
335        origin: edge_to,
336    };
337
338    let new_next_rev = HalfEdgeEntry {
339        next: edge_entry.next,
340        prev: new_prev_handle.not_normalized(),
341        face: edge_entry.face,
342        origin: new_vertex_handle,
343    };
344
345    let new_prev = HalfEdgeEntry {
346        next: edge,
347        prev: new_next_handle.normalized(),
348        face: new_face_handle,
349        origin: new_vertex_handle,
350    };
351
352    let new_prev_rev = HalfEdgeEntry {
353        next: new_next_handle.not_normalized(),
354        prev: edge_entry.prev,
355        face: edge_entry.face,
356        origin: edge_from,
357    };
358
359    dcel.edges.push(EdgeEntry {
360        entries: [new_next, new_next_rev],
361        directed_data: [DE::default(), DE::default()],
362        undirected_data: UE::default(),
363    });
364
365    dcel.edges.push(EdgeEntry {
366        entries: [new_prev, new_prev_rev],
367        directed_data: [DE::default(), DE::default()],
368        undirected_data: UE::default(),
369    });
370
371    dcel.faces.push(FaceEntry {
372        adjacent_edge: Some(edge),
373        data: F::default(),
374    });
375
376    dcel.vertices[new_vertex_handle.index()].out_edge = Some(new_prev_handle.normalized());
377
378    *dcel.half_edge_mut(edge) = HalfEdgeEntry {
379        prev: new_prev_handle.normalized(),
380        next: new_next_handle.normalized(),
381        face: new_face_handle,
382        ..edge_entry
383    };
384
385    dcel.faces[edge_entry.face.index()].adjacent_edge = Some(new_prev_handle.not_normalized());
386
387    dcel.half_edge_mut(edge_entry.next).prev = new_next_handle.not_normalized();
388    dcel.half_edge_mut(edge_entry.prev).next = new_prev_handle.not_normalized();
389
390    new_vertex_handle
391}

393pub fn extend_line<V, DE, UE, F>(
394    dcel: &mut Dcel<V, DE, UE, F>,
395    end_vertex: FixedVertexHandle,
396    new_vertex_handle: FixedVertexHandle,
397) -> FixedVertexHandle
398where
399    DE: Default,
400    UE: Default,
401{
402    // Before:
403    //
404    // <--out_edge-- end_vertex
405    //
406    // After:
407    // <--out_edge-- end_vertex <---new_edge--- new_vertex
408
409    let end_vertex = dcel.vertex(end_vertex);
410    let out_edge = end_vertex.out_edge().expect("end vertex must not isolated");
411    let in_edge = out_edge.rev();
412
413    assert_eq!(
414        out_edge,
415        in_edge.rev(),
416        "end vertex is not at the end of the line"
417    );
418
419    let new_edge = FixedDirectedEdgeHandle::new_normalized(dcel.num_undirected_edges());
420    let new_edge_rev = new_edge.rev();
421
422    let face = out_edge.face().fix();
423
424    let out_edge = out_edge.fix();
425    let in_edge = in_edge.fix();
426    let end_vertex = end_vertex.fix();
427
428    dcel.half_edge_mut(out_edge).prev = new_edge;
429    dcel.half_edge_mut(in_edge).next = new_edge_rev;
430
431    dcel.edges.push(EdgeEntry::new(
432        HalfEdgeEntry {
433            next: out_edge,
434            prev: new_edge_rev,
435            face,
436            origin: new_vertex_handle,
437        },
438        HalfEdgeEntry {
439            next: new_edge,
440            prev: in_edge,
441            face,
442            origin: end_vertex,
443        },
444    ));
445
446    dcel.vertices[new_vertex_handle.index()].out_edge = Some(new_edge);
447
448    new_vertex_handle
449}

451pub fn split_edge_when_all_vertices_on_line<V, DE, UE, F>(
452    dcel: &mut Dcel<V, DE, UE, F>,
453    edge: FixedDirectedEdgeHandle,
454    new_vertex_handle: FixedVertexHandle,
455) -> [FixedDirectedEdgeHandle; 2]
456where
457    DE: Default,
458    UE: Default,
459{
460    let edge = dcel.directed_edge(edge);
461    let rev = edge.rev();
462    assert_eq!(edge.face(), rev.face());
463
464    let new_edge = FixedDirectedEdgeHandle::new_normalized(dcel.num_undirected_edges());
465    let new_edge_rev = new_edge.rev();
466
467    let edge_next = edge.next().fix();
468    let rev_prev = rev.prev().fix();
469
470    let to = edge.to().fix();
471
472    let face = edge.face().fix();
473    let edge = edge.fix();
474    let rev = rev.fix();
475
476    let is_isolated = edge_next == rev;
477
478    // Before:
479    //
480    // from ---edge---> to
481    //
482    // After:
483    //
484    // from ---edge---> new_vertex ---new_edge---> to
485
486    dcel.half_edge_mut(edge).next = new_edge;
487    dcel.half_edge_mut(rev).prev = new_edge_rev;
488
489    dcel.half_edge_mut(rev).origin = new_vertex_handle;
490
491    dcel.vertices[to.index()].out_edge = Some(new_edge_rev);
492
493    let (new_edge_next, new_rev_prev) = if is_isolated {
494        (new_edge_rev, new_edge)
495    } else {
496        dcel.half_edge_mut(edge_next).prev = new_edge;
497        dcel.half_edge_mut(rev_prev).next = new_edge_rev;
498        (edge_next, rev_prev)
499    };
500
501    dcel.edges.push(EdgeEntry::new(
502        HalfEdgeEntry {
503            next: new_edge_next,
504            prev: edge,
505            face,
506            origin: new_vertex_handle,
507        },
508        HalfEdgeEntry {
509            next: rev,
510            prev: new_rev_prev,
511            face,
512            origin: to,
513        },
514    ));
515
516    dcel.vertices[new_vertex_handle.index()].out_edge = Some(new_edge);
517
518    [edge, new_edge]
519}

521/// Splits `edge_handle` only one side. Used to split edges on the convex hull.
522///
523/// Returns the newly inserted vertex and the two resulting parts of the split edge.
524///
525/// The returned edges will point in the same direction as the input edge. The first
526/// returned edge has the same origin as the input edge, the second returned edge has the
527/// same destination as the input edge.
528pub fn split_half_edge<V, DE, UE, F>(
529    dcel: &mut Dcel<V, DE, UE, F>,
530    edge_handle: FixedDirectedEdgeHandle,
531    new_vertex_handle: FixedVertexHandle,
532) -> [FixedDirectedEdgeHandle; 2]
533where
534    DE: Default,
535    UE: Default,
536    F: Default,
537{
538    // Original face:
539    //
540    //      to
541    //      +
542    //      |\
543    //      | \
544    //      |  \
545    //      |   \
546    //      |    \
547    //      |     \
548    //      |     e\
549    //      |en     \
550    //      |  f1    \
551    //      |    ep   \
552    //      +----------+
553    //      v         from
554    //
555    // en = edge next
556    // ep = edge prev
557    //
558    // After splitting e0:
559    //
560    //      to
561    //      +
562    //      |\
563    //      | \
564    //      |  \
565    //      | e2\
566    //      |    \
567    //      | nf  nv
568    //      |    / \
569    //      | e1/   \
570    //      |  /    e\
571    //      | /  f1   \
572    //      +----------+
573    //      v         from
574    //
575    // nf = new face
576    // e1, e2 = new edges
577    // nv = new vertex
578    //
579    // This would return [e, e2]
580    let edge = dcel.directed_edge(edge_handle);
581    let v = edge.prev().from().fix();
582    let to = edge.to().fix();
583    let edge_next = edge.next().fix();
584    let edge_prev = edge.prev().fix();
585
586    let edge_twin = edge.rev().fix();
587    let edge_twin_prev = edge.rev().prev().fix();
588    let edge_twin_face = edge.rev().face().fix();
589
590    let f1 = edge.face().fix();
591    let nf = FixedFaceHandle::new(dcel.faces.len());
592
593    let e1 = FixedDirectedEdgeHandle::new_normalized(dcel.edges.len());
594    let t1 = e1.rev();
595    let e2 = FixedDirectedEdgeHandle::new_normalized(dcel.edges.len() + 1);
596    let t2 = e2.rev();
597
598    let edge1 = HalfEdgeEntry {
599        next: e2,
600        prev: edge_next,
601        origin: v,
602        face: nf,
603    };
604
605    let twin1 = HalfEdgeEntry {
606        next: edge_prev,
607        prev: edge_handle,
608        face: f1,
609        origin: new_vertex_handle,
610    };
611
612    let edge2 = HalfEdgeEntry {
613        next: edge_next,
614        prev: e1,
615        origin: new_vertex_handle,
616        face: nf,
617    };
618
619    let twin2 = HalfEdgeEntry {
620        next: edge_twin,
621        prev: edge_twin_prev,
622        face: edge_twin_face,
623        origin: to,
624    };
625
626    let new_face = FaceEntry {
627        adjacent_edge: Some(e2),
628        data: Default::default(),
629    };
630
631    dcel.edges.push(EdgeEntry::new(edge1, twin1));
632    dcel.edges.push(EdgeEntry::new(edge2, twin2));
633    dcel.faces.push(new_face);
634    dcel.vertices[new_vertex_handle.index()].out_edge = Some(e2);
635
636    dcel.half_edge_mut(edge_twin_prev).next = t2;
637
638    dcel.half_edge_mut(edge_next).prev = e2;
639    dcel.half_edge_mut(edge_prev).prev = t1;
640    dcel.half_edge_mut(edge_twin).prev = t2;
641    dcel.half_edge_mut(edge_next).next = e1;
642    dcel.half_edge_mut(edge_handle).next = t1;
643
644    dcel.half_edge_mut(edge_next).face = nf;
645    dcel.half_edge_mut(edge_twin).origin = new_vertex_handle;
646
647    dcel.vertices[to.index()].out_edge = Some(e2.rev());
648    dcel.faces[f1.index()].adjacent_edge = Some(edge_handle);
649
650    [edge_handle, e2]
651}

653/// Splits `edge_handle`, introducing 6 new half edges, two new faces and one
654/// new vertex.
655///
656/// Returns the newly created vertex handle and the two resulting parts of the split edge.
657/// The returned edges will point in the same direction as the input edge
658pub fn split_edge<V, DE, UE, F>(
659    dcel: &mut Dcel<V, DE, UE, F>,
660    edge_handle: FixedDirectedEdgeHandle,
661    new_vertex_handle: FixedVertexHandle,
662) -> [FixedDirectedEdgeHandle; 2]
663where
664    DE: Default,
665    UE: Default,
666    F: Default,
667{
668    // Original quad:
669    //
670    //     v1          v4
671    //      +----------+
672    //      |\   ep    |
673    //      | \        |
674    //      |  \    f0 |
675    //      |   \e0    |
676    //      |    \     |
677    //      |tn   \  en|
678    //      |      \   |
679    //      |       \  |
680    //      |  f1    \ |
681    //      |    tp   \|
682    //      +----------+
683    //     v2          v3
684    //
685    // After splitting e0:
686    //
687    //      +----------+
688    //      |\   ep   /|
689    //      | \e0    / |
690    //      |  \    /  |
691    //      |   \  /e3 |
692    //      |tn  \/    |
693    //      |    /v0   |
694    //      | e1/  \ en|
695    //      |  /    \  |
696    //      | /    e2\ |
697    //      |/   tp   \|
698    //      +----------+
699    //
700    // All edges are oriented counter clock wise
701    // f0 .. f3 will denote the faces adjacent to e0 .. e3
702    // t0 .. t3 will denote the twins of e0 .. e3
703    //
704    // Splitting edge e0 would return [e0, e2.rev()]
705    let edge = dcel.half_edge(edge_handle);
706    let twin = dcel.half_edge(edge_handle.rev());
707
708    let f0 = edge.face;
709    let f1 = twin.face;
710    let f2 = FixedFaceHandle::new(dcel.faces.len());
711    let f3 = FixedFaceHandle::new(dcel.faces.len() + 1);
712
713    let e0 = edge_handle;
714    let t0 = e0.rev();
715    let e1 = FixedDirectedEdgeHandle::new_normalized(dcel.edges.len());
716    let t1 = e1.rev();
717    let e2 = FixedDirectedEdgeHandle::new_normalized(dcel.edges.len() + 1);
718    let t2 = e2.rev();
719    let e3 = FixedDirectedEdgeHandle::new_normalized(dcel.edges.len() + 2);
720    let t3 = e3.rev();
721    let ep = edge.prev;
722    let en = edge.next;
723    let tn = twin.next;
724    let tp = twin.prev;
725
726    let v0 = new_vertex_handle;
727    let v1 = edge.origin;
728    let v2 = dcel.half_edge(tp).origin;
729    let v3 = twin.origin;
730    let v4 = dcel.half_edge(ep).origin;
731
732    let edge0 = HalfEdgeEntry {
733        next: t3,
734        prev: ep,
735        origin: v1,
736        face: f0,
737    };
738
739    let twin0 = HalfEdgeEntry {
740        next: tn,
741        prev: e1,
742        origin: v0,
743        face: f1,
744    };
745
746    let edge1 = HalfEdgeEntry {
747        next: t0,
748        prev: tn,
749        origin: v2,
750        face: f1,
751    };
752
753    let twin1 = HalfEdgeEntry {
754        next: tp,
755        prev: e2,
756        origin: v0,
757        face: f2,
758    };
759
760    let edge2 = HalfEdgeEntry {
761        next: t1,
762        prev: tp,
763        origin: v3,
764        face: f2,
765    };
766
767    let twin2 = HalfEdgeEntry {
768        next: en,
769        prev: e3,
770        origin: v0,
771        face: f3,
772    };
773
774    let edge3 = HalfEdgeEntry {
775        next: t2,
776        prev: en,
777        origin: v4,
778        face: f3,
779    };
780
781    let twin3 = HalfEdgeEntry {
782        next: ep,
783        prev: e0,
784        origin: v0,
785        face: f0,
786    };
787
788    let face2 = FaceEntry {
789        adjacent_edge: Some(e2),
790        data: F::default(),
791    };
792
793    let face3 = FaceEntry {
794        adjacent_edge: Some(e3),
795        data: F::default(),
796    };
797
798    *dcel.half_edge_mut(e0) = edge0;
799    *dcel.half_edge_mut(t0) = twin0;
800    dcel.edges.push(EdgeEntry::new(edge1, twin1));
801    dcel.edges.push(EdgeEntry::new(edge2, twin2));
802    dcel.edges.push(EdgeEntry::new(edge3, twin3));
803
804    dcel.half_edge_mut(en).next = e3;
805    dcel.half_edge_mut(en).prev = t2;
806    dcel.half_edge_mut(en).face = f3;
807
808    dcel.half_edge_mut(tp).next = e2;
809    dcel.half_edge_mut(tp).prev = t1;
810    dcel.half_edge_mut(tp).face = f2;
811
812    dcel.half_edge_mut(tn).next = e1;
813    dcel.half_edge_mut(ep).prev = t3;
814
815    dcel.vertices[new_vertex_handle.index()].out_edge = Some(t0);
816    dcel.vertices[v3.index()].out_edge = Some(e2);
817
818    dcel.faces[f0.index()].adjacent_edge = Some(e0);
819    dcel.faces[f1.index()].adjacent_edge = Some(e1);
820    dcel.faces.push(face2);
821    dcel.faces.push(face3);
822
823    [e0, e2.rev()]
824}

870pub fn insert_into_triangle<V, DE, UE, F>(
871    dcel: &mut Dcel<V, DE, UE, F>,
872    v: FixedVertexHandle,
873    f0: FixedFaceHandle<InnerTag>,
874) -> FixedVertexHandle
875where
876    DE: Default,
877    UE: Default,
878    F: Default,
879{
880    // All edges are oriented counterclockwise
881    //
882    // Original triangle:
883    //       v1
884    //      / \
885    //     /   \
886    //    /e1   \
887    //   /   f0  \
888    //  /       e0\
889    // v2__e2_____v0
890    //
891    // With v inserted (e0, e1 and e2 as above):
892    //                 .
893    //               / # \
894    //             /   #   \
895    //       f1  /  e4 # e3  \f0
896    //         /     __v__     \
897    //       /   e5_#     #__e8  \
898    //     / ___#             #___ \
899    //   /__#   e6          e7    #__\
900    // /#_____________________________#\
901    //                f2
902
903    let e0 = dcel.faces[f0.index()].adjacent_edge.expect(
904        "Face without adjacent edge should not exist when at least two vertices are present. \
905        This is a bug.",
906    );
907
908    let e1 = dcel.half_edge(e0).next;
909    let e2 = dcel.half_edge(e1).next;
910    let e3 = FixedDirectedEdgeHandle::new_normalized(dcel.edges.len());
911    let e4 = e3.rev();
912    let e5 = FixedDirectedEdgeHandle::new_normalized(dcel.edges.len() + 1);
913    let e6 = e5.rev();
914    let e7 = FixedDirectedEdgeHandle::new_normalized(dcel.edges.len() + 2);
915    let e8 = e7.rev();
916
917    let v0 = dcel.half_edge(e0).origin;
918    let v1 = dcel.half_edge(e1).origin;
919    let v2 = dcel.half_edge(e2).origin;
920
921    let f1 = FixedFaceHandle::new(dcel.faces.len());
922    let f2 = FixedFaceHandle::new(dcel.faces.len() + 1);
923
924    let face1 = FaceEntry {
925        adjacent_edge: Some(e1),
926        data: F::default(),
927    };
928
929    let face2 = FaceEntry {
930        adjacent_edge: Some(e2),
931        data: F::default(),
932    };
933
934    dcel.faces.push(face1);
935    dcel.faces.push(face2);
936
937    dcel.vertices[v.index()].out_edge = Some(e4);
938
939    dcel.half_edge_mut(e0).prev = e8;
940    dcel.half_edge_mut(e0).next = e3;
941    dcel.half_edge_mut(e1).prev = e4;
942    dcel.half_edge_mut(e1).next = e5;
943    dcel.half_edge_mut(e1).face = f1;
944    dcel.half_edge_mut(e2).prev = e6;
945    dcel.half_edge_mut(e2).next = e7;
946    dcel.half_edge_mut(e2).face = f2;
947
948    let edge3 = HalfEdgeEntry {
949        next: e8,
950        prev: e0,
951        origin: v1,
952        face: f0.adjust_inner_outer(),
953    };
954
955    let edge4 = HalfEdgeEntry {
956        next: e1,
957        prev: e5,
958        origin: v,
959        face: f1,
960    };
961
962    let edge5 = HalfEdgeEntry {
963        next: e4,
964        prev: e1,
965        origin: v2,
966        face: f1,
967    };
968
969    let edge6 = HalfEdgeEntry {
970        next: e2,
971        prev: e7,
972        origin: v,
973        face: f2,
974    };
975
976    let edge7 = HalfEdgeEntry {
977        next: e6,
978        prev: e2,
979        origin: v0,
980        face: f2,
981    };
982
983    let edge8 = HalfEdgeEntry {
984        next: e0,
985        prev: e3,
986        origin: v,
987        face: f0.adjust_inner_outer(),
988    };
989
990    dcel.edges.push(EdgeEntry::new(edge3, edge4));
991    dcel.edges.push(EdgeEntry::new(edge5, edge6));
992    dcel.edges.push(EdgeEntry::new(edge7, edge8));
993
994    // Return inserted vertex handle
995    v.adjust_inner_outer()
996}

1049/// Flip an edge in cw direction
1050pub fn flip_cw<V, DE, UE, F>(dcel: &mut Dcel<V, DE, UE, F>, e: FixedUndirectedEdgeHandle) {
1051    let e = e.as_directed();
1052    let e_entry = *dcel.half_edge(e);
1053    let en = e_entry.next;
1054    let ep = e_entry.prev;
1055    let e_face = e_entry.face;
1056    let e_origin = e_entry.origin;
1057
1058    let t = e.rev();
1059    let t_entry = *dcel.half_edge(t);
1060    let tn = t_entry.next;
1061    let tp = t_entry.prev;
1062    let t_face = t_entry.face;
1063    let t_origin = t_entry.origin;
1064
1065    dcel.half_edge_mut(en).next = e;
1066    dcel.half_edge_mut(en).prev = tp;
1067    dcel.half_edge_mut(e).next = tp;
1068    dcel.half_edge_mut(e).prev = en;
1069    dcel.half_edge_mut(e).origin = dcel.half_edge(ep).origin;
1070    dcel.half_edge_mut(tp).next = en;
1071    dcel.half_edge_mut(tp).prev = e;
1072    dcel.half_edge_mut(tp).face = e_face;
1073
1074    dcel.half_edge_mut(tn).next = t;
1075    dcel.half_edge_mut(tn).prev = ep;
1076    dcel.half_edge_mut(t).next = ep;
1077    dcel.half_edge_mut(t).prev = tn;
1078    dcel.half_edge_mut(t).origin = dcel.half_edge(tp).origin;
1079    dcel.half_edge_mut(ep).next = tn;
1080    dcel.half_edge_mut(ep).prev = t;
1081    dcel.half_edge_mut(ep).face = t_face;
1082
1083    dcel.vertices[e_origin.index()].out_edge = Some(tn);
1084    dcel.vertices[t_origin.index()].out_edge = Some(en);
1085
1086    dcel.faces[e_face.index()].adjacent_edge = Some(e);
1087    dcel.faces[t_face.index()].adjacent_edge = Some(t);
1088}

1090/// Vertex removal has two stages: First, the vertex is disconnected from its surroundings (isolated).
1091/// Then, any edges, faces and the vertex itself that have become obsolete are removed.
1092pub fn cleanup_isolated_vertex<V, DE, UE, F>(
1093    dcel: &mut Dcel<V, DE, UE, F>,
1094    isolated: &mut IsolateVertexResult,
1095) {
1096    // Remove disconnected edges, faces and the vertex
1097    isolated.edges_to_remove.sort_unstable();
1098    for edge in isolated.edges_to_remove.iter().rev() {
1099        swap_remove_undirected_edge(dcel, *edge);
1100    }
1101
1102    isolated.faces_to_remove.sort_unstable();
1103    for face in isolated.faces_to_remove.iter().rev() {
1104        swap_remove_face(dcel, *face);
1105    }
1106}
1107
1108fn swap_remove_undirected_edge<V, DE, UE, F>(
1109    dcel: &mut Dcel<V, DE, UE, F>,
1110    edge_handle: FixedUndirectedEdgeHandle,
1111) {
1112    dcel.edges.swap_remove(edge_handle.index());
1113    if dcel.num_undirected_edges() > edge_handle.index() {
1114        let directed = edge_handle.as_directed();
1115        fix_handle_swap(dcel, directed.rev());
1116        fix_handle_swap(dcel, directed);
1117    }
1118}
1119
1120fn fix_handle_swap<V, DE, UE, F>(
1121    dcel: &mut Dcel<V, DE, UE, F>,
1122    edge_handle: FixedDirectedEdgeHandle,
1123) {
1124    // An edge handle was moved to index "edge_handle".
1125    // Make sure to update all references to this edge
1126    //
1127    // Since this method gets only called as part of a swap_remove, the edge that got
1128    // swapped in always had the index dcel.num_undirected_edges .
1129    let old_handle = FixedUndirectedEdgeHandle::new(dcel.num_undirected_edges());
1130    let old_to_new = |handle: FixedDirectedEdgeHandle| {
1131        let undirected = handle.as_undirected();
1132        if undirected == old_handle {
1133            // handle.as_undirected is now edge_handle
1134            if handle.is_normalized() {
1135                edge_handle.as_undirected().normalized()
1136            } else {
1137                edge_handle.as_undirected().not_normalized()
1138            }
1139        } else {
1140            handle
1141        }
1142    };
1143    let edge_next = dcel.half_edge(edge_handle).next;
1144    let edge_prev = dcel.half_edge(edge_handle).prev;
1145    let edge_next = old_to_new(edge_next);
1146    let edge_prev = old_to_new(edge_prev);
1147    dcel.half_edge_mut(edge_prev).next = edge_handle;
1148    dcel.half_edge_mut(edge_next).prev = edge_handle;
1149
1150    let edge_origin = dcel.half_edge(edge_handle).origin;
1151    let edge_face = dcel.half_edge(edge_handle).face;
1152    dcel.vertices[edge_origin.index()].out_edge = Some(edge_handle);
1153    dcel.faces[edge_face.index()].adjacent_edge = Some(edge_handle);
1154}

1156/// Removes a vertex from the DCEL by swapping in another.
1157///
1158/// This operation *will* leave the DCEL in an inconsistent state, the caller must ensure to
1159/// fix the site of the removed vertex appropriately. This method only ensures that
1160/// references to the swapped in vertex are updated accordingly.
1161/// Returns the data of the removed vertex.
1162pub fn swap_remove_vertex<V, DE, UE, F>(
1163    dcel: &mut Dcel<V, DE, UE, F>,
1164    vertex_handle: FixedVertexHandle,
1165) -> RemovalResult<V> {
1166    let data = dcel.vertices.swap_remove(vertex_handle.index()).data;
1167    let mut swapped_in_vertex = None;
1168    if dcel.vertices.len() != vertex_handle.index() {
1169        // Update origin of all out edges of the swapped in vertex
1170        swapped_in_vertex = Some(FixedVertexHandle::new(dcel.vertices.len()));
1171        let to_update: SmallVec<[_; 8]> = dcel
1172            .vertex(vertex_handle)
1173            .out_edges()
1174            .map(|handle| handle.fix())
1175            .collect();
1176        for e in to_update {
1177            dcel.half_edge_mut(e).origin = vertex_handle;
1178        }
1179    };
1180
1181    RemovalResult {
1182        removed_vertex: data,
1183        swapped_in_vertex,
1184    }
1185}
1186
1187/// Removes a face from the DCEL by swapping in another face.
1188///
1189/// This *will* leave the DCEL in an inconsistent state. It is the callers responsibility
1190/// to fix the site around the removed face. This method only ensure that any references to
1191/// the swapped in faces are updated accordingly.
1192fn swap_remove_face<V, DE, UE, F>(dcel: &mut Dcel<V, DE, UE, F>, face: FixedFaceHandle<InnerTag>) {
1193    dcel.faces.swap_remove(face.index());
1194    if dcel.faces.len() > face.index() {
1195        let neighs: SmallVec<[_; 3]> = dcel
1196            .face(face)
1197            .adjacent_edges()
1198            .iter()
1199            .map(|edge| edge.fix())
1200            .collect();
1201        for n in neighs {
1202            dcel.half_edge_mut(n).face = face.adjust_inner_outer();
1203        }
1204    }
1205}

1207pub fn remove_when_degenerate<V, DE, UE, F>(
1208    dcel: &mut Dcel<V, DE, UE, F>,
1209    vertex_to_remove: FixedVertexHandle,
1210) -> RemovalResult<V> {
1211    match dcel.num_vertices() {
1212        0 => panic!("Cannot remove vertex when triangulation is empty"),
1213        1 => remove_when_one_vertex_left(dcel, vertex_to_remove),
1214        2 => remove_when_two_vertices_left(dcel, vertex_to_remove),
1215        _ => remove_when_all_vertices_on_line(dcel, vertex_to_remove),
1216    }
1217}

1219fn remove_when_one_vertex_left<V, DE, UE, F>(
1220    dcel: &mut Dcel<V, DE, UE, F>,
1221    vertex_to_remove: FixedVertexHandle,
1222) -> RemovalResult<V> {
1223    assert_eq!(
1224        vertex_to_remove.index(),
1225        0,
1226        "Attempting to remove invalid vertex"
1227    );
1228
1229    RemovalResult {
1230        removed_vertex: dcel.vertices.pop().unwrap().data,
1231        swapped_in_vertex: None,
1232    }
1233}
1234
1235fn remove_when_two_vertices_left<V, DE, UE, F>(
1236    dcel: &mut Dcel<V, DE, UE, F>,
1237    vertex_to_remove: FixedVertexHandle,
1238) -> RemovalResult<V> {
1239    assert_eq!(dcel.num_faces(), 1);
1240    assert_eq!(dcel.num_vertices(), 2);
1241    assert_eq!(dcel.num_directed_edges(), 2);
1242
1243    let swapped_in_vertex = if vertex_to_remove.index() == 1 {
1244        None
1245    } else {
1246        Some(FixedVertexHandle::new(1))
1247    };
1248
1249    let result = dcel.vertices.swap_remove(vertex_to_remove.index()).data;
1250    dcel.faces[OUTER_FACE_HANDLE.index()].adjacent_edge = None;
1251    dcel.vertices[0].out_edge = None;
1252    dcel.edges.clear();
1253    RemovalResult {
1254        removed_vertex: result,
1255        swapped_in_vertex,
1256    }
1257}
1258
1259fn remove_when_all_vertices_on_line<V, DE, UE, F>(
1260    dcel: &mut Dcel<V, DE, UE, F>,
1261    vertex_to_remove: FixedVertexHandle,
1262) -> RemovalResult<V> {
1263    let out_edges: Vec<_> = dcel.vertex(vertex_to_remove).out_edges().collect();
1264    match &*out_edges {
1265        [out_edge1] => {
1266            // Vertex to remove is at one of the ends of the line
1267            //
1268            // Topology sketch:
1269            // vertex_to_remove ---out_edge1---> vertex_to_update ---o_next---->...
1270
1271            let vertex_to_update = out_edge1.to().fix();
1272            let o_next = out_edge1.next().fix();
1273            let out_edge1 = out_edge1.fix();
1274
1275            dcel.half_edge_mut(o_next).prev = o_next.rev();
1276            dcel.half_edge_mut(o_next.rev()).next = o_next;
1277            dcel.vertices[vertex_to_update.index()].out_edge = Some(o_next);
1278            dcel.faces[OUTER_FACE_HANDLE.index()].adjacent_edge = Some(o_next);
1279
1280            swap_remove_undirected_edge(dcel, out_edge1.as_undirected());
1281
1282            swap_remove_vertex(dcel, vertex_to_remove)
1283        }
1284        [e1, e2] => {
1285            // Topology sketch (with v0 = vertex_to_remove, t1 = e1.rev()):
1286            // v1 ---t1--> v0 ---e2--> v2
1287            //
1288            // Should become this:
1289            // v1 ---------t1--------> v2
1290            let t1 = e1.rev().fix();
1291            let e2_next = e2.next().fix();
1292            let e2_to = e2.to().fix();
1293            let t2_prev = e2.rev().prev().fix();
1294            let e2 = e2.fix();
1295
1296            if e2_next == e2.rev() {
1297                dcel.half_edge_mut(t1).next = t1.rev();
1298                dcel.half_edge_mut(t1.rev()).prev = t1;
1299            } else {
1300                dcel.half_edge_mut(e2_next).prev = t1;
1301                dcel.half_edge_mut(t1).next = e2_next;
1302
1303                dcel.half_edge_mut(t2_prev).next = t1.rev();
1304                dcel.half_edge_mut(t1.rev()).prev = t2_prev;
1305            }
1306
1307            dcel.vertices[e2_to.index()].out_edge = Some(t1.rev());
1308            dcel.half_edge_mut(t1.rev()).origin = e2_to;
1309
1310            dcel.faces[OUTER_FACE_HANDLE.index()].adjacent_edge = Some(t1);
1311
1312            let result = swap_remove_vertex(dcel, vertex_to_remove);
1313            swap_remove_undirected_edge(dcel, e2.as_undirected());
1314            result
1315        }
1316        _ => panic!("Vertex with invalid out edges found. This is a bug."),
1317    }
