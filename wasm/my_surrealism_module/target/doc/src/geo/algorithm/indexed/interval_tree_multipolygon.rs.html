11/// A Y-coordinate wrapper that implements `Ord` for use in the interval tree.
12/// Unlike a full point comparison, this only compares Y values, avoiding
13/// unnecessary X comparisons since we know X is always zero in our usage.
14#[derive(PartialEq, Clone, Copy)]
15struct YValue<T: GeoNum>(T);
16
17impl<T: GeoNum> std::fmt::Debug for YValue<T> {
18    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
19        f.debug_tuple("Y").field(&self.0).finish()
20    }
21}

23impl<T: GeoNum> PartialOrd for YValue<T> {
24    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
25        Some(self.cmp(other))
26    }
27}

29/// Direct Y-value comparison, no X coordinate overhead
30impl<T: GeoNum> Ord for YValue<T> {
31    fn cmp(&self, other: &Self) -> Ordering {
32        self.0.total_cmp(&other.0)
33    }
34}

38impl<T: GeoNum> From<T> for YValue<T> {

44struct YIntervalSegment<T: GeoNum> {
45    segment: (Coord<T>, Coord<T>),
46    /// Pre-computed maximum x-coordinate of the segment endpoints.
47    /// Used for early rejection of segments that don't extend far enough right.
48    /// This avoids computing max(seg.0.x, seg.1.x) for every point query.
49    x_max: T,
50}
51
52/// A [`MultiPolygon`] backed by an [interval tree](https://en.wikipedia.org/wiki/Interval_tree) for fast containment queries
53pub struct IntervalTreeMultiPolygon<T: GeoNum> {
54    y_interval_tree: ITree<YValue<T>, YIntervalSegment<T>>,
55}
56
57impl<T: GeoNum> IntervalTreeMultiPolygon<T> {
58    pub fn new(mp: &MultiPolygon<T>) -> Self {
59        // Chain all exterior and interior line segments from all polygons
60        let segments =
61            mp.0.iter()
62                .flat_map(|polygon| {
63                    polygon.exterior().lines_iter().chain(
64                        polygon
65                            .interiors()
66                            .iter()
67                            .flat_map(|interior| interior.lines_iter()),
68                    )
69                })
70                .map(|line| Self::create_segment(line));
71
72        Self {
73            y_interval_tree: ITree::new(segments),
74        }
75    }
76
77    fn create_segment(line: Line<T>) -> sif_itree::Item<YValue<T>, YIntervalSegment<T>> {
78        let p1 = line.start;
79        let p2 = line.end;
80
81        let (y_min_val, y_max_val) = if p1.y.total_cmp(&p2.y).is_le() {
82            (p1.y, p2.y)
83        } else {
84            (p2.y, p1.y)
85        };
86        let y_min = YValue::from(y_min_val);
87        let y_max = YValue::from(y_max_val);
88
89        // Pre-compute maximum x for early rejection during queries
90        let x_max = if p1.x.total_cmp(&p2.x).is_ge() {
91            p1.x
92        } else {
93            p2.x
94        };
95
96        (
97            y_min..y_max,
98            YIntervalSegment {
99                segment: (p1, p2),
100                x_max,
101            },
102        )
103    }
104
105    pub(crate) fn containment(&self, coord: Coord<T>) -> CoordPos {
106        // Use winding number algorithm with robust predicates
107        // Based on coord_pos_relative_to_ring in coordinate_position.rs
108        let mut winding_number = 0;
109
110        // Query for segments whose Y-interval contains point.y
111        let point_y = YValue::from(coord.y);
112
113        let result = self
114            .y_interval_tree
115            .query(point_y..=point_y, |(_, segment)| {
116                // Early rejection: segment must extend to the right of point
117                if segment.x_max < coord.x {
118                    return ControlFlow::Continue(());
119                }
120
121                let seg = segment.segment;
122
123                // Apply winding number algorithm using robust predicates
124                let mut contribution = 0;
125                if seg.0.y <= coord.y {
126                    if seg.1.y >= coord.y {
127                        let o = <T>::Ker::orient2d(seg.0, seg.1, coord);
128                        if o == Orientation::CounterClockwise && seg.1.y != coord.y {
129                            contribution = 1;
130                        } else if o == Orientation::Collinear
131                            && value_in_between(coord.x, seg.0.x, seg.1.x)
132                        {
133                            // Point on boundary!
134                            return ControlFlow::Break(CoordPos::OnBoundary);
135                        }
136                    }
137                } else {
138                    let o = <T>::Ker::orient2d(seg.0, seg.1, coord);
139                    if o == Orientation::Clockwise {
140                        contribution = -1;
141                    } else if o == Orientation::Collinear
142                        && value_in_between(coord.x, seg.0.x, seg.1.x)
143                    {
144                        // Point on boundary!
145                        return ControlFlow::Break(CoordPos::OnBoundary);
146                    }
147                }
148
149                // Add the contribution directly: the winding order determines the sign
150                winding_number += contribution;
151                ControlFlow::Continue(())
152            });
153
154        match result {
155            ControlFlow::Break(pos) => pos,
156            ControlFlow::Continue(()) => {
157                if winding_number != 0 {
158                    CoordPos::Inside
159                } else {
160                    CoordPos::Outside
161                }
162            }
163        }
164    }
165}
