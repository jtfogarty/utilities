10/// Determine the length of a geometry using the [haversine formula].
11///
12/// [haversine formula]: https://en.wikipedia.org/wiki/Haversine_formula
13///
14/// *Note*: this implementation uses a mean earth radius of 6371.0088 km, based on the [recommendation of
15/// the IUGG](ftp://athena.fsv.cvut.cz/ZFG/grs80-Moritz.pdf)
16pub trait HaversineLength<T, RHS = Self> {
17    /// Determine the length of a geometry using the [haversine formula].
18    ///
19    /// # Units
20    ///
21    /// - return value: meters
22    ///
23    /// # Examples
24    ///
25    /// ```
26    /// use geo::prelude::*;
27    /// use geo::LineString;
28    ///
29    /// let linestring = LineString::<f64>::from(vec![
30    ///     // New York City
31    ///     (-74.006, 40.7128),
32    ///     // London
33    ///     (-0.1278, 51.5074),
34    /// ]);
35    ///
36    /// let length = linestring.haversine_length();
37    ///
38    /// assert_eq!(
39    ///     5_570_230., // meters
40    ///     length.round()
41    /// );
42    /// ```
43    ///
44    /// [haversine formula]: https://en.wikipedia.org/wiki/Haversine_formula
45    fn haversine_length(&self) -> T;
46}
47
48#[allow(deprecated)]
49impl<T> HaversineLength<T> for Line<T>
50where
51    T: CoordFloat + FromPrimitive,
52{
53    fn haversine_length(&self) -> T {
54        Haversine.length(self)
55    }
56}
