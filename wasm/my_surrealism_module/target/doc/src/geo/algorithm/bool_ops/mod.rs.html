19/// Boolean Operations on geometry.
20///
21/// Boolean operations are set operations on geometries considered as a subset
22/// of the 2-D plane. The operations supported are: intersection, union,
23/// symmetric difference (xor), and set-difference on pairs of 2-D geometries and
24/// clipping a 1-D geometry with self.
25///
26/// These operations are implemented on [`Polygon`] and the [`MultiPolygon`]
27/// geometries.
28///
29/// # Validity
30///
31/// Note that the operations are strictly well-defined only on *valid*
32/// geometries. However, the implementation generally works well as long as the
33/// interiors of polygons are contained within their corresponding exteriors.
34///
35/// Degenerate 2-d geoms with 0 area are handled, and ignored by the algorithm.
36/// In particular, taking `union` with an empty geom should remove degeneracies
37/// and fix invalid polygons as long the interior-exterior requirement above is
38/// satisfied.
39///
40/// # Performance
41///
42/// For union operations on a large number of [`Polygon`]s or [`MultiPolygon`]s,
43/// using [`unary_union`] will yield far better performance.
44pub trait BooleanOps {
45    type Scalar: BoolOpsNum;
46
47    /// The exterior and interior rings of the geometry.
48    ///
49    /// It doesn't particularly matter which order they are in, as the topology algorithm counts crossings
50    /// to determine the interior and exterior of the polygon.
51    ///
52    /// It is required that the rings are from valid geometries, that the rings not overlap.
53    /// In the case of a MultiPolygon, this requires that none of its polygon's interiors may overlap.
54    fn rings(&self) -> impl Iterator<Item = &LineString<Self::Scalar>>;
55
56    /// Performs a boolean operation between shapes using the default [`FillRule::EvenOdd`] fill rule.
57    ///
58    /// The `EvenOdd` rule determines filled regions based on the parity of path crossings.
59    /// It correctly handles "holes" in polygons by accounting for path direction and overlapping regions.
60    ///
61    /// This behavior models that of a real-world laser cutter: each shape is treated as a continuous
62    /// cut from solid material. When a new shape overlaps an existing cut, the overlapping area
63    /// is removed, potentially splitting the geometry into multiple disjoint parts.
64    ///
65    /// This fill rule is intuitive for applications like CAD tooling, CNC routing,
66    /// and other manufacturing processes where overlapping shapes subtract from the material
67    /// rather than accumulate.
68    ///
69    /// To use a different fill rule, such as [`FillRule::NonZero`], use [`Self::boolean_op_with_fill_rule`] instead.
70    fn boolean_op(
71        &self,
72        other: &impl BooleanOps<Scalar = Self::Scalar>,
73        op: OpType,
74    ) -> MultiPolygon<Self::Scalar> {
75        self.boolean_op_with_fill_rule(other, op, FillRule::EvenOdd)
76    }
77
78    /// Performs a boolean operation with the specified fill rule.
79    ///
80    /// # Examples
81    ///
82    /// ```
83    /// use geo::algorithm::bool_ops::{BooleanOps, OpType, FillRule};
84    /// use geo::wkt;
85    ///
86    /// let polygon1 = wkt!(POLYGON((0.0 0.0, 10.0 0.0, 10.0 10.0, 0.0 10.0, 0.0 0.0)));
87    /// let polygon2 = wkt!(POLYGON((5.0 5.0, 15.0 5.0, 15.0 15.0, 5.0 15.0, 5.0 5.0)));
88    /// let result = polygon1.boolean_op_with_fill_rule(&polygon2, OpType::Union, FillRule::NonZero);
89    /// assert_eq!(result.0.len(), 1);
90    /// ```
91    fn boolean_op_with_fill_rule(
92        &self,
93        other: &impl BooleanOps<Scalar = Self::Scalar>,
94        op: OpType,
95        fill_rule: FillRule,
96    ) -> MultiPolygon<Self::Scalar> {
97        let subject = self.rings().map(ring_to_shape_path).collect::<Vec<_>>();
98        let clip = other.rings().map(ring_to_shape_path).collect::<Vec<_>>();
99        let shapes = subject.overlay(&clip, op.into(), fill_rule);
100        multi_polygon_from_shapes(shapes)
101    }
102
103    /// Returns the overlapping regions shared by both `self` and `other`.
104    fn intersection(
105        &self,
106        other: &impl BooleanOps<Scalar = Self::Scalar>,
107    ) -> MultiPolygon<Self::Scalar> {
108        self.boolean_op(other, OpType::Intersection)
109    }
110
111    /// Returns the overlapping regions shared by both `self` and `other`, using the specified fill rule.
112    fn intersection_with_fill_rule(
113        &self,
114        other: &impl BooleanOps<Scalar = Self::Scalar>,
115        fill_rule: FillRule,
116    ) -> MultiPolygon<Self::Scalar> {
117        self.boolean_op_with_fill_rule(other, OpType::Intersection, fill_rule)
118    }
119
120    /// Combines the regions of both `self` and `other` into a single geometry, removing
121    /// overlaps and merging boundaries. Consider using [`unary_union`] for efficiently combining several adjacent / overlapping geometries.
122    fn union(&self, other: &impl BooleanOps<Scalar = Self::Scalar>) -> MultiPolygon<Self::Scalar> {
123        self.boolean_op(other, OpType::Union)
124    }
125
126    /// Combines the regions of both `self` and `other` into a single geometry, removing
127    /// overlaps and merging boundaries, using the specified fill rule.
128    fn union_with_fill_rule(
129        &self,
130        other: &impl BooleanOps<Scalar = Self::Scalar>,
131        fill_rule: FillRule,
132    ) -> MultiPolygon<Self::Scalar> {
133        self.boolean_op_with_fill_rule(other, OpType::Union, fill_rule)
134    }
135
136    /// The regions that are in either `self` or `other`, but not in both.
137    fn xor(&self, other: &impl BooleanOps<Scalar = Self::Scalar>) -> MultiPolygon<Self::Scalar> {
138        self.boolean_op(other, OpType::Xor)
139    }
140
141    /// The regions that are in either `self` or `other`, but not in both.
142    fn xor_with_fill_rule(
143        &self,
144        other: &impl BooleanOps<Scalar = Self::Scalar>,
145        fill_rule: FillRule,
146    ) -> MultiPolygon<Self::Scalar> {
147        self.boolean_op_with_fill_rule(other, OpType::Xor, fill_rule)
148    }
149
150    /// The regions of `self` which are not in `other`.
151    fn difference(
152        &self,
153        other: &impl BooleanOps<Scalar = Self::Scalar>,
154    ) -> MultiPolygon<Self::Scalar> {
155        self.boolean_op(other, OpType::Difference)
156    }
157
158    /// The regions of `self` which are not in `other`, using the specified fill rule.
159    fn difference_with_fill_rule(
160        &self,
161        other: &impl BooleanOps<Scalar = Self::Scalar>,
162        fill_rule: FillRule,
163    ) -> MultiPolygon<Self::Scalar> {
164        self.boolean_op_with_fill_rule(other, OpType::Difference, fill_rule)
165    }
166
167    /// Clip a 1-D geometry with self.
168    ///
169    /// Returns the portion of `ls` that lies within `self` (known as the set-theoeretic
170    /// intersection) if `invert` is false, and the difference (`ls - self`) otherwise.
171    fn clip(
172        &self,
173        multi_line_string: &MultiLineString<Self::Scalar>,
174        invert: bool,
175    ) -> MultiLineString<Self::Scalar> {
176        self.clip_with_fill_rule(multi_line_string, invert, FillRule::EvenOdd)
177    }
178
179    /// Clip a 1-D geometry with self.
180    ///
181    /// Returns the portion of `ls` that lies within `self` (known as the set-theoeretic
182    /// intersection) if `invert` is false, and the difference (`ls - self`) otherwise.
183    fn clip_with_fill_rule(
184        &self,
185        multi_line_string: &MultiLineString<Self::Scalar>,
186        invert: bool,
187        fill_rule: FillRule,
188    ) -> MultiLineString<Self::Scalar> {
189        let subject: Vec<Vec<_>> = multi_line_string
190            .iter()
191            .map(|line_string| line_string.coords().map(|c| BoolOpsCoord(*c)).collect())
192            .collect();
193
194        let clip = self.rings().map(ring_to_shape_path).collect::<Vec<_>>();
195
196        let clip_rule = ClipRule {
197            invert,
198            boundary_included: true,
199        };
200        let paths = subject.clip_by(&clip, fill_rule, clip_rule);
201        i_overlay_integration::convert::multi_line_string_from_paths(paths)
202    }
203}

213/// Efficient [union](BooleanOps::union) of many adjacent / overlapping geometries
214///
215/// This is typically much faster than `union`ing a bunch of geometries together one at a time.
216///
217/// Note: Geometries can be wound in either direction, but the winding order must be consistent,
218/// and each polygon's interior rings must be wound opposite to its exterior.
219///
220/// See [Orient] for more information.
221///
222/// [Orient]: crate::algorithm::orient::Orient
223///
224/// # Arguments
225///
226/// `boppables`: A collection of `Polygon` or `MultiPolygons` to union together.
227///
228/// returns the union of all the inputs.
229///
230/// # Examples
231///
232/// ```
233/// use geo::algorithm::unary_union;
234/// use geo::wkt;
235///
236/// let right_piece = wkt!(POLYGON((4. 0.,4. 4.,8. 4.,8. 0.,4. 0.)));
237/// let left_piece = wkt!(POLYGON((0. 0.,0. 4.,4. 4.,4. 0.,0. 0.)));
238///
239/// // touches neither right nor left piece
240/// let separate_piece = wkt!(POLYGON((14. 10.,14. 14.,18. 14.,18. 10.,14. 10.)));
241///
242/// let polygons = vec![left_piece, separate_piece, right_piece];
243/// let actual_output = unary_union(&polygons);
244///
245/// let expected_output = wkt!(MULTIPOLYGON(
246///     // left and right piece have been combined
247///     ((0.0 4.0,0.0 0.0,8.0 0.0,8.0 4.0,0.0 4.0)),
248///     // separate piece remains separate
249///     ((14.0 14.0,14.0 10.0,18.0 10.0,18.0 14.0,14.0 14.0))
250/// ));
251/// assert_eq!(actual_output, expected_output);
252/// ```
253pub fn unary_union<'a, B: BooleanOps + 'a>(
254    boppables: impl IntoIterator<Item = &'a B>,
255) -> MultiPolygon<B::Scalar> {
256    let mut winding_order: Option<WindingOrder> = None;
257    let subject = boppables
258        .into_iter()
259        .flat_map(|boppable| {
260            let rings = boppable.rings();
261            rings
262                .map(|ring| {
263                    if winding_order.is_none() {
264                        winding_order = ring.winding_order();
265                    }
266                    ring_to_shape_path(ring)
267                })
268                .collect::<Vec<_>>()
269        })
270        .collect::<Vec<_>>();
271
272    let fill_rule = if winding_order == Some(WindingOrder::CounterClockwise) {
273        FillRule::Positive
274    } else {
275        FillRule::Negative
276    };
277
278    let shapes = FloatOverlay::with_subj(&subject).overlay(OverlayRule::Subject, fill_rule);
279    multi_polygon_from_shapes(shapes)
280}

282impl<T: BoolOpsNum> BooleanOps for Polygon<T> {
283    type Scalar = T;
284
285    fn rings(&self) -> impl Iterator<Item = &LineString<Self::Scalar>> {
286        std::iter::once(self.exterior()).chain(self.interiors())
287    }
288}

290impl<T: BoolOpsNum> BooleanOps for MultiPolygon<T> {
