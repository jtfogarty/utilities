4use std::fmt;
5
6#[derive(Debug, Clone, PartialEq)]
7pub enum InvalidLine {
8    /// A valid [`Line`] must have at least 2 distinct points to be valid - it must have a non-zero length.
9    IdenticalCoords,
10
11    /// A valid [`Line`] must have finite coordinates.
12    NonFiniteCoord(CoordIndex),
13}
14
15impl fmt::Display for InvalidLine {
16    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
17        match self {
18            InvalidLine::IdenticalCoords => write!(f, "line has identical coordinates"),
19            InvalidLine::NonFiniteCoord(idx) => write!(
20                f,
21                "{} coordinate is non-finite",
22                if idx.0 == 0 { "start" } else { "end" }
23            ),
24        }
25    }
26}
27
28impl std::error::Error for InvalidLine {}
29
30impl<F: GeoFloat> Validation for Line<F> {
31    type Error = InvalidLine;
32
33    fn visit_validation<T>(
34        &self,
35        mut handle_validation_error: Box<dyn FnMut(Self::Error) -> Result<(), T> + '_>,
36    ) -> Result<(), T> {
37        if utils::check_coord_is_not_finite(&self.start) {
38            handle_validation_error(InvalidLine::NonFiniteCoord(CoordIndex(0)))?;
39        }
40        if utils::check_coord_is_not_finite(&self.end) {
41            handle_validation_error(InvalidLine::NonFiniteCoord(CoordIndex(1)))?;
42        }
43        if self.start == self.end {
44            handle_validation_error(InvalidLine::IdenticalCoords)?
45        }
46        Ok(())
47    }
48}
