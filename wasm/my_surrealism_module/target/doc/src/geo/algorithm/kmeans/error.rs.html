4/// Errors that can occur during k-means++ initialisation
5#[derive(Debug, Clone, PartialEq)]
6pub enum KMeansInitError {
7    /// Input data contains `NaN` coordinates
8    ///
9    /// This typically indicates invalid or corrupted input data. Check your point coordinates
10    /// for NaN values before running k-means.
11    NaNCoordinate,
12
13    /// Input data contains infinite coordinates
14    ///
15    /// This typically indicates invalid or corrupted input data. Check your point coordinates
16    /// for infinite values before running k-means.
17    InfiniteCoordinate,
18
19    /// All points are at identical locations, making clustering impossible
20    ///
21    /// When all points have the same coordinates, k-means++ cannot select diverse initial
22    /// centroids because all distances are zero. This makes the weighted sampling distribution
23    /// degenerate.
24    ///
25    /// Consider:
26    /// - Checking if your data has been properly loaded
27    /// - Adding small random perturbations to break ties if appropriate.
28    DegenerateData,
29
30    /// Failed to create weighted distribution for k-means++ sampling
31    ///
32    /// This error occurs when the random weighted sampling distribution cannot be constructed,
33    /// typically because all weights sum to zero or contain invalid values.
34    WeightedDistributionFailed {
35        /// The underlying error from rand's WeightedIndex
36        error: WeightedError,
37    },
38}
39
40impl std::fmt::Display for KMeansInitError {
41    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
42        match self {
43            KMeansInitError::NaNCoordinate => {
44                write!(f, "Input data contains NaN coordinates")
45            }
46            KMeansInitError::InfiniteCoordinate => {
47                write!(f, "Input data contains infinite coordinates")
48            }
49            KMeansInitError::DegenerateData => {
50                write!(
51                    f,
52                    "All points are at identical locations, making clustering impossible"
53                )
54            }
55            KMeansInitError::WeightedDistributionFailed { error } => {
56                write!(
57                    f,
58                    "Failed to create weighted distribution for k-means++ sampling: {}",
59                    error
60                )
61            }
62        }
63    }
64}

66/// Errors that can occur during k-means clustering
67#[derive(Debug, Clone, PartialEq)]
68pub enum KMeansError<T: GeoFloat> {
69    /// Invalid number of clusters requested
70    InvalidK {
71        /// The requested number of clusters
72        k: usize,
73        /// The number of points in the dataset
74        n: usize,
75    },
76
77    /// An empty cluster was encountered and could not be recovered
78    EmptyCluster {
79        /// The iteration number when the error occurred
80        iteration: usize,
81        /// The cluster ID that became empty
82        cluster_id: usize,
83    },
84
85    /// Initialisation failed due to degenerate or invalid data
86    ///
87    /// Contains specific information about the initialisation failure mode.
88    /// See [`KMeansInitError`] for details on each failure type.
89    InitializationFailed(KMeansInitError),
90
91    /// Maximum iterations reached without convergence
92    ///
93    /// The algorithm completed `iterations` iterations without converging (no assignment
94    /// changes and centroid movement below tolerance). The cluster assignments from the
95    /// final iteration are included and may still be useful.
96    ///
97    /// ## Interpreting the Results
98    ///
99    /// Consider using the partial result if:
100    /// - `max_centroid_shift` is relatively small compared to your data scale
101    /// - `changed_assignments` is a small fraction of total points
102    /// - Visual inspection or validation metrics are acceptable
103    ///
104    /// Consider increasing `max_iter` if:
105    /// - `max_centroid_shift` is large relative to tolerance
106    /// - A significant number of points are still changing clusters
107    /// - You require guaranteed convergence
108    MaxIterationsReached {
109        /// Cluster assignments from the final iteration.
110        ///
111        /// These are valid assignments (each point assigned to nearest centroid)
112        /// but may improve with additional iterations.
113        assignments: Vec<usize>,
114        /// Number of iterations completed (equal to max_iter)
115        iterations: usize,
116        /// Maximum distance any centroid shifted in the final iteration
117        max_centroid_shift: T,
118        /// The tolerance threshold for convergence
119        tolerance: T,
120        /// Number of points that changed clusters in the final iteration
121        changed_assignments: usize,
122    },
123}
124
125impl<T: GeoFloat> std::fmt::Display for KMeansError<T> {
126    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
127        match self {
128            KMeansError::InvalidK { k, n } => {
129                write!(
130                    f,
131                    "Invalid k={}: must be > 0 and <= {} (number of points)",
132                    k, n
133                )
134            }
135            KMeansError::EmptyCluster {
136                iteration,
137                cluster_id,
138            } => {
139                write!(
140                    f,
141                    "Empty cluster {} at iteration {} could not be recovered",
142                    cluster_id, iteration
143                )
144            }
145            KMeansError::InitializationFailed(init_error) => {
146                write!(f, "K-means initialisation failed: {}", init_error)
147            }
148            KMeansError::MaxIterationsReached {
149                iterations,
150                max_centroid_shift,
151                tolerance,
152                changed_assignments,
153                ..
154            } => {
155                let shift_f64 = max_centroid_shift
156                    .to_f64()
157                    .expect("max_centroid_shift must be representable as f64");
158                let tol_f64 = tolerance
159                    .to_f64()
160                    .expect("tolerance must be representable as f64");
161                write!(
162                    f,
163                    "k-means did not converge within {} iterations. \
164                     Final centroid shift: {:.6} (tolerance: {:.6}), \
165                     {} points changed clusters in final iteration. \
166                     Consider increasing max_iter or using the partial result.",
167                    iterations, shift_f64, tol_f64, changed_assignments
168                )
169            }
170        }
171    }
172}
