9/// Returns a new Point using the distance to the existing Point and a bearing for the direction on a geodesic.
10///
11/// This uses the geodesic methods given by [Karney (2013)].
12///
13/// [Karney (2013)]:  https://arxiv.org/pdf/1109.4448.pdf
14pub trait GeodesicDestination<T: CoordNum> {
15    /// Returns a new Point using distance to the existing Point and a bearing for the direction.
16    ///
17    /// # Units
18    ///
19    /// - `bearing`: degrees, zero degrees is north
20    /// - `distance`: meters
21    ///
22    /// # Examples
23    ///
24    /// ```rust
25    /// # #[allow(deprecated)]
26    /// use geo::GeodesicDestination;
27    /// use geo::Point;
28    ///
29    /// // Determine the point 10000 km NE of JFK.
30    /// let jfk = Point::new(-73.78, 40.64);
31    /// let northeast_bearing = 45.0;
32    /// let distance = 10e6;
33    ///
34    /// # #[allow(deprecated)]
35    /// let p_1 = jfk.geodesic_destination(northeast_bearing, distance);
36    /// use approx::assert_relative_eq;
37    /// assert_relative_eq!(p_1.x(), 49.052487092959836);
38    /// assert_relative_eq!(p_1.y(), 32.621100463725796);
39    /// ```
40    fn geodesic_destination(&self, bearing: T, distance: T) -> Point<T>;
41}
42
43#[allow(deprecated)]
44impl GeodesicDestination<f64> for Point<f64> {
45    fn geodesic_destination(&self, bearing: f64, distance: f64) -> Point<f64> {
46        Geodesic.destination(*self, bearing, distance)
47    }
48}

56    #[test]
57    fn returns_a_new_point() {
58        let p_1 = Point::new(9.177789688110352, 48.776781529534965);
59        #[allow(deprecated)]
60        let p_2 = p_1.geodesic_destination(45., 10000.);
61
62        assert_relative_eq!(
63            p_2,
64            Point::new(9.27411867078536, 48.8403266058781),
65            epsilon = 1.0e-6
66        );
67
68        #[allow(deprecated)]
69        let distance = p_1.geodesic_distance(&p_2);
70        assert_relative_eq!(distance, 10000., epsilon = 1.0e-6)
71    }
72
73    #[test]
74    fn bearing_zero_is_north() {
75        let p_1 = Point::new(9.177789688110352, 48.776781529534965);
76        #[allow(deprecated)]
77        let p_2 = p_1.geodesic_destination(0., 1000.);
78        assert_relative_eq!(p_1.x(), p_2.x(), epsilon = 1.0e-6);
79        assert!(p_2.y() > p_1.y())
80    }
81
82    #[test]
83    fn bearing_90_is_east() {
