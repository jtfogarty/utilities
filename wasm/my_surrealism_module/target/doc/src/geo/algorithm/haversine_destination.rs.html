8/// Returns a new Point using the distance to the existing Point and a bearing for the direction
9///
10/// *Note*: this implementation uses a mean earth radius of 6371.088 km, based on the [recommendation of
11/// the IUGG](ftp://athena.fsv.cvut.cz/ZFG/grs80-Moritz.pdf)
12pub trait HaversineDestination<T: CoordFloat> {
13    /// Returns a new Point using distance to the existing Point and a bearing for the direction
14    ///
15    /// # Units
16    ///
17    /// - `bearing`: degrees, zero degrees is north
18    /// - `distance`: meters
19    ///
20    /// # Examples
21    ///
22    /// ```rust
23    /// # #[allow(deprecated)]
24    /// use geo::HaversineDestination;
25    /// use geo::Point;
26    /// use approx::assert_relative_eq;
27    ///
28    /// let p_1 = Point::new(9.177789688110352, 48.776781529534965);
29    /// # #[allow(deprecated)]
30    /// let p_2 = p_1.haversine_destination(45., 10000.);
31    /// assert_relative_eq!(p_2, Point::new(9.274409949623548, 48.84033274015048), epsilon = 1e-6)
32    /// ```
33    fn haversine_destination(&self, bearing: T, distance: T) -> Point<T>;
34}
35
36#[allow(deprecated)]
37impl<T> HaversineDestination<T> for Point<T>
38where
39    T: CoordFloat + FromPrimitive,
40{
41    fn haversine_destination(&self, bearing: T, distance: T) -> Point<T> {
42        Haversine.destination(*self, bearing, distance)
43    }
44}

47mod test {
48    use super::*;
49    #[allow(deprecated)]

53    #[test]
54    fn returns_a_new_point() {
55        let p_1 = Point::new(9.177789688110352, 48.776781529534965);
56        #[allow(deprecated)]
57        let p_2 = p_1.haversine_destination(45., 10000.);
58        assert_relative_eq!(
59            p_2,
60            Point::new(9.274409949623548, 48.84033274015048),
61            epsilon = 1.0e-6
62        );
63        #[allow(deprecated)]
64        let distance = p_1.haversine_distance(&p_2);
65        assert_relative_eq!(distance, 10000., epsilon = 1.0e-6)
66    }
67
68    #[test]
69    fn direct_and_indirect_destinations_are_close() {
70        let p_1 = Point::new(9.177789688110352, 48.776781529534965);
71        #[allow(deprecated)]
72        let p_2 = p_1.haversine_destination(45., 10000.);
73        let square_edge = { pow(10000., 2) / 2f64 }.sqrt();
74        #[allow(deprecated)]
75        let p_3 = p_1.haversine_destination(0., square_edge);
76        #[allow(deprecated)]
77        let p_4 = p_3.haversine_destination(90., square_edge);
78        assert_relative_eq!(p_4, p_2, epsilon = 1.0e-6);
79    }
80
81    #[test]
82    fn bearing_zero_is_north() {
83        let p_1 = Point::new(9.177789688110352, 48.776781529534965);
84        #[allow(deprecated)]
85        let p_2 = p_1.haversine_destination(0., 1000.);
86        assert_relative_eq!(p_1.x(), p_2.x(), epsilon = 1.0e-6);
87        assert!(p_2.y() > p_1.y())
88    }

90    #[test]
91    fn should_wrap_correctly() {
92        let pt1 = Point::new(170.0, -30.0);
93        let pt2 = Point::new(-170.0, -30.0);
94
95        for (start, end) in [(pt1, pt2), (pt2, pt1)] {
96            #[allow(deprecated)]
97            let bearing = start.haversine_bearing(end);
98            #[allow(deprecated)]
99            let results: Vec<_> = (0..8)
100                .map(|n| start.haversine_destination(bearing, n as f64 * 250_000.))
101                .collect();
102            assert!(results.iter().all(|pt| pt.x() >= -180.0 && pt.x() <= 180.0));
103        }
104    }
