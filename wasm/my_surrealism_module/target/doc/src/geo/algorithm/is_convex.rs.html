7/// Predicates to test the convexity of a [ `LineString` ].
8/// A closed `LineString` is said to be _convex_ if it
9/// encloses a [convex set]. It is said to be _strictly
10/// convex_ if in addition, no three consecutive vertices
11/// are collinear. It is _collinear_ if all the vertices lie
12/// on the same line.
13///
14/// # Remarks
15///
16/// - Collinearity does not require that the `LineString`
17///   be closed, but the rest of the predicates do.
18///
19/// - This definition is closely related to the notion
20///   of [convexity of polygons][convex set]. In particular, a
21///   [`Polygon`](crate::Polygon) is convex, if and only if its `exterior` is
22///   convex, and `interiors` is empty.
23///
24/// - The [`ConvexHull`] algorithm always returns a strictly
25///   convex `LineString` unless the input is empty or
26///   collinear. The [`graham_hull`] algorithm provides an
27///   option to include collinear points, producing a
28///   (possibly non-strict) convex `LineString`.
29///
30/// # Edge Cases
31///
32/// - the convexity, and collinearity of an empty
33///   `LineString` is _unspecified_ and must not be relied
34///   upon.
35///
36/// - A closed `LineString` with at most three coordinates
37///   (including the possibly repeated first coordinate) is
38///   both convex and collinear. However, the strict convexity
39///   is _unspecified_ and must not be relied upon.
40///
41/// [convex combination]: //en.wikipedia.org/wiki/Convex_combination
42/// [convex set]: //en.wikipedia.org/wiki/Convex_set
43/// [`ConvexHull`]: crate::ConvexHull
44/// [`graham_hull`]: crate::convex_hull::graham_hull
45pub trait IsConvex {
46    /// Test and get the orientation if the shape is convex.
47    /// Tests for strict convexity if `allow_collinear`, and
48    /// only accepts a specific orientation if provided.
49    ///
50    /// The return value is `None` if either:
51    ///
52    /// 1. the shape is not convex
53    ///
54    /// 1. the shape is not strictly convex, and
55    ///    `allow_collinear` is false
56    ///
57    /// 1. an orientation is specified, and some three
58    ///    consecutive vertices where neither collinear, nor
59    ///    in the specified orientation.
60    ///
61    /// In all other cases, the return value is the
62    /// orientation of the shape, or `Orientation::Collinear`
63    /// if all the vertices are on the same line.
64    ///
65    /// **Note.** This predicate is not equivalent to
66    /// `is_collinear` as this requires that the input is
67    /// closed.
68    fn convex_orientation(
69        &self,
70        allow_collinear: bool,
71        specific_orientation: Option<Orientation>,
72    ) -> Option<Orientation>;
73
74    /// Test if the shape is convex.
75    fn is_convex(&self) -> bool {
76        self.convex_orientation(true, None).is_some()
77    }
78
79    /// Test if the shape is convex, and oriented
80    /// counter-clockwise.
81    fn is_ccw_convex(&self) -> bool {
82        self.convex_orientation(true, Some(Orientation::CounterClockwise))
83            .is_some()
84    }
85
86    /// Test if the shape is convex, and oriented clockwise.
87    fn is_cw_convex(&self) -> bool {
88        self.convex_orientation(true, Some(Orientation::Clockwise))
89            .is_some()
90    }
91
92    /// Test if the shape is strictly convex.
93    fn is_strictly_convex(&self) -> bool {
94        self.convex_orientation(false, None).is_some()
95    }
96
97    /// Test if the shape is strictly convex, and oriented
98    /// counter-clockwise.
99    fn is_strictly_ccw_convex(&self) -> bool {
100        self.convex_orientation(false, Some(Orientation::CounterClockwise))
101            == Some(Orientation::CounterClockwise)
102    }
103
104    /// Test if the shape is strictly convex, and oriented
105    /// clockwise.
106    fn is_strictly_cw_convex(&self) -> bool {
107        self.convex_orientation(false, Some(Orientation::Clockwise)) == Some(Orientation::Clockwise)
108    }
109
110    /// Test if the shape lies on a line.
111    fn is_collinear(&self) -> bool;
112}

114impl<T: GeoNum> IsConvex for LineString<T> {
115    fn convex_orientation(
116        &self,
117        allow_collinear: bool,
118        specific_orientation: Option<Orientation>,
119    ) -> Option<Orientation> {
120        if !self.is_closed() || self.0.is_empty() {
121            None
122        } else {
123            // Use sign flip algorithm which detects both non-convexity and self-intersections
124            if is_convex_sign_flips(&self.0) {
125                // If sign flips indicate convex, determine orientation
126                is_convex_shaped(&self.0[1..], allow_collinear, specific_orientation)
127            } else {
128                None
129            }
130        }
131    }
132
133    fn is_collinear(&self) -> bool {
134        self.0.is_empty()
135            || is_convex_shaped(&self.0[1..], true, Some(Orientation::Collinear)).is_some()
136    }
137
138    /// Test if the shape is convex.
139    fn is_convex(&self) -> bool {
140        is_convex_sign_flips(&self.0)
141    }
142}
143/// Check if a LineString is convex using the sign flip algorithm.
144///
145/// This implementation is based on the algorithm described at:
146/// https://math.stackexchange.com/questions/1743995/determine-whether-a-polygon-is-convex-based-on-its-vertices/1745427#1745427
147///
148/// # Algorithm Overview
149///
150/// The sign flip algorithm works by:
151/// 1. Computing edge vectors between consecutive vertices
152/// 2. Tracking sign changes in the x and y components of these vectors
153/// 3. Checking orientation consistency using cross product
154/// 4. A convex polygon has exactly 2 sign flips along each axis
155///
156/// This approach detects both:
157/// - Orientation changes (concave vertices)
158/// - Self-intersections (star polygons)
159///
160/// # Time Complexity
161///
162/// - **Time**: O(n) where n is the number of vertices
163/// - **Early exit**: Returns false immediately when >2 flips are detected or orientation changes
164///
165///
166/// # Edge Cases
167///
168/// - Empty polygons (n = 0): Considered convex
169/// - Single points (n = 1): Considered convex
170/// - Line segments (n = 3): Considered convex
171/// - Triangles (n = 4): Special handling with relaxed flip constraints
172/// - Regular polygons (n ≥ 4): Standard 2-flip rule applies
173/// - Zero-length edges: Ignored in sign flip counting
174/// - Collinear vertices: Detected via cross product consistency
175///
176/// # Examples
177///
178/// ```text
179/// Convex square: [(0,0), (1,0), (1,1), (0,1)] → x_flips=2, y_flips=2 → true
180/// Star polygon: [(0,0), (10,0), (7,-1), (4,5), (6,5), (3,-1)] → >2 flips → false
181/// Triangle: [(0,0), (1,0), (0,1)] → orientation consistent → true
182/// ```
183fn is_convex_sign_flips<T: GeoNum>(coords: &[Coord<T>]) -> bool {
184    // Convexity only makes sense for closed shapes
185    if coords.first() != coords.last() {
186        return false;
187    }
188    let n = coords.len();
189
190    // Edge case 1: Handle degenerate cases first
191    // // These cases are considered convex by definition
192    if n <= 1 {
193        return true; // Empty or single point
194    }
195    if n == 2 {
196        return true; // Single point (with closing duplicate)
197    }
198    // For actual polygons with 3+ vertices (4+ coords including closing), use sign flip algorithm
199    let mut w_orientation: Option<Orientation> = None; // Track orientation consistency via robust predicate
200
201    // Sign flip counters for x-axis
202    let mut x_sign = 0i8; // Current sign: -1 (negative), 0 (zero), 1 (positive)
203    let mut x_first_sign = 0i8; // First non-zero sign encountered
204    let mut x_flips = 0; // Number of sign changes
205
206    // Sign flip counters for y-axis
207    let mut y_sign = 0i8;
208    let mut y_first_sign = 0i8;
209    let mut y_flips = 0;
210
211    // The last vertex is the repeated closing vertex, so we use n - 1
212    let vertices = &coords[..n - 1];
213    let vertex_count = vertices.len();
214
215    // Initialize sliding window: start with the last two actual vertices
216    // This allows us to compute edge vectors for the wraparound case
217    let mut prev = vertices[vertex_count - 1];
218    let mut curr = vertices[0];
219    // Main algorithm loop: process each vertex and its incident edges
220    // Time complexity: O(n): single pass through all vertices
221    // Use iterator with cycle() to handle wraparound
222    for next in vertices.iter().cycle().skip(1).take(vertex_count) {
223        // Compute edge vectors for sign flip tracking
224        let ax = next.x - curr.x; // "after" edge: curr → next
225        let ay = next.y - curr.y;
226
227        // Track sign flips in x direction
228        // The sign flip algorithm principle: a convex polygon traversed in order
229        // will have edge vectors that change sign exactly twice per axis
230
231        match ax.partial_cmp(&T::zero()) {
232            Some(Ordering::Greater) => {
233                match x_sign.cmp(&0) {
234                    Ordering::Equal => x_first_sign = 1,
235                    Ordering::Less => x_flips += 1,
236                    _ => {}
237                }
238                x_sign = 1;
239            }
240            Some(Ordering::Less) => {
241                match x_sign.cmp(&0) {
242                    Ordering::Equal => x_first_sign = -1,
243                    Ordering::Greater => x_flips += 1,
244                    _ => {}
245                }
246                x_sign = -1;
247            }
248            _ => {
249                // ax is 0 or NaN, which doesn't change sign state
250            }
251        }
252
253        // Early exit optimization: if we've seen more than 2 flips, it's not convex
254        if x_flips > 2 {
255            return false;
256        }
257
258        // Track sign flips in y direction (same logic as x)
259        match ay.partial_cmp(&T::zero()) {
260            Some(Ordering::Greater) => {
261                match y_sign.cmp(&0) {
262                    Ordering::Equal => y_first_sign = 1,
263                    Ordering::Less => y_flips += 1,
264                    _ => {}
265                }
266                y_sign = 1;
267            }
268            Some(Ordering::Less) => {
269                match y_sign.cmp(&0) {
270                    Ordering::Equal => y_first_sign = -1,
271                    Ordering::Greater => y_flips += 1,
272                    _ => {}
273                }
274                y_sign = -1;
275            }
276            _ => {
277                // ay is 0 or NaN, which doesn't change sign state
278            }
279        }
280
281        // Early exit optimization: if we've seen more than 2 flips, it's not convex
282        if y_flips > 2 {
283            return false;
284        }
285        // Check orientation consistency using robust orientation predicate
286        // This detects when the polygon "turns" in different directions
287        // (indicating concavity or self-intersection)
288        let current_orientation = T::Ker::orient2d(prev, curr, *next);
289
290        // Skip collinear points for orientation consistency check
291        if current_orientation != Orientation::Collinear {
292            if let Some(established_orientation) = w_orientation {
293                if current_orientation != established_orientation {
294                    // Orientation change detected - not convex
295                    return false;
296                }
297            } else {
298                w_orientation = Some(current_orientation); // Establish initial orientation
299            }
300        }
301        // Advance sliding window to next vertex
302        prev = curr;
303        curr = *next;
304    }
305
306    // Check final/wraparound sign flips
307    // We need to check if the final sign differs from the first sign
308    // to account for the wraparound from last vertex back to first
309    if x_sign != 0 && x_first_sign != 0 && x_sign != x_first_sign {
310        x_flips += 1;
311    }
312    if y_sign != 0 && y_first_sign != 0 && y_sign != y_first_sign {
313        y_flips += 1;
314    }
315
316    // Final convexity check based on sign flip count
317    // Mathematical principle: convex polygons have exactly 2 sign flips per axis
318    // This is because as you traverse a convex polygon, you go:
319    // - Right → Up → Left → Down (or similar pattern) = 2 flips per axis
320    if vertex_count == 3 {
321        // Edge case 2: Triangle special handling
322        // Triangles can have irregular sign flip patterns but are always convex
323        // if orientation is consistent (already checked above with robust predicate)
324        w_orientation.is_some() || (x_flips <= 2 && y_flips <= 2)
325    } else {
326        // Standard case: Regular polygons with 4+ vertices
327        // Must have exactly 2 sign flips per axis for convexity
328        x_flips == 2 && y_flips == 2
329    }
330}

332/// A utility that tests convexity of a sequence of
333/// coordinates. It verifies that for all `0 <= i < n`, the
334/// vertices at positions `i`, `i+1`, `i+2` (mod `n`) have
335/// the same orientation, optionally accepting collinear
336/// triplets, and expecting a specific orientation. The
337/// output is `None` or the only non-collinear orientation,
338/// unless everything is collinear.
339fn is_convex_shaped<T>(
340    coords: &[Coord<T>],
341    allow_collinear: bool,
342    specific_orientation: Option<Orientation>,
343) -> Option<Orientation>
344where
345    T: GeoNum,
346{
347    if coords.is_empty() {
348        // empty here means original linestring had 1 point
349        return Some(Orientation::Collinear);
350    }
351
352    // drop second of adjacent identical points
353    let c0 = coords[0];
354    let mut coords: VecDeque<Coord<T>> = coords
355        .windows(2)
356        .filter_map(|w| if w[0] == w[1] { None } else { Some(w[1]) })
357        .collect();
358    coords.push_front(c0);
359
360    let n = coords.len();
361
362    let orientation_at = |i: usize| {
363        let coord = coords[i];
364        let next = coords[(i + 1) % n];
365        let nnext = coords[(i + 2) % n];
366        (i, T::Ker::orient2d(coord, next, nnext))
367    };
368
369    let find_first_non_collinear = (0..n).map(orientation_at).find_map(|(i, orientation)| {
370        match orientation {
371            Orientation::Collinear => {
372                // If collinear accepted, we skip, otherwise
373                // stop.
374                if allow_collinear {
375                    None
376                } else {
377                    Some((i, orientation))
378                }
379            }
380            _ => Some((i, orientation)),
381        }
382    });
383
384    let (i, first_non_collinear) = if let Some((i, orientation)) = find_first_non_collinear {
385        match orientation {
386            Orientation::Collinear => {
387                // Only happens if !allow_collinear
388                assert!(!allow_collinear);
389                return None;
390            }
391            _ => (i, orientation),
392        }
393    } else {
394        // Empty or everything collinear, and allowed.
395        return Some(Orientation::Collinear);
396    };
397
398    // If a specific orientation is expected, accept only that.
399    if let Some(req_orientation) = specific_orientation {
400        if req_orientation != first_non_collinear {
401            return None;
402        }
403    }
404
405    // Now we have a fixed orientation expected at the rest
406    // of the coords. Loop to check everything matches it.
407    if ((i + 1)..n)
408        .map(orientation_at)
409        .all(|(_, orientation)| match orientation {
410            Orientation::Collinear => allow_collinear,
411            orientation => orientation == first_non_collinear,
412        })
413    {
414        Some(first_non_collinear)
415    } else {
416        None
417    }
418}

427    #[test]
428    fn test_corner_cases() {
429        // This is just tested to ensure there is no panic
430        // due to out-of-index access
431        let empty: LineString = line_string!();
432        assert!(empty.is_collinear());
433        assert!(empty.is_convex());
434        assert!(!empty.is_strictly_ccw_convex());
435
436        let one = line_string![(x: 0., y: 0.)];
437        assert!(one.is_collinear());
438        assert!(one.is_convex());
439        assert!(one.is_cw_convex());
440        assert!(one.is_ccw_convex());
441        assert!(one.is_strictly_convex());
442        assert!(!one.is_strictly_ccw_convex());
443        assert!(!one.is_strictly_cw_convex());
444
445        let one_rep = line_string![(x: 0, y: 0), (x: 0, y: 0)];
446        assert!(one_rep.is_collinear());
447        assert!(one_rep.is_convex());
448        assert!(one_rep.is_cw_convex());
449        assert!(one_rep.is_ccw_convex());
450        assert!(!one_rep.is_strictly_convex());
451        assert!(!one_rep.is_strictly_ccw_convex());
452        assert!(!one_rep.is_strictly_cw_convex());
453
454        let mut two = line_string![(x: 0, y: 0), (x: 1, y: 1)];
455        assert!(two.is_collinear());
456        assert!(!two.is_convex());
457
458        two.close();
459        assert!(two.is_cw_convex());
460        assert!(two.is_ccw_convex());
461        assert!(!two.is_strictly_convex());
462        assert!(!two.is_strictly_ccw_convex());
463        assert!(!two.is_strictly_cw_convex());
464    }
