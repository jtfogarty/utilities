13/// Returns a polygon which covers a geometry. Unlike convex hulls, which also cover
14/// their geometry, a concave hull does so while trying to further minimize its area by
15/// constructing edges such that the exterior of the polygon incorporates points that would
16/// be interior points in a convex hull.
17///
18/// This implementation is a port of Volodymyr Agafonkin's <https://github.com/mapbox/concaveman> which is based on ideas from
19/// the paper [A New Concave Hull Algorithm and Concaveness Measure for n-dimensional Datasets, 2012](https://jise.iis.sinica.edu.tw/JISESearch/fullText?pId=245&code=5A9B97538372AA1).
20///
21/// # Arguments
22/// * `concave_hull_options` - Optional configuration for the concave hull algorithm:
23///   - `concavity` - A relative measure of how concave the hull should be. Lower values result in a more
24///     concave hull. Infinity would result in a convex hull. 2.0 results in a relatively detailed shape. (default: 2.0)
25///   - `length_threshold` - The minimum length of constituent hull edges. Edges shorter than this will not be
26///     drilled down any further. Set to 0.0 for no threshold. (default: 0.0)
27///
28/// # Returns
29/// * A `Polygon` representing a concave hull of the geometry.
30///
31/// # Algorithm
32///
33/// The algorithm works as follows:
34/// 1. Start with the convex hull of all input points as the initial boundary
35/// 2. For each edge of the hull boundary:
36///    - If the edge length exceeds the `length_threshold`, attempt to "drill inward"
37///    - Search for the closest interior point within `max_length = edge_length / concavity` from edge
38///    - Verify the candidate is closer to this edge than adjacent hull edges
39///    - Verify that connecting to this point won't cause intersections with existing hull edges
40///    - Continue searching until a valid candidate is found or no more points are within the `max_length`
41///    - Verify that adding this point won't cause any previously checked interior points to be excluded from the hull and if one is excluded use that point as the candidate
42/// 3. If a valid candidate is found:
43///    - Create two new edges: start→candidate and candidate→end
44///    - Verify at least one of the new edges is less than `max_length`
45///    - Replace the original edge with the two new edges
46///    - Remove the candidate point from further selection
47///    - Add the new edges to the boundary and processing queue for potential further drilling
48/// 4. Repeat until no more edges can be drilled
49///
50/// # Examples
51/// ```
52/// use geo::{polygon, MultiPoint};
53/// use geo::ConcaveHull;
54///
55/// // a collection of points
56/// let points: MultiPoint<_> = vec![
57///     (0.0, 0.0),
58///     (3.0, 0.0),
59///     (1.0, 2.0),
60///     (0.0, 4.0),
61/// ].into();
62///
63/// let correct_hull = polygon![
64///     (x: 3.0, y: 0.0),
65///     (x: 1.0, y: 2.0),
66///     (x: 0.0, y: 4.0),
67///     (x: 0.0, y: 0.0),
68///     (x: 3.0, y: 0.0),
69/// ];
70///
71/// let hull = points.concave_hull();
72/// assert_eq!(hull, correct_hull);
73///
74/// ```
75/// `ConcaveHull` can also be used with custom options.
76/// ```
77/// use geo::{polygon, ConcaveHull, MultiPoint};
78/// use geo::concave_hull::ConcaveHullOptions;
79///
80/// // a collection of points
81/// let points: MultiPoint<_> = vec![
82///     (0.0, 0.0),
83///     (2.0, 0.0),
84///     (1.5, 1.0),
85///     (2.0, 2.0),
86///     (0.0, 2.0),
87/// ].into();
88///
89/// let correct_hull = polygon![
90///     (x: 2.0, y: 0.0),
91///     (x: 1.5, y: 1.0),
92///     (x: 2.0, y: 2.0),
93///     (x: 0.0, y: 2.0),
94///     (x: 0.0, y: 0.0),
95///     (x: 2.0, y: 0.0),
96/// ];
97///
98/// let hull = points.concave_hull_with_options(ConcaveHullOptions {
99///     concavity: 1.0,
100///     length_threshold: 0.0,
101/// });
102/// assert_eq!(hull, correct_hull);
103/// ```
104pub trait ConcaveHull {
105    type Scalar: GeoFloat + RTreeNum;
106    /// Create a concave hull around the geometry set.
107    ///
108    /// See the [module-level documentation](self) for details on the algorithm.
109    fn concave_hull(&self) -> Polygon<Self::Scalar> {
110        self.concave_hull_with_options(ConcaveHullOptions::default())
111    }
112
113    /// Create a concave hull around the geometry set with specified options.
114    ///
115    /// See the [module-level documentation](self) for details on the algorithm and parameters.
116    fn concave_hull_with_options(
117        &self,
118        concave_hull_options: ConcaveHullOptions<Self::Scalar>,
119    ) -> Polygon<Self::Scalar>;
120}

313fn line_to_bbox_distance<T>(line: &Line<T>, bbox: &AABB<Coord<T>>) -> T
314where
315    T: GeoFloat + RTreeNum,
316{
317    // Calculate the euclidean distance from a line to a bounding box.
318    // This function is equivalent to `Euclidean.distance` between a `Rect` and a `Line`,
319    // but is optimized for the R-tree depth-first search used here.
320    // Since lines are likely to be intersecting or contained within the bounding box, resulting
321    // in a distance of zero, calculate each seperately and return early if zero is found.
322
323    // If either line endpoint is contained within the bbox, distance is zero
324    if bbox.contains_point(&line.start) || bbox.contains_point(&line.end) {
325        return T::zero();
326    }
327
328    // If any distances are zero, then return as no further distance calculations needed
329    let c1 = coord! {x: bbox.lower().x, y: bbox.lower().y};
330    let c2 = coord! {x: bbox.lower().x, y: bbox.upper().y};
331    let c3 = coord! {x: bbox.upper().x, y: bbox.upper().y};
332    let c4 = coord! {x: bbox.upper().x, y: bbox.lower().y};
333    let d1 = Euclidean.distance(line, &Line::new(c1, c4));
334    if d1 == T::zero() {
335        return d1;
336    }
337    let d2 = Euclidean.distance(line, &Line::new(c1, c2));
338    if d2 == T::zero() {
339        return d2;
340    }
341    let d3 = Euclidean.distance(line, &Line::new(c4, c3));
342    if d3 == T::zero() {
343        return d3;
344    }
345    let d4 = Euclidean.distance(line, &Line::new(c2, c3));
346    if d4 == T::zero() {
347        return d4;
348    }
349
350    // If the line is contained within or intersecting the bounding box, return the minimum distance
351    partial_min(partial_min(d1, d2), partial_min(d3, d4))
352}

354fn no_hull_intersections<T>(line: &Line<T>, current_hull_tree: &RTree<Line<T>>) -> bool
355where
356    T: GeoFloat + RTreeNum,
357{
358    // Check if the line intersects with any existing hull line
359    // Hull lines which share an endpoint with the line are skipped
360
361    // Create bounding box for the line
362    let min_x = T::min(line.start.x, line.end.x);
363    let max_x = T::max(line.start.x, line.end.x);
364    let min_y = T::min(line.start.y, line.end.y);
365    let max_y = T::max(line.start.y, line.end.y);
366    let bbox = AABB::from_corners(point!([min_x, min_y]), point!([max_x, max_y]));
367
368    // Iterate over all lines in the hull which intersect with the bounding box
369    let hull_lines = current_hull_tree.locate_in_envelope_intersecting(&bbox);
370    for hull_line in hull_lines {
371        // Skip lines which share an endpoint
372        if hull_line.start == line.start
373            || hull_line.start == line.end
374            || hull_line.end == line.start
375            || hull_line.end == line.end
376        {
377            continue;
378        }
379        if line.intersects(hull_line) {
380            return false;
381        }
382    }
383    true
384}

386fn find_excluded_point<T>(
387    start_line: Line<T>,
388    end_line: Line<T>,
389    previously_checked_points: &[Coord<T>],
390) -> Option<Coord<T>>
391where
392    T: GeoFloat,
393{
394    // Check previously checked interior points to see if any would lie outside the hull if the new lines were added.
395    // If so, return that point as the candidate to ensure all interior points remain within the hull.
396
397    // Create a triangle which represents the area which would be excluded from the hull
398    let triangle = Triangle::new(start_line.start, start_line.end, end_line.end);
399    for point in previously_checked_points {
400        // If the point is contained within the triangle, it would be outside the hull and so return it as the candidate
401        if triangle.contains(point) {
402            return Some(*point);
403        }
404    }
405    None
406}

408fn find_candidate<T>(
409    hull_edge: &CurrentHullEdge<T>,
410    max_length: &T,
411    current_hull_edges: &[CurrentHullEdge<T>],
412    interior_points_tree: &RTree<Coord<T>>,
413    current_hull_tree: &RTree<Line<T>>,
414) -> Option<Coord<T>>
415where
416    T: GeoFloat + RTreeNum,
417{
418    let line = hull_edge.line;
419
420    // Initialize priority queue with R-tree root node
421    let mut queue: BinaryHeap<NodeQueueItem<T>> = BinaryHeap::new();
422    queue.push(NodeQueueItem {
423        tree_node: RTreeNodeRef::Parent(interior_points_tree.root()),
424        distance: T::zero(),
425    });
426
427    // Keep track of nearest interior points which failed checks
428    let mut previously_checked_points: Vec<Coord<T>> = Vec::new();
429
430    // Perform depth-first search through the R-tree
431    while let Some(node) = queue.pop() {
432        match node.tree_node {
433            RTreeNodeRef::Parent(parent) => {
434                // Add the children of a parent node to the queue if they are within the max distance
435                for child in parent.children() {
436                    match child {
437                        RTreeNode::Parent(p) => {
438                            let distance = line_to_bbox_distance(&line, &p.envelope());
439                            if distance <= *max_length {
440                                queue.push(NodeQueueItem {
441                                    tree_node: RTreeNodeRef::Parent(p),
442                                    distance,
443                                });
444                            }
445                        }
446                        RTreeNode::Leaf(l) => {
447                            let distance = Euclidean.distance(*l, &line);
448                            if distance <= *max_length {
449                                queue.push(NodeQueueItem {
450                                    tree_node: RTreeNodeRef::Leaf(l),
451                                    distance,
452                                });
453                            }
454                        }
455                    }
456                }
457            }
458            RTreeNodeRef::Leaf(leaf) => {
459                // Check if candidate point is further from adjacent hull lines
460                if node.distance
461                    >= Euclidean.distance(*leaf, &current_hull_edges[hull_edge.prev_i].line)
462                    || node.distance
463                        >= Euclidean.distance(*leaf, &current_hull_edges[hull_edge.next_i].line)
464                {
465                    previously_checked_points.push(*leaf);
466                    continue;
467                }
468
469                let start_line = Line::new(line.start, *leaf);
470                let end_line = Line::new(*leaf, line.end);
471
472                // Check if using candidate point would cause intersections with hull lines
473                if no_hull_intersections(&start_line, current_hull_tree)
474                    && no_hull_intersections(&end_line, current_hull_tree)
475                {
476                    // Check if any of the previously checked interior points would lie outside the hull if the new lines
477                    // were added and use that point as the candidate if so
478                    if let Some(point) =
479                        find_excluded_point(start_line, end_line, &previously_checked_points)
480                    {
481                        return Some(point);
482                    }
483                    return Some(*leaf);
484                }
485                previously_checked_points.push(*leaf);
486            }
487        }
488    }
489    None
490}

492fn order_concave_hull<T>(current_hull_edges: Vec<CurrentHullEdge<T>>) -> LineString<T>
493where
494    T: GeoFloat,
495{
496    // Order the constituent concave hull edges and return as a `LineString`
497    let mut ordered_coords: Vec<Coord<T>> = Vec::with_capacity(current_hull_edges.len() + 1);
498    let mut current_i = 0;
499    ordered_coords.push(current_hull_edges[current_i].line.start);
500
501    for _ in 0..current_hull_edges.len() {
502        let next_i = current_hull_edges[current_i].next_i;
503        let line = current_hull_edges[current_i].line;
504        ordered_coords.push(line.end);
505        current_i = next_i;
506    }
507    LineString::from(ordered_coords)
508}

510fn remove_interior_point<T>(coord: &Coord<T>, tree: &mut RTree<Coord<T>>)
511where
512    T: GeoFloat + RTreeNum,
513{
514    // Remove all instances of the point from the R-tree
515    let n = tree.nearest_neighbors(coord).len();
516    for _ in 0..n {
517        tree.remove(coord);
518    }
519}

521fn concave_hull_with_options<T>(
522    coords: &mut [Coord<T>],
523    concave_hull_options: ConcaveHullOptions<T>,
524) -> Polygon<T>
525where
526    T: GeoFloat + RTreeNum,
527{
528    debug_assert!(concave_hull_options.concavity >= T::zero());
529    // Ensure concavity is non-negative
530    let concavity: T = T::max(T::zero(), concave_hull_options.concavity);
531
532    // Compute initial convex hull
533    let convex_hull = qhull::quick_hull(coords);
534
535    // Return convex hull if less than 4 points
536    if coords.len() < 4 {
537        return Polygon::new(convex_hull, vec![]);
538    }
539
540    // Build R-trees for interior points and hull lines
541    let mut interior_points_tree: RTree<Coord<T>> = RTree::bulk_load(coords.to_owned());
542    let mut current_hull_tree: RTree<Line<T>> = RTree::bulk_load(convex_hull.lines().collect());
543
544    let mut edge_queue: VecDeque<CurrentHullEdge<T>> = VecDeque::new();
545    let mut current_hull_edges: Vec<CurrentHullEdge<T>> = Vec::new();
546
547    // Populate edge queue and current hull edges with convex hull edges
548    let hull_length = convex_hull.lines().len();
549    for (i, line) in convex_hull.lines().enumerate() {
550        let edge = CurrentHullEdge {
551            line,
552            i,
553            prev_i: if i == 0 { hull_length - 1 } else { i - 1 },
554            next_i: if i == hull_length - 1 { 0 } else { i + 1 },
555        };
556        current_hull_edges.push(edge.clone());
557        edge_queue.push_back(edge);
558
559        // Remove hull points from interior points
560        if i == 0 {
561            remove_interior_point(&line.start, &mut interior_points_tree);
562        }
563        remove_interior_point(&line.end, &mut interior_points_tree);
564    }
565
566    while let Some(hull_edge) = edge_queue.pop_front() {
567        let line = hull_edge.line;
568        let length = Euclidean.length(&line);
569
570        // Only consider drilling down if line length exceeds threshold
571        if length > concave_hull_options.length_threshold {
572            // Calculate maximum length for new hull edges
573            let max_length = length / concavity;
574
575            if let Some(candidate_point) = find_candidate(
576                &hull_edge,
577                &max_length,
578                &current_hull_edges,
579                &interior_points_tree,
580                &current_hull_tree,
581            ) {
582                // Create new hull lines from start→candidate and candidate→end
583                let start_line = Line::new(line.start, candidate_point);
584                let end_line = Line::new(candidate_point, line.end);
585
586                // Verify at least one of the new edges is less than max_length
587                if partial_min(Euclidean.length(&start_line), Euclidean.length(&end_line))
588                    < max_length
589                {
590                    // Remove candidate point from interior points
591                    remove_interior_point(&candidate_point, &mut interior_points_tree);
592
593                    // Update current hull tree
594                    current_hull_tree.remove(&line);
595                    current_hull_tree.insert(start_line);
596                    current_hull_tree.insert(end_line);
597
598                    // Set end edges' index as the length of current hull
599                    let end_edge_i = current_hull_edges.len();
600
601                    // Set new hull edges with indexes of adjacent edges
602                    let start_hull_edge = CurrentHullEdge {
603                        line: start_line,
604                        i: hull_edge.i,
605                        prev_i: hull_edge.prev_i,
606                        next_i: end_edge_i,
607                    };
608                    let end_hull_edge = CurrentHullEdge {
609                        line: end_line,
610                        i: end_edge_i,
611                        prev_i: hull_edge.i,
612                        next_i: hull_edge.next_i,
613                    };
614
615                    // Replace the current edge with the new start edge
616                    current_hull_edges[hull_edge.i] = start_hull_edge.clone();
617
618                    // Push new end edge to current hull edges
619                    current_hull_edges.push(end_hull_edge.clone());
620
621                    // Push both new edges to queue
622                    edge_queue.push_back(start_hull_edge);
623                    edge_queue.push_back(end_hull_edge);
624
625                    continue;
626                }
627            }
628        }
629    }
630    Polygon::new(order_concave_hull(current_hull_edges), vec![])
631}
