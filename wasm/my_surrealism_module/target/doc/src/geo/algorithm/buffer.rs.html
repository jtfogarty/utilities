25/// Create a new geometry whose boundary is offset the specified distance from the input.
26///
27/// The buffer operation creates polygons which represent all points within a specified distance from the input geometry.
28/// For example, buffering a point creates a circle, buffering a line creates a "pill" shape, and buffering a polygon
29/// creates a larger polygon (or a smaller one if a negative distance is requested).
30///
31/// # Examples
32///
33/// Basic buffering with default style:
34/// ```
35/// use geo::{Point, Buffer, MultiPolygon};
36///
37/// let point = Point::new(0.0, 0.0);
38/// // Creates an approximated circle with radius 2.0
39/// let buffered: MultiPolygon = point.buffer(2.0);
40/// ```
41///
42/// Default buffering rounds the point where geometries meet and where lines end.
43/// Use a custom style for more control:
44/// ```
45/// use geo::{wkt, MultiPolygon, Buffer};
46/// use geo::algorithm::buffer::{BufferStyle, LineCap, LineJoin};
47///
48/// let lines = wkt! {            
49///     MULTILINESTRING(
50///         (0. 0.,2. 0.,1. 2.),
51///         (0. -1.,2. 1.,3. 3.)
52///     )
53/// };
54/// let style = BufferStyle::new(0.5)
55///     .line_cap(LineCap::Square)
56///     .line_join(LineJoin::Miter(1.0));
57/// let buffered: MultiPolygon = lines.buffer_with_style(style);
58/// ```
59pub trait Buffer {
60    type Scalar: BoolOpsNum + 'static;
61
62    /// Create a new geometry whose boundary is offset the specified distance from the input.
63    ///
64    /// By default, buffering uses rounded joins and end caps.
65    /// See [`buffer_with_style`](Self::buffer_with_style) for more control.
66    ///
67    /// # Arguments
68    ///
69    /// * `distance` - The buffer distance. Positive values create an outward buffer,
70    ///   negative values create an inward buffer (for polygons).
71    ///
72    /// # Returns
73    ///
74    /// A `MultiPolygon` representing the buffered geometry.
75    ///
76    /// # Examples
77    ///
78    /// ```
79    /// # use approx::assert_relative_eq;
80    /// use geo::{Point, Buffer, MultiPolygon};
81    ///
82    /// let point = Point::new(5.0, 5.0);
83    /// let circle: MultiPolygon = point.buffer(3.0);
84    ///
85    /// use geo::algorithm::Area;
86    /// // Creates an approximately circular polygon centered at (5, 5) with radius 30
87    /// assert_relative_eq!(circle.unsigned_area(), std::f64::consts::PI * 3.0f64.powi(2), epsilon = 2e-1);
88    /// ```
89    fn buffer(&self, distance: Self::Scalar) -> MultiPolygon<Self::Scalar> {
90        let default_style = BufferStyle::new(distance);
91        self.buffer_with_style(default_style)
92    }
93
94    /// Create a new geometry whose boundary is offset the specified distance from the input using
95    /// the specific styling options where lines intersect (line joins) and end (end caps).
96    /// For default (rounded) styling, see [`buffer`](Self::buffer).
97    ///
98    /// This method allows control over the buffer appearance through the `BufferStyle` parameter.
99    ///
100    /// # Arguments
101    ///
102    /// * `style` - A `BufferStyle` that specifies the distance, line caps, and line joins.
103    ///
104    /// # Returns
105    ///
106    /// A `MultiPolygon` representing the buffered geometry.
107    ///
108    /// # Examples
109    ///
110    /// ```
111    /// use geo::{wkt, Buffer};
112    /// use geo::algorithm::buffer::{BufferStyle, LineCap, LineJoin};
113    ///
114    /// let line_string = wkt! { LINESTRING (0. 0.,2. 0.,1. 2.) };
115    /// let style = BufferStyle::new(1.5)
116    ///     .line_cap(LineCap::Square)
117    ///     .line_join(LineJoin::Bevel);
118    /// let buffered = line_string.buffer_with_style(style);
119    /// ```
120    fn buffer_with_style(&self, style: BufferStyle<Self::Scalar>) -> MultiPolygon<Self::Scalar>;
121}

123/// Configuration for buffer styling operations.
124///
125/// `BufferStyle` controls how the buffer operation creates the resulting polygon,
126/// including the distance from the original geometry and how line segments are joined or capped.
127///
128/// # Examples
129///
130/// ```
131/// use geo::algorithm::buffer::{BufferStyle, LineCap, LineJoin};
132///
133/// // Default round style
134/// let round_style = BufferStyle::new(2.0);
135///
136/// // Square caps with mitered joins
137/// let square_style = BufferStyle::new(2.0)
138///     .line_cap(LineCap::Square)
139///     .line_join(LineJoin::Miter(1.0));
140/// ```
141pub struct BufferStyle<T: BoolOpsNum> {
142    distance: T,
143    line_cap: LineCap<BoolOpsCoord<T>, T>,
144    line_join: LineJoin<T>,
145}

157impl<T: BoolOpsNum> BufferStyle<T> {
158    /// Creates a new `BufferStyle` with the specified distance and default round caps/joins.
159    ///
160    /// # Arguments
161    ///
162    /// * `distance` - The buffer distance from the original geometry
163    ///
164    /// # Examples
165    ///
166    /// ```
167    /// use geo::algorithm::buffer::BufferStyle;
168    ///
169    /// let style = BufferStyle::new(1.5);
170    /// ```
171    pub fn new(distance: T) -> Self {
172        BufferStyle {
173            distance,
174            line_cap: LineCap::Round(Self::default_join_angle()),
175            line_join: LineJoin::Round(Self::default_join_angle()),
176        }
177    }
178
179    /// Sets how two edges of a geometry should meet at vertices.
180    ///
181    /// Note: This has no effect on point geometries.
182    ///
183    /// # Arguments
184    ///
185    /// * `line_join` - The join style: `Round`, `Miter`, or `Bevel`
186    ///
187    /// # Examples
188    ///
189    /// ```
190    /// use geo::algorithm::buffer::{Buffer, BufferStyle, LineJoin};
191    ///
192    /// let style = BufferStyle::new(2.0).line_join(LineJoin::Miter(1.0));
193    /// ```
194    pub fn line_join(mut self, line_join: LineJoin<T>) -> Self {
195        self.line_join = line_join;
196        self
197    }
198
199    /// Sets how the ends of linear geometries and points should be capped.
200    ///
201    /// This only affects `Line`, `LineString`, `MultiLineString`, `Point`, and `MultiPoint` geometries.
202    /// Two dimensional geometries, like Polygons, ignore setting.
203    ///
204    /// # Arguments
205    ///
206    /// * `line_cap` - The cap style: `Round`, `Square`, `Butt`, or `Custom`
207    ///
208    /// # Examples
209    ///
210    /// ```
211    /// use geo::algorithm::buffer::{Buffer, BufferStyle, LineCap};
212    ///
213    /// let style = BufferStyle::new(2.0).line_cap(LineCap::Square);
214    /// ```
215    pub fn line_cap(mut self, line_cap: LineCap<BoolOpsCoord<T>, T>) -> Self {
216        self.line_cap = line_cap;
217        self
218    }
219
220    // Annoyingly, i_overlay doesn't implement Clone for LineJoin
221    fn clone_line_join(&self) -> LineJoin<T> {
222        match self.line_join {
223            LineJoin::Bevel => LineJoin::Bevel,
224            LineJoin::Miter(angle) => LineJoin::Miter(angle),
225            LineJoin::Round(angle) => LineJoin::Round(angle),
226        }
227    }
228
229    // Annoyingly, i_overlay doesn't implement Clone for LineCap
230    fn clone_line_cap(&self) -> LineCap<BoolOpsCoord<T>, T> {
231        match &self.line_cap {
232            LineCap::Butt => LineCap::Butt,
233            LineCap::Round(angle) => LineCap::Round(*angle),
234            LineCap::Square => LineCap::Square,
235            LineCap::Custom(points) => LineCap::Custom(points.clone()),
236        }
237    }
238
239    fn default_join_angle() -> T {
240        // This is arbitrary, but the results seem to match pretty closely with JTS's own default.
241        T::from_f32(0.20).expect("valid float constant")
242    }
243
244    // Used by i_overlay for buffering (Multi)Polygons
245    fn outline_style(&self) -> OutlineStyle<T> {
246        OutlineStyle::new(self.distance).line_join(self.clone_line_join())
247    }
248
249    // Used by i_overlay for buffering (Multi)LineStrings
250    fn stroke_style(&self) -> StrokeStyle<BoolOpsCoord<T>, T> {
251        // "Buffer width" is like radius, whereas "stroke width" is more like diameter, so double the
252        // "stroke width" to reconcile semantics with "buffer width".
253        let two = T::one() + T::one();
254        StrokeStyle::new(self.distance * two)
255            .line_join(self.clone_line_join())
256            .end_cap(self.clone_line_cap())
257            .start_cap(self.clone_line_cap())
258    }
259}

268impl<F: BoolOpsNum + 'static> Buffer for Point<F> {
269    type Scalar = F;
270    fn buffer_with_style(&self, style: BufferStyle<Self::Scalar>) -> MultiPolygon<Self::Scalar> {
271        if style.distance <= F::zero() {
272            return MultiPolygon::empty();
273        }
274
275        match style.line_cap {
276            // Without length, i_overlay can't sensible reason about orientation of template points.
277            LineCap::Custom(_) => MultiPolygon::empty(),
278            LineCap::Butt => MultiPolygon::empty(),
279            LineCap::Square => {
280                let a = coord!(x: self.x() - style.distance, y: self.y() - style.distance);
281                let b = coord!(x: self.x() + style.distance, y: self.y() + style.distance);
282                MultiPolygon(vec![Rect::new(a, b).to_polygon()])
283            }
284            LineCap::Round(angle) => {
285                // approximate a circle
286                let num_segments = (2.0 * std::f64::consts::PI / angle.to_f64()).ceil() as usize;
287                let center = self.0;
288                let radius = style.distance;
289
290                let mut coords = Vec::with_capacity(num_segments + 1);
291                for i in 0..num_segments {
292                    let angle = F::from_f64(
293                        2.0 * std::f64::consts::PI / num_segments as f64 * i as f64
294                            + std::f64::consts::PI,
295                    )
296                    .expect("valid float constant");
297                    let x = center.x + radius * Float::cos(angle);
298                    let y = center.y + radius * Float::sin(angle);
299                    coords.push(Coord { x, y });
300                }
301                // Close the ring
302                coords.push(coords[0]);
303
304                let polygon = Polygon::new(LineString::new(coords), vec![]);
305                MultiPolygon::new(vec![polygon])
306            }
307        }
308    }
309}

311impl<F: BoolOpsNum + 'static> Buffer for MultiPoint<F> {
312    type Scalar = F;
313    fn buffer_with_style(&self, style: BufferStyle<Self::Scalar>) -> MultiPolygon<Self::Scalar> {
314        if style.distance <= F::zero() {
315            return MultiPolygon::empty();
316        }
317
318        if self.0.is_empty() {
319            return MultiPolygon::empty();
320        }
321
322        // Buffer each point individually
323        let buffered_points: Vec<MultiPolygon<F>> = self
324            .0
325            .iter()
326            .map(|point| point.buffer_with_style(style.clone()))
327            .collect();
328
329        // Union all the buffered circles to merge overlapping ones
330        unary_union(buffered_points.iter())
331    }
332}

334impl<F: BoolOpsNum + 'static> Buffer for LineString<F> {
335    type Scalar = F;
336    fn buffer_with_style(&self, style: BufferStyle<Self::Scalar>) -> MultiPolygon<Self::Scalar> {
337        if style.distance <= F::zero() {
338            return MultiPolygon::empty();
339        }
340        match self.dimensions() {
341            Dimensions::Empty => MultiPolygon::empty(),
342            // Treat degenerate line like a point, similar to JTS, whereas i_overlay returns an
343            // empty result
344            Dimensions::ZeroDimensional => Point(self.0[0]).buffer_with_style(style),
345            Dimensions::OneDimensional => {
346                let subject = line_string_to_shape_path(self);
347                let shapes = subject.stroke(style.stroke_style(), false);
348                multi_polygon_from_shapes(shapes)
349            }
350            Dimensions::TwoDimensional => unreachable!("linestring can't be 2 dimensional"),
351        }
352    }
353}

355impl<F: BoolOpsNum + 'static> Buffer for MultiLineString<F> {
356    type Scalar = F;
357    fn buffer_with_style(&self, style: BufferStyle<Self::Scalar>) -> MultiPolygon<Self::Scalar> {
358        if style.distance <= F::zero() {
359            return MultiPolygon::empty();
360        }
361
362        let mut degenerate_points = Vec::new();
363        let mut subject = Vec::with_capacity(self.0.len());
364        for line_string in self {
365            match line_string.dimensions() {
366                Dimensions::Empty => {}
367                // Treat degenerate line like a point, similar to JTS, whereas i_overlay returns an
368                // empty result
369                Dimensions::ZeroDimensional => degenerate_points.push(Point(line_string[0])),
370                Dimensions::OneDimensional => subject.push(line_string_to_shape_path(line_string)),
371                Dimensions::TwoDimensional => unreachable!("linestring can't be 2 dimensional"),
372            }
373        }
374
375        let stroked_lines = if subject.is_empty() {
376            MultiPolygon::empty()
377        } else {
378            let shapes = subject.stroke(style.stroke_style(), false);
379            multi_polygon_from_shapes(shapes)
380        };
381
382        if degenerate_points.is_empty() {
383            stroked_lines
384        } else {
385            MultiPoint(degenerate_points)
386                .buffer_with_style(style)
387                .union(&stroked_lines)
388        }
389    }
390}

414    fn buffer_with_style(&self, style: BufferStyle<Self::Scalar>) -> MultiPolygon<Self::Scalar> {
415        if style.distance <= F::zero() {
416            return MultiPolygon::empty();
417        }
418        match self.dimensions() {
419            Dimensions::TwoDimensional => unreachable!("line can't be 2-D"),
420            Dimensions::Empty => unreachable!("line can't be empty"),

423            Dimensions::ZeroDimensional => Point(self.start).buffer_with_style(style),
424            Dimensions::OneDimensional => {
425                let subject: Vec<_> = [self.start, self.end]
426                    .iter()
427                    .map(|c| BoolOpsCoord(*c))
428                    .collect();
429                let shapes = subject.stroke(style.stroke_style(), false);
430                multi_polygon_from_shapes(shapes)
431            }
432        }
433    }
