20    fn extract_offset_shapes(
21        &self,
22        filter: Vec<bool>,
23        main_direction: ContourDirection,
24        min_area: u64,
25    ) -> IntShapes {
26        let clockwise = main_direction == ContourDirection::Clockwise;
27        let mut buffer = filter;
28        let visited = buffer.as_mut_slice();
29        let mut shapes = Vec::new();
30        let mut holes = Vec::new();
31        let mut anchors = Vec::new();
32
33        let mut link_index = 0;
34        let mut is_all_anchors_sorted = true;
35        while link_index < visited.len() {
36            if visited.is_visited(link_index) {
37                link_index += 1;
38                continue;
39            }
40
41            let left_top_link = GraphUtil::find_left_top_link(self.links, self.nodes, link_index, visited);
42            let link = self.link(left_top_link);
43            let is_hole = link.fill & SUBJ_TOP == SUBJ_TOP;
44            let mut bold = link.is_bold();
45            let direction = is_hole == clockwise;
46
47            let start_data = StartPathData::new(direction, link, left_top_link);
48
49            let mut contour = self.get_fill_contour(&start_data, direction, &mut bold, visited);
50            if !bold {
51                link_index += 1;
52                continue;
53            }
54
55            let (is_valid, is_modified) = contour.validate(min_area, true);
56
57            if !is_valid {
58                link_index += 1;
59                continue;
60            }
61
62            if is_hole {
63                let mut v_segment = if clockwise {
64                    VSegment {
65                        a: contour[1],
66                        b: contour[2],
67                    }
68                } else {
69                    VSegment {
70                        a: contour[0],
71                        b: contour[contour.len() - 1],
72                    }
73                };
74                if is_modified {
75                    let most_left = contour.left_bottom_segment();
76                    if most_left != v_segment {
77                        v_segment = most_left;
78                        is_all_anchors_sorted = false;
79                    }
80                };
81
82                debug_assert_eq!(v_segment, contour.left_bottom_segment());
83                let id = ContourIndex::new_hole(holes.len());
84                anchors.push(IdSegment::with_segment(id, v_segment));
85                holes.push(contour);
86            } else {
87                shapes.push(vec![contour]);
88            }
89        }
90
91        if !is_all_anchors_sorted {
92            anchors.sort_by(|s0, s1| s0.v_segment.a.cmp(&s1.v_segment.a));
93        }
94
95        shapes.join_sorted_holes(holes, anchors, clockwise);
96
97        shapes
98    }

100    fn get_fill_contour(
101        &self,
102        start_data: &StartPathData,
103        clockwise: bool,
104        bold: &mut bool,
105        visited: &mut [bool],
106    ) -> IntContour {
107        let mut link_id = start_data.link_id;
108        let mut node_id = start_data.node_id;
109        let last_node_id = start_data.last_node_id;
110
111        visited.visit(link_id);
112
113        let mut contour = IntContour::new();
114        contour.push(start_data.begin);
115
116        // Find a closed tour
117        while node_id != last_node_id {
118            link_id = GraphUtil::next_link(
119                self.links,
120                self.nodes,
121                link_id,
122                node_id,
123                clockwise,
124                visited,
125            );
126
127            let link = unsafe { self.links.get_unchecked(link_id) };
128            *bold = *bold || link.is_bold();
129
130            node_id = contour.push_node_and_get_other(link, node_id);
131
132            visited.visit(link_id);
133        }
134
135        contour
136    }
