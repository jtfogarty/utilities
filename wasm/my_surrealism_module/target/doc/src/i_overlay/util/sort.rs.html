8pub(crate) trait SmartBinSort<U> {
9    fn smart_bin_sort_by<F>(&mut self, solver: &Solver, compare: F)
10    where
11        F: Fn(&Self::Item, &Self::Item) -> Ordering + Sync;
12
13    type Item: Send;
14}
15
16impl<T, U> SmartBinSort<U> for [T]
17where
18    T: BinKey<U> + Clone + Send,
19    U: Copy + Ord + BinLayoutOp,
20{
21    fn smart_bin_sort_by<F>(&mut self, _solver: &Solver, compare: F)
22    where
23        F: Fn(&T, &T) -> Ordering + Sync,
24    {
25        #[cfg(feature = "allow_multithreading")]
26        {
27            if let Some(multithreading) = _solver.multithreading {
28                if self.len() > multithreading.par_sort_min_size {
29                    self.par_sort_unstable_by(compare);
30                    return;
31                }
32            }
33        }
34
35        // Fallback to standard sort if multithreading is not enabled
36        self.sort_with_bins(compare)
37    }
38
39    type Item = T;
40}

47    #[derive(Debug, Clone, Copy, PartialEq, Eq)]
48    struct Point {
49        x: i32,
50        y: i32,
51    }
52
53    impl PartialOrd for Point {
54        #[inline(always)]
55        fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
56            Some(self.cmp(other))
57        }
58    }
59
60    impl Ord for Point {
61        #[inline(always)]
62        fn cmp(&self, other: &Self) -> Ordering {
63            let x = self.x == other.x;
64            if x && self.y == other.y {
65                Ordering::Equal
66            } else if self.x < other.x || x && self.y < other.y {
67                Ordering::Less
68            } else {
69                Ordering::Greater
70            }
71        }
72    }
73
74    impl BinKey<i32> for Point {
75        #[inline(always)]
76        fn bin_key(&self) -> i32 {
77            self.x
78        }
79
80        #[inline(always)]
81        fn bin_index(&self, layout: &BinLayout<i32>) -> usize {
82            layout.index(self.x.into())
83        }
84    }
