13/// The `FloatStringOverlay` struct is a builder for overlaying geometric shapes by converting
14/// floating-point geometry to integer space. It provides methods for adding paths and shapes,
15/// as well as for converting the overlay into a `FloatStringGraph`.
16pub struct FloatStringOverlay<P: FloatPointCompatible<T>, T: FloatNumber> {
17    pub(super) overlay: StringOverlay,
18    pub(super) adapter: FloatPointAdapter<P, T>,
19}
20
21impl<P: FloatPointCompatible<T>, T: FloatNumber> FloatStringOverlay<P, T> {
22    /// Constructs a new `FloatStringOverlay`, a builder for overlaying geometric shapes
23    /// by converting float-based geometry to integer space, using a pre-configured adapter.
24    ///
25    /// - `adapter`: A `FloatPointAdapter` instance responsible for coordinate conversion between
26    ///   float and integer values, ensuring accuracy during geometric transformations.
27    /// - `capacity`: Initial capacity for storing segments, ideally matching the total number of
28    ///   segments for efficient memory allocation.
29    #[inline]
30    pub fn with_adapter(adapter: FloatPointAdapter<P, T>, capacity: usize) -> Self {
31        Self { overlay: StringOverlay::new(capacity), adapter }
32    }
33
34    /// Creates a new `FloatOverlay` instance and initializes it with subject and clip shapes.
35    /// - `shape`: A `ShapeResource` define the shape.
36    ///   `ShapeResource` can be one of the following:
37    ///     - `Contour`: A contour representing a closed path. This path is interpreted as closed, so it doesn’t require the start and endpoint to be the same for processing.
38    ///     - `Contours`: A collection of contours, each representing a closed path.
39    ///     - `Shapes`: A collection of shapes, where each shape may consist of multiple contours.
40    /// - `string`: A `ShapeResource` define the string paths.
41    ///   `ShapeResource` can be one of the following:
42    ///     - `Path`: A path representing a string line.
43    ///     - `Paths`: A collection of paths, each representing a string line.
44    ///     - `Vec<Paths>`: A collection of grouped paths, where each group may consist of multiple paths.
45    pub fn with_shape_and_string<R0, R1>(shape: &R0, string: &R1) -> Self
46    where
47        R0: ShapeResource<P, T>,
48        R1: ShapeResource<P, T>,
49        P: FloatPointCompatible<T>,
50        T: FloatNumber,
51    {
52        let iter = shape.iter_paths().chain(string.iter_paths()).flatten();
53        let adapter = FloatPointAdapter::with_iter(iter);
54        let shape_capacity = shape.iter_paths().fold(0, |s, c| s + c.len());
55        let string_capacity = string.iter_paths().fold(0, |s, c| s + c.len());
56
57        Self::with_adapter(adapter, shape_capacity + string_capacity)
58            .unsafe_add_shapes(shape)
59            .unsafe_add_string_lines(string)
60    }
61
62    /// Adds a shapes to the overlay.
63    /// - `source`: A `ShapeResource` that define shape.
64    ///   `ShapeResource` can be one of the following:
65    ///     - `Contour`: A contour representing a closed path. This path is interpreted as closed, so it doesn’t require the start and endpoint to be the same for processing.
66    ///     - `Contours`: A collection of contours, each representing a closed path.
67    ///     - `Shapes`: A collection of shapes, where each shape may consist of multiple contours.
68    /// - `shape_type`: Specifies the role of the added paths in the overlay operation, either as `Subject` or `Clip`.
69    #[inline]
70    pub fn unsafe_add_shapes<S: ShapeResource<P, T>>(mut self, source: &S) -> Self {
71        for contour in source.iter_paths() {
72            self = self.unsafe_add_shape_contour(contour);
73        }
74        self
75    }
76
77    /// Adds a string line paths to the overlay.
78    /// - `resource`: A `ShapeResource` that define shape.
79    ///   `ShapeResource` can be one of the following:
80    ///     - `Path`: A path representing a string line.
81    ///     - `Paths`: A collection of paths, each representing a string line.
82    ///     - `Vec<Paths>`: A collection of grouped paths, where each group may consist of multiple paths.
83    #[inline]
84    pub fn unsafe_add_string_lines<S: ShapeResource<P, T>>(mut self, resource: &S) -> Self {
85        for path in resource.iter_paths() {
86            self = self.unsafe_add_string_line(path);
87        }
88        self
89    }
90
91    /// Adds a closed shape path to the overlay.
92    /// - `contour`: An array of points that form a closed path.
93    /// - **Safety**: Marked `unsafe` because it assumes the path is fully contained within the bounding box.
94    #[inline]
95    pub fn unsafe_add_shape_contour(mut self, contour: &[P]) -> Self {
96        self.overlay.add_shape_contour_iter(contour.iter().map(|p| self.adapter.float_to_int(p)));
97        self
98    }
99
100    /// Adds an open string line path to the overlay.
101    /// - `path`: A path representing a string line.
102    /// - **Safety**: Marked `unsafe` because it assumes each path is fully contained within the bounding box.
103    #[inline]
104    pub fn unsafe_add_string_line(mut self, path: &[P]) -> Self {
105        for w in path.windows(2) {
106            let a = self.adapter.float_to_int(&w[0]);
107            let b = self.adapter.float_to_int(&w[1]);
108            self.overlay.add_string_line([a, b]);
109        }
110
111        self
112    }

114    /// Converts the current overlay into an `FloatStringGraph` based on the specified build rule.
115    /// The resulting graph is the foundation for performing boolean operations, and it's optimized for such operations based on the provided build rule.
116    /// - `fill_rule`: Fill rule to determine filled areas (non-zero, even-odd, positive, negative).
117    /// - Returns: A `FloatStringGraph` containing the graph representation of the overlay's geometry.
118    #[inline]
119    pub fn build_graph_view(&mut self, fill_rule: FillRule) -> Option<FloatStringGraph<P, T>> {
120        self.build_graph_view_with_solver(fill_rule, Default::default())
121    }
122
123    /// Converts the current overlay into an `FloatStringGraph` based on the specified build rule and solver.
124    /// This method allows for finer control over the boolean operation process by passing a custom solver.
125    /// - `fill_rule`: Fill rule to determine filled areas (non-zero, even-odd, positive, negative).
126    /// - `solver`: A custom solver for optimizing or modifying the graph creation process.
127    /// - Returns: A `FloatStringGraph` containing the graph representation of the overlay's geometry.
128    #[inline]
129    pub fn build_graph_view_with_solver(&mut self, fill_rule: FillRule, solver: Solver) -> Option<FloatStringGraph<P, T>> {
130        let graph = self.overlay.build_graph_view_with_solver(fill_rule, solver)?;
131        Some(FloatStringGraph { graph, adapter: self.adapter.clone() })
132    }
133
134    /// Executes a single Boolean operation on the current geometry using the specified build and clip rules.
135    ///
136    /// ### Parameters:
137    /// - `fill_rule`: Fill rule to determine filled areas (non-zero, even-odd, positive, negative).
138    /// - `clip_rule`: Clip rule to determine how the boundary and inversion settings affect the result.
139    /// - `solver`: Type of solver to use.
140    /// - Returns: A `Paths<P>` collection of string lines that meet the clipping conditions.
141    #[inline]
142    pub fn clip_string_lines_with_solver(self, fill_rule: FillRule, clip_rule: ClipRule, solver: Solver) -> Paths<P> {
143        let paths = self.overlay.clip_string_lines_with_solver(fill_rule, clip_rule, solver);
144        paths.to_float(&self.adapter)
145    }
