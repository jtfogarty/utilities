4/// Represents the selection strategy or algorithm for processing geometric data, aimed at optimizing performance under various conditions.
5///
6/// This enum allows for the explicit selection of a computational approach to geometric data processing. The choice of solver is crucial as it directly affects the efficiency of operations, especially in relation to the complexity and size of the dataset involved.
7///
8/// Cases:
9/// - `List`: A linear list-based approach for organizing and processing geometric data. Typically, performs better for smaller datasets, approximately with fewer than 10,000 edges, due to its straightforward processing model. For small to moderate datasets, this method can offer a balance of simplicity and speed.
10/// - `Tree`: Implements a tree-based data structure (e.g., a binary search tree or a spatial partitioning tree) to manage geometric data. This method is generally more efficient for larger datasets or scenarios requiring complex spatial queries, as it can significantly reduce the number of comparisons needed for operations. However, its performance advantage becomes more apparent as the dataset size exceeds a certain threshold (roughly estimated at 10,000 edges).
11/// - `Auto`: Delegates the choice of solver to the system, which determines the most suitable approach based on the size and complexity of the dataset. This option is designed to dynamically select between `list` and `tree` strategies, aiming to optimize performance without requiring a priori knowledge of the data's characteristics. It's the recommended choice for users looking for a balance between performance and ease of use, as it adapts to the specific requirements of each operation.
12#[derive(Debug, Clone, Copy, PartialEq)]
13pub enum Strategy {
14    List,
15    Tree,
16    Frag,
17    Auto,
18}

20/// Represents the precision level used by the solver to determine
21/// the tolerance for snapping to the nearest edge ends.
22///
23/// The precision determines a radius calculated as `2^value`,
24/// where `value` starts at `start` and increases in increments
25/// defined by `progression` in each iteration.
26///
27/// - `start`: The initial exponent value.
28/// - `progression`: The step size for incrementing the exponent
29///   in each iteration.
30#[derive(Debug, Clone, Copy, PartialEq)]
31pub struct Precision {
32    /// The initial exponent value for the radius calculation.
33    pub start: usize,
34    /// The amount by which the exponent increases in each iteration.
35    pub progression: usize,
36}
37
38impl Precision {
39    /// Absolute precision with no progression.
40    /// (Radius remains at `2^0 = 1`)
41    pub const ABSOLUTE: Precision = Self {
42        start: 0,
43        progression: 0,
44    };
45
46    /// High precision, starting at `2^0 = 1` and doubling every loop.
47    pub const HIGH: Precision = Self {
48        start: 0,
49        progression: 1,
50    };
51
52    /// Medium-high precision, starting at `2^1 = 2` and doubling every loop.
53    pub const MEDIUM_HIGH: Precision = Self {
54        start: 1,
55        progression: 1,
56    };
57
58    /// Medium precision, starting at `2^0 = 1` and quadrupling every loop.
59    pub const MEDIUM: Precision = Self {
60        start: 0,
61        progression: 2,
62    };
63
64    /// Medium-low precision, starting at `2^2 = 4` and quadrupling every loop.
65    pub const MEDIUM_LOW: Precision = Self {
66        start: 2,
67        progression: 2,
68    };
69
70    /// Low precision, starting at `2^2 = 4` and increasing by a factor of 8 every loop.
71    pub const LOW: Precision = Self {
72        start: 2,
73        progression: 3,
74    };
75}

94#[derive(Clone, Copy)]
95pub struct Solver {
96    pub strategy: Strategy,
97    pub precision: Precision,
98    pub multithreading: Option<MultithreadOptions>,
99}
100
101impl Default for Solver {
102    fn default() -> Self {
103        Solver::AUTO
104    }
105}
106
107impl Solver {
108    pub const LIST: Self = Self {
109        strategy: List,
110        precision: Precision::HIGH,
111        multithreading: Some(MultithreadOptions {
112            par_sort_min_size: MultithreadOptions::DEFAULT_PAR_SORT_MIN_SIZE,
113        }),
114    };
115
116    pub const TREE: Self = Self {
117        strategy: Tree,
118        precision: Precision::HIGH,
119        multithreading: Some(MultithreadOptions {
120            par_sort_min_size: MultithreadOptions::DEFAULT_PAR_SORT_MIN_SIZE,
121        }),
122    };
123
124    pub const FRAG: Self = Self {
125        strategy: Frag,
126        precision: Precision::HIGH,
127        multithreading: Some(MultithreadOptions {
128            par_sort_min_size: MultithreadOptions::DEFAULT_PAR_SORT_MIN_SIZE,
129        }),
130    };
131
132    pub const AUTO: Self = Self {
133        strategy: Auto,
134        precision: Precision::HIGH,
135        multithreading: Some(MultithreadOptions {
136            par_sort_min_size: MultithreadOptions::DEFAULT_PAR_SORT_MIN_SIZE,
137        }),
138    };
139
140    const MAX_SPLIT_LIST_COUNT: usize = 4_000;
141    const MIN_FRAGMENT_COUNT: usize = 16_000;
142    const MAX_FILL_LIST_COUNT: usize = 8_000;
143
144    pub fn with_precision(precision: Precision) -> Self {
145        Self {
146            strategy: Auto,
147            precision,
148            multithreading: Some(MultithreadOptions {
149                par_sort_min_size: MultithreadOptions::DEFAULT_PAR_SORT_MIN_SIZE,
150            }),
151        }
152    }
153
154    pub fn with_strategy_and_precision(strategy: Strategy, precision: Precision) -> Self {
155        Self {
156            strategy,
157            precision,
158            multithreading: Some(MultithreadOptions {
159                par_sort_min_size: MultithreadOptions::DEFAULT_PAR_SORT_MIN_SIZE,
160            }),
161        }
162    }
163
164    pub(crate) fn is_list_split<C: Send>(&self, segments: &[Segment<C>]) -> bool {
165        match self.strategy {
166            List => true,
167            Tree | Frag => false,
168            Auto => segments.len() < Self::MAX_SPLIT_LIST_COUNT,
169        }
170    }
171
172    pub(crate) fn is_fragmentation_required<C: Send>(&self, segments: &[Segment<C>]) -> bool {
173        segments.len() > Self::MIN_FRAGMENT_COUNT || self.strategy == Frag
174    }
175
176    pub(crate) fn is_list_fill<C: Send>(&self, segments: &[Segment<C>]) -> bool {
177        match self.strategy {
178            List => true,
179            Tree | Frag => false,
180            Auto => segments.len() < Self::MAX_FILL_LIST_COUNT,
181        }
182    }
