26impl OverlayGraph<'_> {
27    /// Extracts shapes from the overlay graph based on the specified overlay rule. This method is used to retrieve the final geometric shapes after boolean operations have been applied. It's suitable for most use cases where the minimum area of shapes is not a concern.
28    /// - `overlay_rule`: The boolean operation rule to apply when extracting shapes from the graph, such as union or intersection.
29    /// - `buffer`: Reusable buffer, optimisation purpose only.
30    /// - Returns: A vector of `IntShape`, representing the geometric result of the applied overlay rule.
31    /// # Shape Representation
32    /// The output is a `IntShapes`, where:
33    /// - The outer `Vec<IntShape>` represents a set of shapes.
34    /// - Each shape `Vec<IntContour>` represents a collection of contours, where the first contour is the outer boundary, and all subsequent contours are holes in this boundary.
35    /// - Each path `Vec<IntPoint>` is a sequence of points, forming a closed path.
36    ///
37    /// Note: Outer boundary paths have a counterclockwise order, and holes have a clockwise order.
38    #[inline]
39    pub fn extract_shapes(&self, overlay_rule: OverlayRule, buffer: &mut BooleanExtractionBuffer) -> IntShapes {
40        self.links.filter_by_overlay_into(overlay_rule, &mut buffer.visited);
41        self.extract(overlay_rule, buffer)
42    }
43
44    /// Extracts the flat contours from the overlay graph based on the specified overlay rule.
45    ///
46    /// This method performs a Boolean operation (e.g., union or intersection) and stores the result
47    /// directly into a flat buffer of contours, without nesting them into shapes (i.e., no hole-joining or grouping).
48    ///
49    /// It is optimized for performance and suitable when raw contour data is sufficient,
50    /// such as during intermediate processing, visualization, or tesselation.
51    ///
52    /// - `overlay_rule`: The boolean operation rule to apply (e.g., union, intersection, xor).
53    /// - `buffer`: Reusable working buffer to avoid reallocations.
54    /// - `output`: A flat buffer to which the resulting valid contours will be written.
55    #[inline]
56    pub fn extract_contours_into(&self, overlay_rule: OverlayRule, buffer: &mut BooleanExtractionBuffer, output: &mut FlatContoursBuffer) {
57        self.links.filter_by_overlay_into(overlay_rule, &mut buffer.visited);
58        self.extract_contours(overlay_rule, buffer, output);
59    }

61    fn extract(
62        &self,
63        overlay_rule: OverlayRule,
64        buffer: &mut BooleanExtractionBuffer,
65    ) -> IntShapes {
66        let clockwise = self.options.output_direction == ContourDirection::Clockwise;
67
68        let mut shapes = Vec::new();
69        let mut holes = Vec::new();
70        let mut anchors = Vec::new();
71
72        buffer.points.reserve_capacity(buffer.visited.len());
73
74        let mut link_index = 0;
75        let mut anchors_already_sorted = true;
76        while link_index < buffer.visited.len() {
77            if buffer.visited.is_visited(link_index) {
78                link_index += 1;
79                continue;
80            }
81
82            let left_top_link = GraphUtil::find_left_top_link(self.links, self.nodes, link_index, &buffer.visited);
83            let link = unsafe { self.links.get_unchecked(left_top_link) };
84            let is_hole = overlay_rule.is_fill_top(link.fill);
85
86            let direction = is_hole == clockwise;
87            let start_data = StartPathData::new(direction, link, left_top_link);
88
89            self.find_contour(&start_data, direction, buffer);
90            let (is_valid, is_modified) =
91                buffer.points.validate(self.options.min_output_area, self.options.preserve_output_collinear);
92
93            if !is_valid {
94                link_index += 1;
95                continue;
96            }
97
98            let contour = buffer.points.as_slice().to_vec();
99
100            if is_hole {
101                let mut v_segment = if clockwise {
102                    VSegment {
103                        a: contour[1],
104                        b: contour[2],
105                    }
106                } else {
107                    VSegment {
108                        a: contour[0],
109                        b: contour[contour.len() - 1],
110                    }
111                };
112                if is_modified {
113                    let most_left = contour.left_bottom_segment();
114                    if most_left != v_segment {
115                        v_segment = most_left;
116                        anchors_already_sorted = false;
117                    }
118                };
119
120                debug_assert_eq!(v_segment, contour.left_bottom_segment());
121                let id_data = ContourIndex::new_hole(holes.len());
122                anchors.push(IdSegment::with_segment(id_data, v_segment));
123                holes.push(contour);
124            } else {
125                shapes.push(vec![contour]);
126            }
127        }
128
129        if !anchors_already_sorted {
130            anchors.sort_by(|s0, s1| s0.v_segment.a.cmp(&s1.v_segment.a));
131        }
132
133        shapes.join_sorted_holes(holes, anchors, clockwise);
134
135        shapes
136    }

138    fn find_contour(
139        &self,
140        start_data: &StartPathData,
141        clockwise: bool,
142        buffer: &mut BooleanExtractionBuffer,
143    ) {
144        let mut link_id = start_data.link_id;
145        let mut node_id = start_data.node_id;
146        let last_node_id = start_data.last_node_id;
147
148        buffer.visited.visit(link_id);
149        buffer.points.clear();
150        buffer.points.push(start_data.begin);
151
152        // Find a closed tour
153        while node_id != last_node_id {
154            link_id = GraphUtil::next_link(
155                self.links,
156                self.nodes,
157                link_id,
158                node_id,
159                clockwise,
160                &buffer.visited,
161            );
162
163            let link = unsafe { self.links.get_unchecked(link_id) };
164            node_id = buffer.points.push_node_and_get_other(link, node_id);
165
166            buffer.visited.visit(link_id);
167        }
168    }

170    fn extract_contours(
171        &self,
172        overlay_rule: OverlayRule,
173        buffer: &mut BooleanExtractionBuffer,
174        output: &mut FlatContoursBuffer
175    ) {
176        let clockwise = self.options.output_direction == ContourDirection::Clockwise;
177        let len = buffer.visited.len();
178        buffer.points.reserve_capacity(len);
179        output.clear_and_reserve(len, 4);
180
181        let mut link_index = 0;
182        while link_index < len {
183            if buffer.visited.is_visited(link_index) {
184                link_index += 1;
185                continue;
186            }
187
188            let left_top_link = GraphUtil::find_left_top_link(self.links, self.nodes, link_index, &buffer.visited);
189            let link = unsafe { self.links.get_unchecked(left_top_link) };
190            let is_hole = overlay_rule.is_fill_top(link.fill);
191
192            let direction = is_hole == clockwise;
193            let start_data = StartPathData::new(direction, link, left_top_link);
194
195            self.find_contour(&start_data, direction, buffer);
196            let (is_valid, _) =
197                buffer.points.validate(self.options.min_output_area, self.options.preserve_output_collinear);
198
199            if !is_valid {
200                link_index += 1;
201                continue;
202            }
203
204            output.add_contour(buffer.points.as_slice());
205        }
206    }

221                begin: link.b.point,
222                node_id: link.a.id,
223                link_id,
224                last_node_id: link.b.id,

226        } else {
227            Self {
228                begin: link.a.point,
229                node_id: link.b.id,

231                last_node_id: link.a.id,
232            }
233        }

237pub(crate) trait GraphContour {
238    fn validate(&mut self, min_output_area: u64, preserve_output_collinear: bool) -> (bool, bool);
239    fn push_node_and_get_other(&mut self, link: &OverlayLink, node_id: usize) -> usize;
240}
241
242impl GraphContour for IntContour {
243    #[inline]
244    fn validate(&mut self, min_output_area: u64, preserve_output_collinear: bool) -> (bool, bool) {
245        let is_modified = if !preserve_output_collinear {
246            self.simplify_contour()
247        } else {
248            false
249        };
250
251        if self.len() < 3 {
252            return (false, is_modified);
253        }
254
255        if min_output_area == 0 {
256            return (true, is_modified);
257        }
258        let area = self.unsafe_area();
259        let abs_area = area.unsigned_abs() >> 1;
260        let is_valid = abs_area >= min_output_area;
261
262        (is_valid, is_modified)
263    }
264
265    #[inline]
266    fn push_node_and_get_other(&mut self, link: &OverlayLink, node_id: usize) -> usize {
267        if link.a.id == node_id {
268            self.push(link.a.point);
269            link.b.id
270        } else {
271            self.push(link.b.point);
272            link.a.id
273        }
274    }
275}

279    fn is_not_visited(&self, index: usize) -> bool;

281}

283impl Visit for [bool] {
284    #[inline(always)]
285    fn is_visited(&self, index: usize) -> bool {
286        unsafe { *self.get_unchecked(index) }
287    }
288
289    #[inline(always)]
290    fn is_not_visited(&self, index: usize) -> bool {
291        !unsafe { *self.get_unchecked(index) }
292    }
293
294    #[inline(always)]
295    fn visit(&mut self, index: usize) {
296        unsafe { *self.get_unchecked_mut(index) = true }
297    }

304    #[inline]
305    pub(crate) fn find_left_top_link(links: &[OverlayLink], nodes: &[OverlayNode], link_index: usize, visited: &[bool]) -> usize {
306        let top = unsafe { links.get_unchecked(link_index) };
307        debug_assert!(top.is_direct());
308        let node = unsafe { nodes.get_unchecked(top.a.id) };
309
310        match node {
311            OverlayNode::Bridge(bridge) => Self::find_left_top_link_on_bridge(links, bridge),
312            OverlayNode::Cross(indices) => {
313                Self::find_left_top_link_on_indices(links, top, link_index, indices, visited)
314            }
315        }
316    }
317
318    #[inline(always)]
319    fn find_left_top_link_on_indices(
320        links: &[OverlayLink],
321        link: &OverlayLink,
322        link_index: usize,
323        indices: &[usize],
324        visited: &[bool],
325    ) -> usize {
326        let mut top_index = link_index;
327        let mut top = link;
328
329        // find most top link
330
331        for &i in indices.iter() {
332            if i == link_index {
333                continue;
334            }
335            let link = unsafe { links.get_unchecked(i) };
336            if !link.is_direct()
337                || Triangle::is_clockwise_point(top.a.point, top.b.point, link.b.point)
338            {
339                continue;
340            }
341
342            if visited.is_visited(i) {
343                continue;
344            }
345
346            top_index = i;
347            top = link;
348        }
349
350        top_index
351    }
352
353    #[inline(always)]
354    fn find_left_top_link_on_bridge(links: &[OverlayLink], bridge: &[usize; 2]) -> usize {
355        let l0 = unsafe { links.get_unchecked(bridge[0]) };
356        let l1 = unsafe { links.get_unchecked(bridge[1]) };
357        if Triangle::is_clockwise_point(l0.a.point, l0.b.point, l1.b.point) {
358            bridge[0]
359        } else {
360            bridge[1]
361        }
362    }
363
364    #[inline(always)]
365    pub(crate) fn next_link(
366        links: &[OverlayLink],
367        nodes: &[OverlayNode],
368        link_id: usize,
369        node_id: usize,
370        clockwise: bool,
371        visited: &[bool],
372    ) -> usize {
373        let node = unsafe { nodes.get_unchecked(node_id) };
374        match node {
375            OverlayNode::Bridge(bridge) => {
376                if bridge[0] == link_id {
377                    bridge[1]
378                } else {
379                    bridge[0]
380                }
381            }
382            OverlayNode::Cross(indices) => {
383                GraphUtil::find_nearest_link_to(links, link_id, node_id, clockwise, indices, visited)
384            }
385        }
386    }
387
388    #[inline]
389    fn find_nearest_link_to(
390        links: &[OverlayLink],
391        target_index: usize,
392        node_id: usize,
393        clockwise: bool,
394        indices: &[usize],
395        visited: &[bool],
396    ) -> usize {
397        let mut is_first = true;
398        let mut first_index = 0;
399        let mut second_index = usize::MAX;
400        let mut pos = 0;
401        for (i, &link_index) in indices.iter().enumerate() {
402            if visited.is_not_visited(link_index) {
403                if is_first {
404                    first_index = link_index;
405                    is_first = false;
406                } else {
407                    second_index = link_index;
408                    pos = i;
409                    break;
410                }
411            }
412        }
413
414        if second_index == usize::MAX {
415            return first_index;
416        }
417
418        let target = unsafe { links.get_unchecked(target_index) };
419        let (c, a) = if target.a.id == node_id {
420            (target.a.point, target.b.point)
421        } else {
422            (target.b.point, target.a.point)
423        };
424
425        // more the one vectors
426        let b = unsafe { links.get_unchecked(first_index) }.other(node_id).point;
427        let mut vector_solver = NearestVector::new(c, a, b, first_index, clockwise);
428
429        // add second vector
430        vector_solver.add(unsafe { links.get_unchecked(second_index) }.other(node_id).point, second_index);
431
432        // check the rest vectors
433        for &link_index in indices.iter().skip(pos + 1) {
434            if visited.is_not_visited(link_index) {
435                let p = unsafe { links.get_unchecked(link_index) }.other(node_id).point;
436                vector_solver.add(p, link_index);
437            }
438        }
439
440        vector_solver.best_id
441    }
