23/// Configuration options for polygon Boolean operations using [`Overlay`].
24///
25/// These options control precision, simplification, and contour filtering
26/// during the Boolean operation process. You can use this to adjust output
27/// direction, eliminate small artifacts, or retain collinear points.
28#[derive(Debug, Clone, Copy)]
29pub struct IntOverlayOptions {
30    /// Preserve collinear points in the input before Boolean operations.
31    pub preserve_input_collinear: bool,
32
33    /// Desired direction for output contours (default outer: CCW / hole: CW).
34    pub output_direction: ContourDirection,
35
36    /// Preserve collinear points in the output after Boolean operations.
37    pub preserve_output_collinear: bool,
38
39    /// Minimum area threshold to include a contour in the result.
40    pub min_output_area: u64,
41}
42
43/// Specifies the type of shape being processed, influencing how the shape participates in Boolean operations.
44/// Note: All operations except for `Difference` are commutative, meaning the order of `Subject` and `Clip` shapes does not impact the outcome.
45/// - `Subject`: The primary shape(s) for operations. Acts as the base layer in the operation.
46/// - `Clip`: The modifying shape(s) that are applied to the `Subject`. Determines how the `Subject` is altered or intersected.
47#[derive(Debug, Clone, Copy, PartialEq)]
48pub enum ShapeType {
49    Subject,
50    Clip,
51}
52
53/// Represents the winding direction of a contour.
54#[derive(Debug, Clone, Copy, PartialEq)]
55pub enum ContourDirection {
56    CounterClockwise,
57    Clockwise,
58}

251    /// Convert into vector shapes from the added paths or shapes, applying the specified build and overlay rules. This method is particularly useful for development purposes and for creating visualizations in educational demos, where understanding the impact of different rules on the final geometry is crucial.
252    /// - `fill_rule`: The build rule to use for the shapes.
253    /// - `overlay_rule`: The overlay rule to apply.
254    pub fn build_shape_vectors(
255        &mut self,
256        fill_rule: FillRule,
257        overlay_rule: OverlayRule,
258    ) -> Vec<VectorShape> {
259        self.split_solver
260            .split_segments(&mut self.segments, &self.solver);
261        if self.segments.is_empty() {
262            return Vec::new();
263        }
264        self.graph_builder
265            .build_boolean_overlay(
266                fill_rule,
267                overlay_rule,
268                self.options,
269                &self.solver,
270                &self.segments,
271            )
272            .extract_shape_vectors(overlay_rule)
273    }

288    /// Convert into `OverlayGraph` from the added paths or shapes using the specified build rule. This graph is the foundation for executing boolean operations, allowing for the analysis and manipulation of the geometric data. The `OverlayGraph` created by this method represents a preprocessed state of the input shapes, optimized for the application of boolean operations based on the provided build rule.
289    /// - `fill_rule`: Specifies the rule for determining filled areas within the shapes, influencing how the resulting graph represents intersections and unions.
290    #[inline]
291    pub fn build_graph_view(&mut self, fill_rule: FillRule) -> Option<OverlayGraph> {
292        self.split_solver
293            .split_segments(&mut self.segments, &self.solver);
294        if self.segments.is_empty() {
295            return None;
296        }
297        let graph = self.graph_builder.build_boolean_all(
298            fill_rule,
299            self.options,
300            &self.solver,
301            &self.segments,
302        );
303
304        Some(graph)
305    }

307    /// Executes a single Boolean operation on the current geometry using the specified overlay and build rules.
308    /// This method provides a streamlined approach for performing a Boolean operation without generating
309    /// an entire `OverlayGraph`. Ideal for cases where only one Boolean operation is needed, `overlay`
310    /// saves on computational resources by building only the necessary links, optimizing CPU usage by 0-20%
311    /// compared to a full graph-based approach.
312    ///
313    /// ### Parameters:
314    /// - `overlay_rule`: The boolean operation rule to apply, determining how shapes are combined or subtracted.
315    /// - `fill_rule`: Specifies the rule for determining filled areas within the shapes, influencing how the resulting graph represents intersections and unions.
316    /// - Returns: A vector of `IntShape` that meet the specified area criteria, representing the cleaned-up geometric result.
317    /// # Shape Representation
318    /// The output is a `IntShapes`, where:
319    /// - The outer `Vec<IntShape>` represents a set of shapes.
320    /// - Each shape `Vec<IntContour>` represents a collection of contours, where the first contour is the outer boundary, and all subsequent contours are holes in this boundary.
321    /// - Each path `Vec<IntPoint>` is a sequence of points, forming a closed path.
322    ///
323    /// Note: Outer boundary paths have a counterclockwise order, and holes have a clockwise order.
324    /// ### Usage:
325    /// This function is suitable when a single, optimized Boolean operation is required on the provided
326    /// geometry. For example:
327    ///
328    /// ```rust
329    /// use i_float::int::point::IntPoint;
330    /// use i_float::int_pnt;
331    /// use i_overlay::core::fill_rule::FillRule;
332    /// use i_overlay::core::overlay::Overlay;
333    /// use i_overlay::core::overlay_rule::OverlayRule;
334    ///
335    /// let left_rect = [int_pnt!(0, 0), int_pnt!(0, 10), int_pnt!(10, 10), int_pnt!(10, 0)];
336    /// let right_rect = [int_pnt!(10, 0), int_pnt!(10, 10), int_pnt!(20, 10), int_pnt!(20, 0)];
337    /// let mut overlay = Overlay::with_contour(&left_rect, &right_rect);
338    ///
339    /// let result = overlay.overlay(OverlayRule::Union, FillRule::EvenOdd);
340    /// ```
341    ///
342    /// This method is particularly useful in scenarios where the geometry only needs one overlay operation
343    /// without subsequent modifications. By excluding unnecessary graph structures, it optimizes performance,
344    /// particularly for complex or resource-intensive geometries.
345    #[inline]
346    pub fn overlay(&mut self, overlay_rule: OverlayRule, fill_rule: FillRule) -> IntShapes {
347        self.split_solver
348            .split_segments(&mut self.segments, &self.solver);
349        if self.segments.is_empty() {
350            return Vec::new();
351        }
352        let mut buffer = self.boolean_buffer.take().unwrap_or_default();
353        let shapes = self
354            .graph_builder
355            .build_boolean_overlay(
356                fill_rule,
357                overlay_rule,
358                self.options,
359                &self.solver,
360                &self.segments,
361            )
362            .extract_shapes(overlay_rule, &mut buffer);
363        self.boolean_buffer = Some(buffer);
364        shapes
365    }
